import { test, expect } from '@playwright/test';

test('copy note body button copies the body to clipboard', async ({ page }) => {
  await page.goto('http://localhost:5175');
  // Assuming a note is already created and selected
  await page.fill('#note-title', 'Test Note');
  await page.fill('#note-body', 'This is a test note body.');
  await page.fill('#note-tags', 'test,copy');
  await page.click('button:has-text("Create Note")');
  await page.waitForSelector('#cancel-order');
  await page.click('#cancel-order');
  await page.locator('button:has-text("Copy")').click();

  const text = await page.evaluate(() => navigator.clipboard.readText());
  expect(text).toBe('This is a test note body.');
});
