import { test, expect } from '@playwright/test';

test('Copy note body', async ({ page }) => {
  await page.goto('/'); // Navigate to the Home page

  // Assuming there's a note with ID 1 and body "This is a test note."
  await page.locate('button').withText('Note #1').click();
  await page.locate('button').withText('Copy').click();

  // Check if the note body is copied to the clipboard
  const clipboard = await page.evaluate(() => navigator.clipboard.readText());
  expect(clipboard).toBe('This is a test note.');
});
