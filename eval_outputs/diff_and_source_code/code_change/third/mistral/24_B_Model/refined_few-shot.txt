import { test, expect } from '@playwright/test';

test('copy note body to clipboard', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await page.fill('#note-title', 'Test Note');
  await page.fill('#note-body', 'This is a test note body.');
  await page.fill('#note-tags', 'test, tag');
  await page.click('button:has-text("Create Note")');
  await page.click('button:has-text("Note #1")');
  await page.click('text=Copy');
  await expect(page.locator('body')).toHaveText('This is a test note body.');
});
