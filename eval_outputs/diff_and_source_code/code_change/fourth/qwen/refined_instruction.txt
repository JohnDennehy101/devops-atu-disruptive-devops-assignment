import { test, expect } from '@playwright/test';

test('filter notes by tag', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Assuming there are notes with tags 'work' and 'personal'
  const tagFilterInput = page.getByPlaceholder('Filter by tag');
  await tagFilterInput.fill('work');

  // Wait for the displayed notes to update
  await page.waitForTimeout(1000);

  // Check if only notes with the 'work' tag are displayed
  const displayedNotes = page.getByRole('button', { name: /note #\d+/ });
  const displayedNoteTitles = await displayedNotes.allInnerTexts();
  const filteredNotes = displayedNoteTitles.filter(title => title.includes('work'));

  expect(filteredNotes.length).toBeGreaterThan(0);
  expect(displayedNoteTitles.length).toBe(filteredNotes.length);
});
