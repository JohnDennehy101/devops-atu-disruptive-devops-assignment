import { test, expect } from '@playwright/test';

test('filter notes by tag', async ({ page }) => {
  // Step 1: Navigate to the home page
  await page.goto('/');

  // Step 2: Check that the "Your Notes" section displays the correct count of notes
  const initialNoteCount = await page.locator('text=Your Notes').textContent();
  const initialCount = parseInt(initialNoteCount?.match(/\d+/)?.[0] || '0', 10);
  expect(initialCount).toBeGreaterThan(0);

  // Step 3: Enter a tag in the filter input and check that the displayed notes match the filter
  const filterInput = page.locator('input[placeholder="Filter by tag"]');
  await filterInput.fill('exampleTag');
  await page.waitForTimeout(1000); // Wait for filtering to complete

  const filteredNotes = await page.locator('button').all();
  const filteredCount = filteredNotes.length;
  expect(filteredCount).toBeLessThanOrEqual(initialCount);

  // Step 4: Clear the filter input and verify that all notes are displayed again
  await filterInput.clear();
  await page.waitForTimeout(1000); // Wait for filtering to complete

  const allNotes = await page.locator('button').all();
  const allCount = allNotes.length;
  expect(allCount).toBe(initialCount);
});
