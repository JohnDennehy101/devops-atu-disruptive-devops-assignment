import { test, expect } from '@playwright/test';

test('filter notes by tag shows matching notes and displays message when no match', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await page.fill('#note-title', 'Tagged Note');
  await page.fill('#note-body', 'This note has tags');
  await page.fill('#note-tags', 'urgent, work');
  await page.click('button:has-text("Create Note")');
  await expect(page.getByText('Note #')).toBeVisible();
  await page.fill('#note-title', 'Another Note');
  await page.fill('#note-body', 'This note has different tags');
  await page.fill('#note-tags', 'personal');
  await page.click('button:has-text("Create Note")');
  const tagFilterInput = page.getByPlaceholder('Filter by tag');
  await expect(tagFilterInput).toBeVisible();
  await tagFilterInput.fill('urgent');
  await expect(page.getByText('Your Notes (1)')).toBeVisible();
  await tagFilterInput.fill('nonexistenttag');
  await expect(page.getByText('Your Notes (0)')).toBeVisible();
  await expect(page.getByText('No notes match the filter')).toBeVisible();
  await tagFilterInput.fill('');
  await expect(page.getByText('Your Notes (2)')).toBeVisible();
});
