import { test, expect } from '@playwright/test';

test.describe('Tag filter on Your Notes sidebar', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
  });

  test('should display a "Filter by tag" input field', async ({ page }) => {
    const filterInput = page.getByPlaceholder('Filter by tag');
    await expect(filterInput).toBeVisible();
  });

  test('should show all saved notes when tag filter is empty', async ({ page }) => {
    // Create a note with a specific tag
    await page.getByRole('textbox', { name: 'Title' }).fill('Note With Tag');
    await page.getByRole('textbox', { name: 'Body' }).fill('Body content');
    await page.getByRole('textbox', { name: 'Tags' }).fill('javascript');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Create another note with a different tag
    await page.getByRole('textbox', { name: 'Title' }).fill('Another Note');
    await page.getByRole('textbox', { name: 'Body' }).fill('Another body');
    await page.getByRole('textbox', { name: 'Tags' }).fill('python');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // With empty filter, both notes should be visible
    const filterInput = page.getByPlaceholder('Filter by tag');
    await expect(filterInput).toHaveValue('');

    const noteButtons = page.getByRole('button', { name: /Note #/ });
    await expect(noteButtons).toHaveCount(2);
  });

  test('should filter notes by tag when typing in the filter input', async ({ page }) => {
    // Create a note with tag "javascript"
    await page.getByRole('textbox', { name: 'Title' }).fill('JS Note');
    await page.getByRole('textbox', { name: 'Body' }).fill('JavaScript content');
    await page.getByRole('textbox', { name: 'Tags' }).fill('javascript');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Create a note with tag "python"
    await page.getByRole('textbox', { name: 'Title' }).fill('Python Note');
    await page.getByRole('textbox', { name: 'Body' }).fill('Python content');
    await page.getByRole('textbox', { name: 'Tags' }).fill('python');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Filter by "javascript"
    const filterInput = page.getByPlaceholder('Filter by tag');
    await filterInput.fill('javascript');

    // Only the JS note should be visible
    const noteButtons = page.getByRole('button', { name: /Note #/ });
    await expect(noteButtons).toHaveCount(1);
  });

  test('should show "No notes match the filter" when no tags match', async ({ page }) => {
    // Create a note first
    await page.getByRole('textbox', { name: 'Title' }).fill('Test Note');
    await page.getByRole('textbox', { name: 'Body' }).fill('Test body');
    await page.getByRole('textbox', { name: 'Tags' }).fill('existing-tag');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Wait for the note to appear in the sidebar
    await expect(page.getByRole('button', { name: /Note #/ })).toHaveCount(1);

    // Type a tag that doesn't exist
    const filterInput = page.getByPlaceholder('Filter by tag');
    await filterInput.fill('nonexistent-tag');

    // Should show the no match message
    await expect(page.getByText('No notes match the filter')).toBeVisible();
  });

  test('should update the displayed note count based on the filter', async ({ page }) => {
    // Create two notes with different tags
    await page.getByRole('textbox', { name: 'Title' }).fill('First Note');
    await page.getByRole('textbox', { name: 'Body' }).fill('First body');
    await page.getByRole('textbox', { name: 'Tags' }).fill('alpha');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByRole('textbox', { name: 'Title' }).fill('Second Note');
    await page.getByRole('textbox', { name: 'Body' }).fill('Second body');
    await page.getByRole('textbox', { name: 'Tags' }).fill('beta');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Verify initial count shows 2
    await expect(page.getByText('Your Notes (2)')).toBeVisible();

    // Filter by "alpha"
    const filterInput = page.getByPlaceholder('Filter by tag');
    await filterInput.fill('alpha');

    // Count should update to 1
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
  });

  test('should perform case-insensitive tag filtering', async ({ page }) => {
    // Create a note with mixed-case tag
    await page.getByRole('textbox', { name: 'Title' }).fill('Case Note');
    await page.getByRole('textbox', { name: 'Body' }).fill('Case body');
    await page.getByRole('textbox', { name: 'Tags' }).fill('JavaScript');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Filter with lowercase
    const filterInput = page.getByPlaceholder('Filter by tag');
    await filterInput.fill('javascript');

    // The note should still appear
    const noteButtons = page.getByRole('button', { name: /Note #/ });
    await expect(noteButtons).toHaveCount(1);
  });

  test('should show all notes again after clearing the filter', async ({ page }) => {
    // Create two notes
    await page.getByRole('textbox', { name: 'Title' }).fill('Note A');
    await page.getByRole('textbox', { name: 'Body' }).fill('Body A');
    await page.getByRole('textbox', { name: 'Tags' }).fill('tagA');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByRole('textbox', { name: 'Title' }).fill('Note B');
    await page.getByRole('textbox', { name: 'Body' }).fill('Body B');
    await page.getByRole('textbox', { name: 'Tags' }).fill('tagB');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Filter to show only one
    const filterInput = page.getByPlaceholder('Filter by tag');
    await filterInput.fill('tagA');
    await expect(page.getByRole('button', { name: /Note #/ })).toHaveCount(1);

    // Clear the filter
    await filterInput.clear();

    // Both notes should be visible again
    await expect(page.getByRole('button', { name: /Note #/ })).toHaveCount(2);
  });
});
