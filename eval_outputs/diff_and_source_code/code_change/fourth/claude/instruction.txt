import { test, expect } from "@playwright/test";

test.describe("Tag filter on Home page", () => {
  test("should filter saved notes by tag", async ({ page }) => {
    // Seed two notes via API
    const note1 = await (
      await page.request.post("/api/notes", {
        data: { title: "Work Note", body: "Body 1", tags: ["work", "urgent"] },
      })
    ).json();

    const note2 = await (
      await page.request.post("/api/notes", {
        data: { title: "Personal Note", body: "Body 2", tags: ["personal"] },
      })
    ).json();

    // Store both note IDs in localStorage so the sidebar shows them
    await page.addInitScript(
      (ids: number[]) => {
        localStorage.setItem("noteIds", JSON.stringify(ids));
      },
      [note1.id, note2.id]
    );

    await page.goto("/");

    // Both notes should be visible in the sidebar
    await expect(page.getByRole("button", { name: `Note #${note1.id}` })).toBeVisible();
    await expect(page.getByRole("button", { name: `Note #${note2.id}` })).toBeVisible();

    // The heading should show count of 2
    await expect(page.getByRole("heading", { name: /Your Notes \(2\)/ })).toBeVisible();

    // Type a tag filter that matches only note1
    const filterInput = page.getByPlaceholder("Filter by tag");
    await expect(filterInput).toBeVisible();
    await filterInput.fill("work");

    // Only the matching note should remain
    await expect(page.getByRole("button", { name: `Note #${note1.id}` })).toBeVisible();
    await expect(page.getByRole("button", { name: `Note #${note2.id}` })).not.toBeVisible();

    // Heading count should update to 1
    await expect(page.getByRole("heading", { name: /Your Notes \(1\)/ })).toBeVisible();
  });

  test("should show 'No notes match the filter' when filter matches nothing", async ({ page }) => {
    // Seed a note
    const note = await (
      await page.request.post("/api/notes", {
        data: { title: "Some Note", body: "Body", tags: ["alpha"] },
      })
    ).json();

    await page.addInitScript(
      (ids: number[]) => {
        localStorage.setItem("noteIds", JSON.stringify(ids));
      },
      [note.id]
    );

    await page.goto("/");

    // Note is visible initially
    await expect(page.getByRole("button", { name: `Note #${note.id}` })).toBeVisible();

    // Filter by a tag that doesn't exist
    await page.getByPlaceholder("Filter by tag").fill("nonexistent");

    // Should show the empty-state message
    await expect(page.getByText("No notes match the filter")).toBeVisible();

    // The heading should show 0
    await expect(page.getByRole("heading", { name: /Your Notes \(0\)/ })).toBeVisible();
  });

  test("should show 'No notes yet' when there are no saved notes", async ({ page }) => {
    await page.addInitScript(() => {
      localStorage.removeItem("noteIds");
    });

    await page.goto("/");

    await expect(page.getByText("No notes yet")).toBeVisible();
    await expect(page.getByRole("heading", { name: /Your Notes \(0\)/ })).toBeVisible();
  });

  test("should restore full list when filter is cleared", async ({ page }) => {
    const note1 = await (
      await page.request.post("/api/notes", {
        data: { title: "Note A", body: "A", tags: ["frontend"] },
      })
    ).json();

    const note2 = await (
      await page.request.post("/api/notes", {
        data: { title: "Note B", body: "B", tags: ["backend"] },
      })
    ).json();

    await page.addInitScript(
      (ids: number[]) => {
        localStorage.setItem("noteIds", JSON.stringify(ids));
      },
      [note1.id, note2.id]
    );

    await page.goto("/");

    const filterInput = page.getByPlaceholder("Filter by tag");

    // Filter down to one note
    await filterInput.fill("frontend");
    await expect(page.getByRole("button", { name: `Note #${note2.id}` })).not.toBeVisible();

    // Clear the filter
    await filterInput.clear();

    // Both notes should reappear
    await expect(page.getByRole("button", { name: `Note #${note1.id}` })).toBeVisible();
    await expect(page.getByRole("button", { name: `Note #${note2.id}` })).toBeVisible();
    await expect(page.getByRole("heading", { name: /Your Notes \(2\)/ })).toBeVisible();
  });
});
