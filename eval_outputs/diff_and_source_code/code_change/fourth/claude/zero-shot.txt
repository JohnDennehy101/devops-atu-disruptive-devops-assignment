import { test, expect } from "@playwright/test";

test.describe("Tag filter on Home page", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/");
  });

  test("should display a tag filter input", async ({ page }) => {
    const filterInput = page.getByPlaceholder("Filter by tag");
    await expect(filterInput).toBeVisible();
  });

  test("should show 'No notes yet' when no notes exist", async ({ page }) => {
    await page.evaluate(() => localStorage.removeItem("noteIds"));
    await page.reload();
    await expect(page.getByText("No notes yet")).toBeVisible();
  });

  test("should filter notes by tag and show count", async ({ page }) => {
    // Create first note with tag "work"
    await page.getByLabel("Title").fill("Work Note");
    await page.getByLabel("Body").fill("This is a work note");
    await page.getByLabel("Tags").fill("work, important");
    await page.getByRole("button", { name: "Create Note" }).click();

    // Create second note with tag "personal"
    await page.getByLabel("Title").fill("Personal Note");
    await page.getByLabel("Body").fill("This is a personal note");
    await page.getByLabel("Tags").fill("personal, fun");
    await page.getByRole("button", { name: "Create Note" }).click();

    // Both notes should be listed
    await expect(page.getByText("Your Notes (2)")).toBeVisible();

    // Filter by "work"
    const filterInput = page.getByPlaceholder("Filter by tag");
    await filterInput.fill("work");

    // Only the work note should be displayed
    await expect(page.getByText("Your Notes (1)")).toBeVisible();
  });

  test("should show 'No notes match the filter' when filter has no matches", async ({ page }) => {
    // Create a note
    await page.getByLabel("Title").fill("Test Note");
    await page.getByLabel("Body").fill("Test body");
    await page.getByLabel("Tags").fill("existing");
    await page.getByRole("button", { name: "Create Note" }).click();

    await expect(page.getByText("Your Notes (1)")).toBeVisible();

    // Filter by a non-existent tag
    const filterInput = page.getByPlaceholder("Filter by tag");
    await filterInput.fill("nonexistent");

    await expect(page.getByText("Your Notes (0)")).toBeVisible();
    await expect(page.getByText("No notes match the filter")).toBeVisible();
  });

  test("should show all notes when filter is cleared", async ({ page }) => {
    // Create a note
    await page.getByLabel("Title").fill("Tagged Note");
    await page.getByLabel("Body").fill("Body content");
    await page.getByLabel("Tags").fill("alpha");
    await page.getByRole("button", { name: "Create Note" }).click();

    await expect(page.getByText("Your Notes (1)")).toBeVisible();

    // Filter then clear
    const filterInput = page.getByPlaceholder("Filter by tag");
    await filterInput.fill("nonexistent");
    await expect(page.getByText("Your Notes (0)")).toBeVisible();

    await filterInput.clear();
    await expect(page.getByText("Your Notes (1)")).toBeVisible();
  });

  test("should filter case-insensitively", async ({ page }) => {
    await page.getByLabel("Title").fill("Case Note");
    await page.getByLabel("Body").fill("Body");
    await page.getByLabel("Tags").fill("JavaScript");
    await page.getByRole("button", { name: "Create Note" }).click();

    await expect(page.getByText("Your Notes (1)")).toBeVisible();

    const filterInput = page.getByPlaceholder("Filter by tag");
    await filterInput.fill("javascript");

    await expect(page.getByText("Your Notes (1)")).toBeVisible();
  });

  test("should filter by partial tag match", async ({ page }) => {
    await page.getByLabel("Title").fill("Partial Note");
    await page.getByLabel("Body").fill("Body");
    await page.getByLabel("Tags").fill("programming");
    await page.getByRole("button", { name: "Create Note" }).click();

    await expect(page.getByText("Your Notes (1)")).toBeVisible();

    const filterInput = page.getByPlaceholder("Filter by tag");
    await filterInput.fill("prog");

    await expect(page.getByText("Your Notes (1)")).toBeVisible();
  });
});
