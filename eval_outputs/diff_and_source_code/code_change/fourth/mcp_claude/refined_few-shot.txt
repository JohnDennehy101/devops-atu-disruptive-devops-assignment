import { test, expect } from '@playwright/test';

test.describe('Tag filter on Your Notes sidebar', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
  });

  test('filter by tag input is visible', async ({ page }) => {
    await expect(page.getByPlaceholder('Filter by tag')).toBeVisible();
  });

  test('shows all saved notes when filter is empty', async ({ page }) => {
    const noteButtons = page.locator('button', { hasText: /^Note #/ });
    const count = await noteButtons.count();
    const heading = page.getByRole('heading', { level: 3 });
    await expect(heading).toContainText(`Your Notes (${count})`);
  });

  test('filters notes by tag and updates count', async ({ page }) => {
    const heading = page.getByRole('heading', { level: 3 });
    const initialText = await heading.textContent();
    const initialCount = parseInt(initialText!.match(/\d+/)![0], 10);

    // Type a non-matching tag to filter out all notes
    await page.getByPlaceholder('Filter by tag').fill('zzz_nonexistent_tag');
    await expect(heading).toContainText('Your Notes (0)');
    await expect(page.getByText('No notes match the filter')).toBeVisible();

    // Clear the filter to restore all notes
    await page.getByPlaceholder('Filter by tag').fill('');
    await expect(heading).toContainText(`Your Notes (${initialCount})`);
    await expect(page.getByText('No notes match the filter')).not.toBeVisible();
  });

  test('shows "No notes match the filter" when tag filter has no matches', async ({ page }) => {
    await page.getByPlaceholder('Filter by tag').fill('unmatchable_tag_xyz');
    await expect(page.getByText('No notes match the filter')).toBeVisible();
    const noteButtons = page.locator('button', { hasText: /^Note #/ });
    await expect(noteButtons).toHaveCount(0);
  });

  test('creates a note with a tag then filters by that tag', async ({ page }) => {
    const uniqueTag = `e2etag_${Date.now()}`;

    // Create a note with the unique tag
    await page.getByRole('textbox', { name: 'Title' }).fill('Filter Test Note');
    await page.getByRole('textbox', { name: 'Body' }).fill('Body for filter test');
    await page.getByRole('textbox', { name: 'Tags (comma-separated)' }).fill(uniqueTag);
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Wait for the new note to appear in sidebar
    const heading = page.getByRole('heading', { level: 3 });
    await expect(heading).toContainText(/Your Notes \(\d+\)/);

    // Filter by the unique tag — should show exactly 1 note
    await page.getByPlaceholder('Filter by tag').fill(uniqueTag);
    await expect(heading).toContainText('Your Notes (1)');
    const noteButtons = page.locator('button', { hasText: /^Note #/ });
    await expect(noteButtons).toHaveCount(1);

    // Filter by a different tag — the created note should disappear
    await page.getByPlaceholder('Filter by tag').fill('other_tag_no_match');
    await expect(heading).toContainText('Your Notes (0)');
    await expect(page.getByText('No notes match the filter')).toBeVisible();
  });

  test('tag filter is case-insensitive', async ({ page }) => {
    const baseTag = `CaseTest_${Date.now()}`;

    // Create a note with mixed-case tag
    await page.getByRole('textbox', { name: 'Title' }).fill('Case Test Note');
    await page.getByRole('textbox', { name: 'Body' }).fill('Body for case test');
    await page.getByRole('textbox', { name: 'Tags (comma-separated)' }).fill(baseTag);
    await page.getByRole('button', { name: 'Create Note' }).click();

    const heading = page.getByRole('heading', { level: 3 });
    await expect(heading).toContainText(/Your Notes \(\d+\)/);

    // Filter with lowercase version of the tag
    await page.getByPlaceholder('Filter by tag').fill(baseTag.toLowerCase());
    await expect(heading).not.toContainText('Your Notes (0)');
    const noteButtons = page.locator('button', { hasText: /^Note #/ });
    expect(await noteButtons.count()).toBeGreaterThanOrEqual(1);
  });
});
