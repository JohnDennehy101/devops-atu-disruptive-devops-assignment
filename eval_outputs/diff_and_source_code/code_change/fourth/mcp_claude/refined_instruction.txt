import { test, expect } from '@playwright/test';

test.describe('Tag Filter Feature', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('filter by tag input is visible', async ({ page }) => {
    await expect(page.getByRole('textbox', { name: 'Filter by tag' })).toBeVisible();
  });

  test('shows all notes when filter is empty', async ({ page }) => {
    // Create two notes with different tags
    await page.getByLabel('Title').fill('Note Alpha');
    await page.getByLabel('Body').fill('Body alpha');
    await page.getByLabel(/Tags/).fill('javascript');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByLabel('Title').fill('Note Beta');
    await page.getByLabel('Body').fill('Body beta');
    await page.getByLabel(/Tags/).fill('python');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (2)')).toBeVisible();

    // Filter input is empty — both notes should be listed
    const filterInput = page.getByRole('textbox', { name: 'Filter by tag' });
    await expect(filterInput).toHaveValue('');
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(2);
  });

  test('filters notes by matching tag', async ({ page }) => {
    // Create two notes with different tags
    await page.getByLabel('Title').fill('JS Note');
    await page.getByLabel('Body').fill('About javascript');
    await page.getByLabel(/Tags/).fill('javascript, frontend');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByLabel('Title').fill('Python Note');
    await page.getByLabel('Body').fill('About python');
    await page.getByLabel(/Tags/).fill('python, backend');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (2)')).toBeVisible();

    // Filter by "javascript" — only one note should match
    await page.getByRole('textbox', { name: 'Filter by tag' }).fill('javascript');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(1);
  });

  test('shows "No notes match the filter" when no tags match', async ({ page }) => {
    // Create a note
    await page.getByLabel('Title').fill('Some Note');
    await page.getByLabel('Body').fill('Some body');
    await page.getByLabel(/Tags/).fill('existingtag');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    // Type a non-matching filter
    await page.getByRole('textbox', { name: 'Filter by tag' }).fill('nonexistenttag');
    await expect(page.getByText('Your Notes (0)')).toBeVisible();
    await expect(page.getByText('No notes match the filter')).toBeVisible();
  });

  test('shows "No notes yet" when there are no saved notes', async ({ page }) => {
    await expect(page.getByText('Your Notes (0)')).toBeVisible();
    await expect(page.getByText('No notes yet')).toBeVisible();
  });

  test('clearing filter restores all notes', async ({ page }) => {
    // Create two notes
    await page.getByLabel('Title').fill('First');
    await page.getByLabel('Body').fill('Body one');
    await page.getByLabel(/Tags/).fill('alpha');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByLabel('Title').fill('Second');
    await page.getByLabel('Body').fill('Body two');
    await page.getByLabel(/Tags/).fill('beta');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (2)')).toBeVisible();

    // Filter to show only one
    const filterInput = page.getByRole('textbox', { name: 'Filter by tag' });
    await filterInput.fill('alpha');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    // Clear filter — both notes restored
    await filterInput.fill('');
    await expect(page.getByText('Your Notes (2)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(2);
  });

  test('tag filter is case-insensitive', async ({ page }) => {
    await page.getByLabel('Title').fill('Case Test');
    await page.getByLabel('Body').fill('Testing case');
    await page.getByLabel(/Tags/).fill('JavaScript');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    // Filter with lowercase should still match "JavaScript"
    await page.getByRole('textbox', { name: 'Filter by tag' }).fill('javascript');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(1);
  });

  test('tag filter supports partial matching', async ({ page }) => {
    await page.getByLabel('Title').fill('Partial Match');
    await page.getByLabel('Body').fill('Testing partial');
    await page.getByLabel(/Tags/).fill('frontend');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    // "front" should partially match "frontend"
    await page.getByRole('textbox', { name: 'Filter by tag' }).fill('front');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(1);
  });

  test('note count in heading updates with filter', async ({ page }) => {
    // Create three notes
    await page.getByLabel('Title').fill('Note A');
    await page.getByLabel('Body').fill('Body A');
    await page.getByLabel(/Tags/).fill('shared, unique-a');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByLabel('Title').fill('Note B');
    await page.getByLabel('Body').fill('Body B');
    await page.getByLabel(/Tags/).fill('shared, unique-b');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (2)')).toBeVisible();

    await page.getByLabel('Title').fill('Note C');
    await page.getByLabel('Body').fill('Body C');
    await page.getByLabel(/Tags/).fill('unique-c');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (3)')).toBeVisible();

    // Filter by "shared" — should show 2 notes
    await page.getByRole('textbox', { name: 'Filter by tag' }).fill('shared');
    await expect(page.getByText('Your Notes (2)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(2);

    // Filter by "unique-a" — should show 1 note
    await page.getByRole('textbox', { name: 'Filter by tag' }).fill('unique-a');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(1);
  });
});
