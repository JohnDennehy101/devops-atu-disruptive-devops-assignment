import { test, expect } from '@playwright/test';

test('test tag filter functionality', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Add a note with a specific tag
  await page.fill('#note-title', 'Test Note');
  await page.fill('#note-body', 'This is a test note.');
  await page.fill('#note-tags', 'testTag');
  await page.click('text=Create Note');

  // Verify the note is added
  await page.waitForSelector('text=Note #1');
  await expect(page.locator('text=Note #1')).toBeVisible();

  // Filter notes by tag
  await page.fill('input[placeholder="Filter by tag"]', 'testTag');
  await page.waitForSelector('text=Note #1');
  await expect(page.locator('text=Note #1')).toBeVisible();

  // Clear the filter
  await page.fill('input[placeholder="Filter by tag"]', '');
  await page.waitForSelector('text=Note #1');
  await expect(page.locator('text=Note #1')).toBeVisible();

  // Add another note without the tag
  await page.fill('#note-title', 'Another Note');
  await page.fill('#note-body', 'This is another test note.');
  await page.fill('#note-tags', '');
  await page.click('text=Create Note');

  // Verify the second note is added
  await page.waitForSelector('text=Note #2');
  await expect(page.locator('text=Note #1')).toBeVisible();
  await expect(page.locator('text=Note #2')).toBeVisible();

  // Filter notes by tag again
  await page.fill('input[placeholder="Filter by tag"]', 'testTag');
  await page.waitForSelector('text=Note #1');
  await expect(page.locator('text=Note #1')).toBeVisible();
  await expect(page.locator('text=Note #2')).toBeHidden();
});
