import { test, expect } from '@playwright/test';

test('filter notes by tag', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await page.locator('#tag-filter').fill('test');
  await expect(page.locator('#note-list')).toContainText('Note 1');
  await expect(page.locator('#note-list')).not.toContainText('Note 2');
});
