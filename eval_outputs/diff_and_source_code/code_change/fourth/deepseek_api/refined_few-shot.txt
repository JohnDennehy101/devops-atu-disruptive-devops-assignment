import { test, expect } from '@playwright/test';

test('tag filter functionality', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Create notes with different tags
  await page.fill('#note-title', 'Note with tag1');
  await page.fill('#note-body', 'Body for tag1 note');
  await page.fill('#note-tags', 'tag1, important');
  await page.click('button[type="submit"]');
  await expect(page.getByText('Creating...')).toBeHidden();

  await page.fill('#note-title', 'Note with tag2');
  await page.fill('#note-body', 'Body for tag2 note');
  await page.fill('#note-tags', 'tag2, urgent');
  await page.click('button[type="submit"]');
  await expect(page.getByText('Creating...')).toBeHidden();

  await page.fill('#note-title', 'Note with both tags');
  await page.fill('#note-body', 'Body for both tags note');
  await page.fill('#note-tags', 'tag1, tag2');
  await page.click('button[type="submit"]');
  await expect(page.getByText('Creating...')).toBeHidden();

  // Check initial count shows all notes
  await expect(page.getByText('Your Notes (3)')).toBeVisible();

  // Filter by tag1
  await page.getByPlaceholder('Filter by tag').fill('tag1');
  await expect(page.getByText('Your Notes (2)')).toBeVisible();
  await expect(page.getByText('Note #')).toHaveCount(2);

  // Filter by tag2
  await page.getByPlaceholder('Filter by tag').fill('tag2');
  await expect(page.getByText('Your Notes (2)')).toBeVisible();
  await expect(page.getByText('Note #')).toHaveCount(2);

  // Filter by urgent
  await page.getByPlaceholder('Filter by tag').fill('urgent');
  await expect(page.getByText('Your Notes (1)')).toBeVisible();
  await expect(page.getByText('Note #')).toHaveCount(1);

  // Filter with no matches
  await page.getByPlaceholder('Filter by tag').fill('nonexistent');
  await expect(page.getByText('Your Notes (0)')).toBeVisible();
  await expect(page.getByText('No notes match the filter')).toBeVisible();

  // Clear filter
  await page.getByPlaceholder('Filter by tag').fill('');
  await expect(page.getByText('Your Notes (3)')).toBeVisible();
  await expect(page.getByText('Note #')).toHaveCount(3);
});

test('filter message when no saved notes', async ({ page }) => {
  await page.goto('http://localhost:5175');
  
  // Clear any existing saved notes
  await page.evaluate(() => {
    localStorage.removeItem('noteIds');
  });
  await page.reload();
  
  await expect(page.getByText('Your Notes (0)')).toBeVisible();
  await expect(page.getByText('No notes yet')).toBeVisible();
  
  // Filter input should still be present but empty
  const filterInput = page.getByPlaceholder('Filter by tag');
  await expect(filterInput).toBeVisible();
  await expect(filterInput).toHaveValue('');
  
  // Typing in filter when no notes exist
  await filterInput.fill('anytag');
  await expect(page.getByText('Your Notes (0)')).toBeVisible();
  await expect(page.getByText('No notes yet')).toBeVisible();
});
