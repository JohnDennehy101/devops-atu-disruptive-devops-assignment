import { test, expect } from '@playwright/test';

test('home page has title', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await expect(page.locator('h1')).getByText('Note Taking App');
});

test('create note form exists', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await expect(page.locator('form')).toBeVisible();
  await expect(page.locator('label[for="note-title"]')).getByText('Title');
  await expect(page.locator('label[for="note-body"]')).getByText('Body');
  await expect(page.locator('label[for="note-tags"]')).getByText('Tags (comma-separated)');
});

test('saved notes section exists', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await expect(page.locator('h3')).getByText('Your Notes (0)');
  await expect(page.locator('div[class*="space-y-2"]')).toBeVisible();
});

test('no notes message is displayed when no notes are saved', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await expect(page.locator('p')).getByText('No notes yet');
});
