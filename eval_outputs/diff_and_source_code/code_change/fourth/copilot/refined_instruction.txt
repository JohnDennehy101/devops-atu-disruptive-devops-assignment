import { test, expect } from '@playwright/test';

test('should filter notes by tag', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Verify initial state
  await expect(page.getByText('Your Notes (0)')).toBeVisible();
  await expect(page.getByText('No notes yet')).toBeVisible();

  // Create a new note
  await page.getByRole('textbox', { name: 'Title' }).fill('Test Note');
  await page.getByRole('textbox', { name: 'Body' }).fill('This is a test note.');
  await page.getByRole('textbox', { name: 'Tags (comma-separated)' }).fill('test, example');
  await page.getByRole('button', { name: 'Create Note' }).click();

  // Verify the note is created
  await expect(page.getByText('Your Notes (1)')).toBeVisible();
  await expect(page.getByText('Note #1')).toBeVisible();

  // Add another note
  await page.getByRole('textbox', { name: 'Title' }).fill('Another Note');
  await page.getByRole('textbox', { name: 'Body' }).fill('This is another test note.');
  await page.getByRole('textbox', { name: 'Tags (comma-separated)' }).fill('example, sample');
  await page.getByRole('button', { name: 'Create Note' }).click();

  // Verify the second note is created
  await expect(page.getByText('Your Notes (2)')).toBeVisible();
  await expect(page.getByText('Note #2')).toBeVisible();

  // Filter notes by tag
  await page.getByRole('textbox', { placeholder: 'Filter by tag' }).fill('test');
  await expect(page.getByText('Your Notes (1)')).toBeVisible();
  await expect(page.getByText('Note #1')).toBeVisible();
  await expect(page.getByText('Note #2')).not.toBeVisible();

  // Clear the filter
  await page.getByRole('textbox', { placeholder: 'Filter by tag' }).fill('');
  await expect(page.getByText('Your Notes (2)')).toBeVisible();
  await expect(page.getByText('Note #1')).toBeVisible();
  await expect(page.getByText('Note #2')).toBeVisible();

  // Apply a filter with no matches
  await page.getByRole('textbox', { placeholder: 'Filter by tag' }).fill('nonexistent');
  await expect(page.getByText('Your Notes (0)')).toBeVisible();
  await expect(page.getByText('No notes match the filter')).toBeVisible();
});
