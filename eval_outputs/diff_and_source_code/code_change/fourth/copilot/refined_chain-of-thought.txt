import { test, expect } from '@playwright/test';

test('Filter notes by tag', async ({ page }) => {
  // Navigate to the home page
  await page.goto('http://localhost:5175');

  // Verify the "Filter by tag" input is present
  const filterInput = page.locator('input[placeholder="Filter by tag"]');
  await expect(filterInput).toBeVisible();

  // Verify the initial state of "Your Notes" count
  const notesCount = page.locator('h3:has-text("Your Notes")');
  await expect(notesCount).toContainText('Your Notes (0)');

  // Add a new note
  await page.fill('#note-title', 'Test Note');
  await page.fill('#note-body', 'This is a test note.');
  await page.fill('#note-tags', 'test, example');
  await page.click('button:has-text("Create Note")');

  // Wait for the note to appear in the list
  const noteButton = page.locator('button:has-text("Note #1")');
  await expect(noteButton).toBeVisible();

  // Verify the updated "Your Notes" count
  await expect(notesCount).toContainText('Your Notes (1)');

  // Filter notes by a matching tag
  await filterInput.fill('test');
  await expect(notesCount).toContainText('Your Notes (1)');

  // Filter notes by a non-matching tag
  await filterInput.fill('nonexistent');
  await expect(notesCount).toContainText('Your Notes (0)');
  const noNotesMessage = page.locator('text=No notes match the filter');
  await expect(noNotesMessage).toBeVisible();

  // Clear the filter and verify all notes are displayed again
  await filterInput.fill('');
  await expect(notesCount).toContainText('Your Notes (1)');
  await expect(noteButton).toBeVisible();
});
