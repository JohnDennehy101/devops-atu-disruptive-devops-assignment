import { test, expect } from '@playwright/test';

test('filters saved notes by tag and shows matching count and empty-filter state', async ({ page }) => {
  await page.goto('http://localhost:5175');

  const titleInput = page.locator('#note-title');
  const bodyInput = page.locator('#note-body');
  const tagsInput = page.locator('#note-tags');
  const createButton = page.getByRole('button', { name: 'Create Note' });

  await titleInput.fill('First Note');
  await bodyInput.fill('Body for first note');
  await tagsInput.fill('work, urgent');
  await createButton.click();

  await expect(page.getByRole('heading', { name: /Your Notes \(1\)/ })).toBeVisible();
  await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(1);

  await titleInput.fill('Second Note');
  await bodyInput.fill('Body for second note');
  await tagsInput.fill('personal, ideas');
  await createButton.click();

  await expect(page.getByRole('heading', { name: /Your Notes \(2\)/ })).toBeVisible();
  await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(2);

  const filterInput = page.getByPlaceholder('Filter by tag');

  await filterInput.fill('work');
  await expect(page.getByRole('heading', { name: 'Your Notes (1)' })).toBeVisible();
  await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(1);
  await expect(page.getByText('No notes match the filter')).toHaveCount(0);

  await filterInput.fill('WORK');
  await expect(page.getByRole('heading', { name: 'Your Notes (1)' })).toBeVisible();
  await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(1);

  await filterInput.fill('personal');
  await expect(page.getByRole('heading', { name: 'Your Notes (1)' })).toBeVisible();
  await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(1);

  await filterInput.fill('missing-tag');
  await expect(page.getByRole('heading', { name: 'Your Notes (0)' })).toBeVisible();
  await expect(page.getByText('No notes match the filter')).toBeVisible();
  await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(0);

  await filterInput.fill('');
  await expect(page.getByRole('heading', { name: 'Your Notes (2)' })).toBeVisible();
  await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(2);
});

test('shows no notes yet when there are no saved notes', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await expect(page.getByRole('heading', { name: 'Your Notes (0)' })).toBeVisible();
  await expect(page.getByPlaceholder('Filter by tag')).toBeVisible();
  await expect(page.getByText('No notes yet')).toBeVisible();
});
