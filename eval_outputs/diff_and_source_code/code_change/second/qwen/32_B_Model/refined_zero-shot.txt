import { test, expect } from '@playwright/test';

test('character count for form body is displayed correctly', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Type in the body textarea
  const bodyTextarea = page.locator('#note-body');
  await bodyTextarea.type('Hello, world!');

  // Check the character count display
  const charCountDisplay = page.locator('.text-xs.text-gray-500.dark:text-gray-400.mt-1');
  await expect(charCountDisplay).toHaveText('13 / 5000');
});
