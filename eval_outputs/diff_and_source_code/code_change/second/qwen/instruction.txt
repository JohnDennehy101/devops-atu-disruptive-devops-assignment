import { test, expect } from '@playwright/test';

test('Character count indicator updates correctly', async ({ page }) => {
  // Step 1: Navigate to the home page
  await page.goto('/');

  // Step 2: Check if the character count indicator is displayed
  const charCountIndicator = await page.getByRole('paragraph', { name: '0 / 5000' });
  await expect(charCountIndicator).toBeVisible();

  // Step 3: Enter text into the body field and verify that the character count updates correctly
  const bodyField = await page.getByLabel('Body');
  await bodyField.fill('Hello, World!');
  const updatedCharCountIndicator = await page.getByRole('paragraph', { name: '13 / 5000' });
  await expect(updatedCharCountIndicator).toHaveText('13 / 5000');

  // Step 4: Clear the text and verify that the character count resets to zero
  await bodyField.clear();
  const resetCharCountIndicator = await page.getByRole('paragraph', { name: '0 / 5000' });
  await expect(resetCharCountIndicator).toHaveText('0 / 5000');
});
