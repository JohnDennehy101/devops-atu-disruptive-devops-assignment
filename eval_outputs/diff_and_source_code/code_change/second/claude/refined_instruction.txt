import { test, expect } from '@playwright/test';

test('displays character count for note body field', async ({ page }) => {
  await page.goto('http://localhost:5175');

  const bodyTextarea = page.getByRole('textbox', { name: 'Body' });
  await expect(bodyTextarea).toBeVisible();

  await expect(page.getByText('0 / 5000')).toBeVisible();

  await bodyTextarea.fill('Hello');
  await expect(page.getByText('5 / 5000')).toBeVisible();

  await bodyTextarea.fill('Testing character count');
  await expect(page.getByText('23 / 5000')).toBeVisible();
});
