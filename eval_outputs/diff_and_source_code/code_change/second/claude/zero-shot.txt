import { test, expect } from '@playwright/test';

test.describe('Body character counter', () => {
  test('should display character count as 0 / 5000 when body is empty', async ({ page }) => {
    await page.goto('/');
    await expect(page.getByText('0 / 5000')).toBeVisible();
  });

  test('should update character count as user types in the body field', async ({ page }) => {
    await page.goto('/');
    const bodyField = page.getByLabel('Body');
    await bodyField.fill('Hello');
    await expect(page.getByText('5 / 5000')).toBeVisible();
  });

  test('should display correct character count for longer text', async ({ page }) => {
    await page.goto('/');
    const bodyField = page.getByLabel('Body');
    const text = 'a'.repeat(100);
    await bodyField.fill(text);
    await expect(page.getByText('100 / 5000')).toBeVisible();
  });
});
