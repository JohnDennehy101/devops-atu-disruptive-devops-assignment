test('shows character count', async ({ page }) => {
  await page.goto('/');
  await page.locator('#note-body').fill('a'.repeat(5001));
  await expect(page.locator('.text-xs.text-gray-500.dark:text-gray-400.mt-1')).toBeVisible();
  await expect(page.locator('.text-xs.text-gray-500.dark:text-gray-400.mt-1')).toHaveText('5001 / 5000');
});
