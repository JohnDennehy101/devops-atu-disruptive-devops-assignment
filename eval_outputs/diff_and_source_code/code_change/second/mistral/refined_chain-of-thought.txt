import { test, expect } from '@playwright/test';

test('Check new paragraph element and its content', async ({ page }) => {
  await page.goto('http://localhost:5175/');
  const para = page.locator('p.text-xs.text-gray-500.dark:text-gray-400.mt-1');
  await expect(para).toBeVisible();
  await expect(para).toHaveText(/0 \/ 5000/);
});

test('Verify paragraph text updates with form body length', async ({ page }) => {
  await page.goto('http://localhost:5175/');
  const para = page.locator('p.text-xs.text-gray-500.dark:text-gray-400.mt-1');
  const formBody = page.locator('#note-body');
  await expect(para).toHaveText(/0 \/ 5000/);
  await formBody.fill('Test note body');
  await expect(para).toHaveText(/16 \/ 5000/);
  await formBody.fill('');
  await expect(para).toHaveText(/0 \/ 5000/);
});
