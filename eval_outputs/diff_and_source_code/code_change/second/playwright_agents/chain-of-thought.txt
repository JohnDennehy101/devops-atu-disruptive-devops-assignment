import { test, expect } from '@playwright/test';

test.describe('Character Count Display', () => {
  test('should display accurate count at and beyond the 5000 character boundary', async ({ page }) => {
    // 1. Navigate to http://localhost:5175 with a fresh localStorage (seed state)
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();

    // expect: The 'Create New Note' form is visible and the character count reads '0 / 5000'
    await expect(page.getByText('Create New Note')).toBeVisible();
    await expect(page.getByText('0 / 5000')).toBeVisible();

    // 2. Fill the Body textarea with exactly 5000 characters (repeat the letter 'A' 5000 times)
    const bodyTextarea = page.getByRole('textbox', { name: 'Body' });
    await bodyTextarea.fill('A'.repeat(5000));

    // expect: The character count below the Body textarea shows '5000 / 5000'
    await expect(page.getByText('5000 / 5000')).toBeVisible();

    // 3. Type one additional character into the Body textarea so it contains 5001 characters
    await bodyTextarea.press('End');
    await bodyTextarea.pressSequentially('B');

    // expect: The character count updates to '5001 / 5000', correctly reflecting the over-limit length
    await expect(page.getByText('5001 / 5000')).toBeVisible();

    // expect: The 'Create Note' button remains visible and is not disabled by the character count alone
    await expect(page.getByRole('button', { name: 'Create Note' })).toBeVisible();
  });
});

// File: app/e2e/generated/character-count-realtime.spec.ts
// spec: app/specs/character-count.plan.md
// seed: app/e2e/seed-generator.spec.ts


test.describe('Character Count Display', () => {
  test('should update character count in real-time as text is typed and deleted', async ({ page }) => {
    // 1. Navigate to http://localhost:5175 with a fresh localStorage (seed state)
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();

    // expect: The 'Create New Note' form is visible and the character count reads '0 / 5000'
    await expect(page.getByText('Create New Note')).toBeVisible();
    await expect(page.getByText('0 / 5000')).toBeVisible();

    // 2. Click into the Body textarea and type the text 'Hello World' (11 characters)
    const bodyTextarea = page.getByRole('textbox', { name: 'Body' });
    await bodyTextarea.fill('Hello World');

    // expect: The character count below the Body textarea updates immediately to '11 / 5000'
    await expect(page.getByText('11 / 5000')).toBeVisible();

    // 3. Append additional text to the Body textarea so it contains 'Hello World This is a test' (26 characters)
    await bodyTextarea.fill('Hello World This is a test');

    // expect: The character count updates to '26 / 5000' reflecting the new total length
    await expect(page.getByText('26 / 5000')).toBeVisible();

    // 4. Clear all text from the Body textarea (select all and delete)
    await bodyTextarea.click();
    await page.keyboard.press('ControlOrMeta+a');
    await page.keyboard.press('Delete');

    // expect: The character count resets to '0 / 5000' once the textarea is empty
    await expect(page.getByText('0 / 5000')).toBeVisible();
  });
});
