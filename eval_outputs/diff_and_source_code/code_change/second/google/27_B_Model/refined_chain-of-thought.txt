import { test, expect } from '@playwright/test';

test('verify formBody character count', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await page.locator('textarea[id="note-body"]').fill('This is a test note body with a long text to verify the character count.');
  const formBodyLocator = page.locator('textarea[id="note-body"]');
  const formBodyLength = await formBodyLocator.getAttribute('value').then(value => value?.length || 0);
  const characterCountElement = page.locator('p.text-xs.text-gray-500.dark\\:text-gray-400.mt-1');
  const characterCountText = await characterCountElement.textContent();
  expect(characterCountText).toContain(`${formBodyLength} / 5000`);
});
