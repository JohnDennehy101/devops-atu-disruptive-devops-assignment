import { test, expect } from '@playwright/test';

test.describe('Home page Clear All button', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
  });

  test('should not show Clear All button when there are no notes', async ({ page }) => {
    const clearAllButton = page.getByRole('button', { name: 'Clear all' });
    await expect(clearAllButton).toBeHidden();
  });

  test('should show Clear All button when there are notes', async ({ page }) => {
    await page.getByLabel('Title').fill('Test Note');
    await page.getByLabel('Body').fill('Test body content');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
    const clearAllButton = page.getByRole('button', { name: 'Clear all' });
    await expect(clearAllButton).toBeVisible();
    await expect(clearAllButton).toHaveClass(/text-red-600/);
  });

  test('should clear all notes when Clear All button is clicked', async ({ page }) => {
    await page.getByLabel('Title').fill('First Note');
    await page.getByLabel('Body').fill('First body');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await page.getByLabel('Title').fill('Second Note');
    await page.getByLabel('Body').fill('Second body');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByText('Your Notes (2)')).toBeVisible();
    const clearAllButton = page.getByRole('button', { name: 'Clear all' });
    await clearAllButton.click();
    await expect(page.getByText('Your Notes (0)')).toBeVisible();
    await expect(page.getByText('No notes yet')).toBeVisible();
    await expect(clearAllButton).toBeHidden();
  });

  test('should clear form fields and selection after clearing all notes', async ({ page }) => {
    await page.getByLabel('Title').fill('Note to clear');
    await page.getByLabel('Body').fill('Body to clear');
    await page.getByLabel('Tags (comma-separated)').fill('tag1, tag2');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await page.getByText('Note #').first().click();
    await page.getByRole('button', { name: 'Edit' }).click();
    await expect(page.getByLabel('Title')).toHaveValue('Note to clear');
    await expect(page.getByLabel('Body')).toHaveValue('Body to clear');
    await expect(page.getByLabel('Tags (comma-separated)')).toHaveValue('tag1, tag2');
    await page.getByRole('button', { name: 'Clear all' }).click();
    await expect(page.getByLabel('Title')).toHaveValue('');
    await expect(page.getByLabel('Body')).toHaveValue('');
    await expect(page.getByLabel('Tags (comma-separated)')).toHaveValue('');
    await expect(page.getByText('Note Details')).toBeHidden();
  });

  test('should persist clear all after page reload', async ({ page }) => {
    await page.getByLabel('Title').fill('Persistent Note');
    await page.getByLabel('Body').fill('Persistent body');
    await page.getByRole('button', { name: 'Create Note' }).click();
    await page.getByRole('button', { name: 'Clear all' }).click();
    await page.reload();
    await expect(page.getByText('Your Notes (0)')).toBeVisible();
    await expect(page.getByText('No notes yet')).toBeVisible();
  });
});
