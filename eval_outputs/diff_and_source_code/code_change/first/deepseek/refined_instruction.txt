import { test, expect } from '@playwright/test';

test('should create a new note', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Fill out the note creation form
  await page.getByRole('textbox', { name: 'Title' }).fill('Test Note');
  await page.getByRole('textbox', { name: 'Body' }).fill('This is a test note body.');
  await page.getByRole('textbox', { name: 'Tags (comma-separated)' }).fill('test, note');

  // Submit the form to create the note
  await page.getByRole('button', { name: 'Create Note' }).click();

  // Wait for the note to appear in the list
  await page.getByRole('button', { name: 'Note #' }).first().click();

  // Verify that the note details are correct
  await expect(page.getByText('Test Note')).toBeVisible();
  await expect(page.getByText('This is a test note body.')).toBeVisible();
  await expect(page.getByText('test, note')).toBeVisible();
});

test('should edit a note', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Load an existing note
  await page.getByRole('button', { name: 'Note #' }).first().click();

  // Start editing the note
  await page.getByRole('button', { name: 'Edit' }).click();

  // Modify the note details
  await page.getByRole('textbox', { name: 'Title' }).fill('Updated Note Title');
  await page.getByRole('textbox', { name: 'Body' }).fill('This is an updated note body.');
  await page.getByRole('textbox', { name: 'Tags (comma-separated)' }).fill('updated, note');

  // Save the changes
  await page.getByRole('button', { name: 'Save' }).click();

  // Verify that the note details are updated
  await expect(page.getByText('Updated Note Title')).toBeVisible();
  await expect(page.getByText('This is an updated note body.')).toBeVisible();
  await expect(page.getByText('updated, note')).toBeVisible();
});

test('should delete a note', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Load an existing note
  await page.getByRole('button', { name: 'Note #' }).first().click();

  // Delete the note
  await page.getByRole('button', { name: 'Delete' }).click();

  // Verify that the note is no longer in the list
  await expect(page.getByRole('button', { name: 'Note #' }).first()).toBeHidden();
});

test('should clear all notes', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Create a new note
  await page.getByRole('textbox', { name: 'Title' }).fill('Temp Note');
  await page.getByRole('textbox', { name: 'Body' }).fill('This is a temporary note.');
  await page.getByRole('textbox', { name: 'Tags (comma-separated)' }).fill('temp');
  await page.getByRole('button', { name: 'Create Note' }).click();

  // Clear all notes
  await page.getByRole('button', { name: 'Clear all' }).click();

  // Verify that no notes are present
  await expect(page.getByText('No notes yet')).toBeVisible();
});
