import { test, expect } from '@playwright/test';

test('should create, edit, and delete a note', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Create a new note
  await page.fill('input[name="note-title"]', 'Test Note');
  await page.fill('textarea[name="note-body"]', 'This is a test note.');
  await page.fill('input[name="note-tags"]', 'test, note');
  await page.click('button:has-text("Create Note")');

  // Wait for the note to appear in the list
  await expect(page.locator('button:has-text("Note #")')).toHaveText(/Note #\d+/);

  // Edit the note
  const noteId = await page.locator('button:has-text("Note #")').first().innerText();
  const noteIdNumber = parseInt(noteId.match(/\d+/)[0], 10);
  await page.click(`button:has-text("Edit")`);
  await page.fill('input[name="note-title"]', 'Edited Note');
  await page.fill('textarea[name="note-body"]', 'This is an edited test note.');
  await page.fill('input[name="note-tags"]', 'edited, test, note');
  await page.click('button:has-text("Save")');

  // Wait for the note to be updated
  await expect(page.locator('button:has-text("Note #")')).toHaveText(/Note #\d+/);

  // Delete the note
  await page.click(`button:has-text("Delete")`);
  await expect(page.locator('button:has-text("Note #")')).not.toHaveText(/Note #\d+/);
});
