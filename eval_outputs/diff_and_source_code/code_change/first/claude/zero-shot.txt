import { test, expect } from "@playwright/test";

test.describe("Clear All Notes functionality", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/");
  });

  test("should display subtitle text about local storage", async ({ page }) => {
    await expect(
      page.getByText("Create and manage notes. Your notes are stored locally.")
    ).toBeVisible();
  });

  test("should not show 'Clear all' button when there are no notes", async ({
    page,
  }) => {
    await page.evaluate(() => localStorage.setItem("noteIds", "[]"));
    await page.reload();
    await expect(page.getByText("Clear all")).not.toBeVisible();
  });

  test("should show 'Clear all' button when there are saved notes", async ({
    page,
  }) => {
    await page.evaluate(() => localStorage.setItem("noteIds", "[1, 2]"));
    await page.reload();
    await expect(page.getByText("Clear all")).toBeVisible();
  });

  test("should display note count in 'Your Notes' heading", async ({
    page,
  }) => {
    await page.evaluate(() => localStorage.setItem("noteIds", "[1, 2, 3]"));
    await page.reload();
    await expect(page.getByText("Your Notes (3)")).toBeVisible();
  });

  test("should clear all notes when 'Clear all' button is clicked", async ({
    page,
  }) => {
    await page.evaluate(() => localStorage.setItem("noteIds", "[1, 2]"));
    await page.reload();

    await expect(page.getByText("Your Notes (2)")).toBeVisible();
    await page.getByText("Clear all").click();

    await expect(page.getByText("Your Notes (0)")).toBeVisible();
    await expect(page.getByText("No notes yet")).toBeVisible();
    await expect(page.getByText("Clear all")).not.toBeVisible();

    const storedIds = await page.evaluate(() =>
      localStorage.getItem("noteIds")
    );
    expect(storedIds).toBe("[]");
  });

  test("should reset form state when clearing all notes", async ({ page }) => {
    await page.evaluate(() => localStorage.setItem("noteIds", "[1]"));
    await page.reload();

    await page.getByText("Clear all").click();

    await expect(page.locator("#note-title")).toHaveValue("");
    await expect(page.locator("#note-body")).toHaveValue("");
    await expect(page.locator("#note-tags")).toHaveValue("");
  });

  test("should show 'Create New Note' heading after clearing all notes", async ({
    page,
  }) => {
    await page.evaluate(() => localStorage.setItem("noteIds", "[1]"));
    await page.reload();

    await page.getByText("Clear all").click();

    await expect(page.getByText("Create New Note")).toBeVisible();
  });
});
