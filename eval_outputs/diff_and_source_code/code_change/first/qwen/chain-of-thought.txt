import { test, expect } from '@playwright/test';

test('Clear all notes functionality', async ({ page }) => {
  // Step 1: Navigate to the Home page
  await page.goto('/');

  // Step 2: Check if the "Clear all" button is present
  const clearAllButton = page.locator('button:has-text("Clear all")');
  await expect(clearAllButton).toBeVisible();

  // Step 3: Click on the "Clear all" button
  await clearAllButton.click();

  // Step 4: Verify that the "Your Notes" count is 0
  const notesCount = page.locator('h3:has-text("Your Notes (0)")');
  await expect(notesCount).toBeVisible();

  // Step 5: Verify that the local storage is cleared
  const localStorageValue = await page.evaluate(() => localStorage.getItem('noteIds'));
  expect(localStorageValue).toBe('[]');
});
