import { test, expect } from '@playwright/test';

test('displays the correct number of notes', async ({ page }) => {
  await page.goto('http://localhost:5175');
  const noteCount = await page.locator('h3').textContent();
  expect(noteCount).toContain('Your Notes (0)');
});

test('displays the clear all button when there are notes', async ({ page }) => {
  await page.goto('http://localhost:5175');
  // Simulate adding a note (this part would need to be implemented based on actual app functionality)

test('clears all notes when the clear all button is clicked', async ({ page }) => {
  await page.goto('http://localhost:5175');
  // Simulate adding a note (this part would need to be implemented based on actual app functionality)
  await page.fill('#note-title', 'Test Note');

  await page.fill('#note-tags', 'test, note');

  // Click the clear all button
