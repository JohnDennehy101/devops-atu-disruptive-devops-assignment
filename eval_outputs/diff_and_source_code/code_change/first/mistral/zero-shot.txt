import { test, expect } from '@playwright/test';

test('should render the Home page', async ({ page }) => {
  await page.goto('/');
  await expect(page).toHaveTitle('Note Taking App');
  await expect(page.locator('h1')).toHaveText('Note Taking App');
  await expect(page.locator('p')).toHaveText('Create and manage notes. Your notes are stored locally.');
  await expect(page.locator('h3')).toHaveText('Your Notes (0)');
  await expect(page.locator('button')).toHaveText('Clear all');
});
