import { test, expect } from '@playwright/test';

test('clear all notes button functionality', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Add some notes to simulate existing notes
  await page.fill('#note-title', 'Test Note 1');
  await page.fill('#note-body', 'This is the body of Test Note 1');
  await page.click('button:has-text("Create Note")');
  await page.fill('#note-title', 'Test Note 2');
  await page.fill('#note-body', 'This is the body of Test Note 2');
  await page.click('button:has-text("Create Note")');

  // Verify the "Clear all" button is visible
  const clearAllButton = page.locator('button:has-text("Clear all")');
  await expect(clearAllButton).toBeVisible();

  // Click the "Clear all" button
  await clearAllButton.click();

  // Verify that all notes are cleared
  await expect(page.locator('text=No notes yet')).toBeVisible();
  await expect(clearAllButton).toBeHidden();
});
