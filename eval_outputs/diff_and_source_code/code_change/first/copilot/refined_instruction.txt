import { test, expect } from '@playwright/test';

test('should display the correct title and description on the home page', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await expect(page.getByRole('heading', { name: 'Note Taking App' })).toBeVisible();
  await expect(page.getByText('Create and manage notes. Your notes are stored locally.')).toBeVisible();
});

test('should display "Your Notes" section with count', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await expect(page.getByRole('heading', { name: /Your Notes \(\d+\)/ })).toBeVisible();
});

test('should show "Clear all" button when there are saved notes', async ({ page }) => {
  await page.goto('http://localhost:5175');
  const clearAllButton = page.getByRole('button', { name: 'Clear all' });
  if (await clearAllButton.isVisible()) {
    await expect(clearAllButton).toBeVisible();
  } else {
    await expect(page.getByText('No notes yet')).toBeVisible();
  }
});

test('should create a new note', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await page.getByRole('textbox', { name: 'Title' }).fill('Test Note');
  await page.getByRole('textbox', { name: 'Body' }).fill('This is a test note body.');
  await page.getByRole('textbox', { name: 'Tags (comma-separated)' }).fill('test, note');
  await page.getByRole('button', { name: 'Create Note' }).click();
  await expect(page.getByText('Test Note')).toBeVisible();
});

test('should clear all notes when "Clear all" button is clicked', async ({ page }) => {
  await page.goto('http://localhost:5175');
  const clearAllButton = page.getByRole('button', { name: 'Clear all' });
  if (await clearAllButton.isVisible()) {
    await clearAllButton.click();
    await expect(page.getByText('No notes yet')).toBeVisible();
  }
});

test('should edit an existing note', async ({ page }) => {
  await page.goto('http://localhost:5175');
  const noteButton = page.getByText(/Note #\d+/);
  if (await noteButton.isVisible()) {
    await noteButton.click();
    await page.getByRole('button', { name: 'Edit' }).click();
    await page.getByRole('textbox', { name: 'Title' }).fill('Updated Note Title');
    await page.getByRole('button', { name: 'Save' }).click();
    await expect(page.getByText('Updated Note Title')).toBeVisible();
  }
});

test('should delete an existing note', async ({ page }) => {
  await page.goto('http://localhost:5175');
  const noteButton = page.getByText(/Note #\d+/);
  if (await noteButton.isVisible()) {
    await noteButton.click();
    await page.getByRole('button', { name: 'Delete' }).click();
    await page.on('dialog', dialog => dialog.accept());
    await expect(noteButton).not.toBeVisible();
  }
});
