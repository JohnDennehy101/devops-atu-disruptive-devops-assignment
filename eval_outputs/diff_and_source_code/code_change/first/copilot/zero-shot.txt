import { test, expect } from '@playwright/test';

test.describe('Home Page', () => {
  test('should display the correct initial UI', async ({ page }) => {
    await page.goto('/');

    // Check for the header
    await expect(page.locator('h1')).toHaveText('Note Taking App');

    // Check for the description paragraph
    await expect(page.locator('p.text-gray-500')).toHaveText(
      'Create and manage notes. Your notes are stored locally.'
    );

    // Check for the "Your Notes" section
    await expect(page.locator('h3')).toHaveText('Your Notes (0)');
    await expect(page.locator('p.text-gray-500')).toHaveText('No notes yet');
  });

  test('should create a new note', async ({ page }) => {
    await page.goto('/');

    // Fill out the form
    await page.fill('#note-title', 'Test Note');
    await page.fill('#note-body', 'This is a test note.');
    await page.fill('#note-tags', 'test, playwright');

    // Submit the form
    await page.click('button:has-text("Create Note")');

    // Verify the note appears in the list
    await expect(page.locator('h3')).toHaveText('Your Notes (1)');
    await expect(page.locator('button:has-text("Note #1")')).toBeVisible();
  });

  test('should clear all notes', async ({ page }) => {
    await page.goto('/');

    // Create a note
    await page.fill('#note-title', 'Test Note');
    await page.fill('#note-body', 'This is a test note.');
    await page.fill('#note-tags', 'test, playwright');
    await page.click('button:has-text("Create Note")');

    // Verify the "Clear all" button is visible
    await expect(page.locator('button:has-text("Clear all")')).toBeVisible();

    // Click the "Clear all" button
    await page.click('button:has-text("Clear all")');

    // Verify the notes list is empty
    await expect(page.locator('h3')).toHaveText('Your Notes (0)');
    await expect(page.locator('p.text-gray-500')).toHaveText('No notes yet');
  });
});
