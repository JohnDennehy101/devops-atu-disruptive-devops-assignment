import { test, expect } from '@playwright/test';

test('shows intro text and clears all saved notes from the sidebar', async ({ page }) => {
  await page.goto('http://localhost:5175');

  await page.evaluate(() => {
    localStorage.clear();
  });
  await page.reload();

  await expect(page.getByText('Create and manage notes. Your notes are stored locally.')).toBeVisible();
  await expect(page.getByText('Your Notes (0)')).toBeVisible();
  await expect(page.getByRole('button', { name: 'Clear all' })).toHaveCount(0);

  await page.getByLabel('Title').fill('First note');
  await page.getByLabel('Body').fill('Body for first note');
  await page.getByLabel('Tags (comma-separated)').fill('one, two');
  await page.getByRole('button', { name: 'Create Note' }).click();

  await expect(page.getByText(/Your Notes \(1\)/)).toBeVisible();
  await expect(page.getByRole('button', { name: 'Clear all' })).toBeVisible();

  await page.getByLabel('Title').fill('Second note');
  await page.getByLabel('Body').fill('Body for second note');
  await page.getByLabel('Tags (comma-separated)').fill('three');
  await page.getByRole('button', { name: 'Create Note' }).click();

  await expect(page.getByText(/Your Notes \(2\)/)).toBeVisible();

  const noteButtons = page.locator('button', { hasText: /^Note #\d+$/ });
  await expect(noteButtons).toHaveCount(2);

  await noteButtons.first().click();
  await expect(page.getByText('Note Details')).toBeVisible();

  await page.getByRole('button', { name: 'Clear all' }).click();

  await expect(page.getByText('Your Notes (0)')).toBeVisible();
  await expect(page.getByText('No notes yet')).toBeVisible();
  await expect(page.getByText('Note Details')).toHaveCount(0);
  await expect(page.getByLabel('Title')).toHaveValue('');
  await expect(page.getByLabel('Body')).toHaveValue('');
  await expect(page.getByLabel('Tags (comma-separated)')).toHaveValue('');
  await expect(page.getByRole('button', { name: 'Clear all' })).toHaveCount(0);

  await expect.poll(async () => {
    return await page.evaluate(() => localStorage.getItem('noteIds'));
  }).toBe('[]');
});
