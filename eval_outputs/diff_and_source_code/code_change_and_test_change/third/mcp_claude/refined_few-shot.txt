import { test, expect } from '@playwright/test';

test('Copy button copies note body to clipboard', async ({ page, context }) => {
  await context.grantPermissions(['clipboard-read', 'clipboard-write']);

  await page.goto('http://localhost:5175');

  const bodyText = 'Text to copy from note';
  await page.getByLabel('Title').fill('Copy test');
  await page.getByLabel('Body').fill(bodyText);
  await page.getByLabel(/Tags/).fill('tag1');
  await page.getByRole('button', { name: 'Create Note' }).click();

  await page.getByRole('button', { name: /Note #\d+/ }).click();
  await expect(page.getByRole('heading', { name: 'Note Details' })).toBeVisible();

  await page.getByRole('button', { name: 'Copy' }).click();

  const clipboardText = await page.evaluate(() => navigator.clipboard.readText());
  expect(clipboardText).toBe(bodyText);
});

test('Copy button is not visible when no note is selected', async ({ page }) => {
  await page.goto('http://localhost:5175');

  await expect(page.getByRole('button', { name: 'Copy' })).not.toBeVisible();
});

test('Copy button copies the correct body when switching notes', async ({ page, context }) => {
  await context.grantPermissions(['clipboard-read', 'clipboard-write']);

  await page.goto('http://localhost:5175');

  const firstBody = 'First note body content';
  const secondBody = 'Second note body content';

  await page.getByLabel('Title').fill('First note');
  await page.getByLabel('Body').fill(firstBody);
  await page.getByLabel(/Tags/).fill('a');
  await page.getByRole('button', { name: 'Create Note' }).click();

  await page.getByLabel('Title').fill('Second note');
  await page.getByLabel('Body').fill(secondBody);
  await page.getByLabel(/Tags/).fill('b');
  await page.getByRole('button', { name: 'Create Note' }).click();

  const noteButtons = page.getByRole('button', { name: /Note #\d+/ });
  const count = await noteButtons.count();

  // Click the last note (second created)
  await noteButtons.nth(count - 1).click();
  await expect(page.getByRole('heading', { name: 'Note Details' })).toBeVisible();
  await page.getByRole('button', { name: 'Copy' }).click();

  let clipboardText = await page.evaluate(() => navigator.clipboard.readText());
  expect(clipboardText).toBe(secondBody);

  // Click the second-to-last note (first created)
  await noteButtons.nth(count - 2).click();
  await expect(page.getByRole('heading', { name: 'Note Details' })).toBeVisible();
  await page.getByRole('button', { name: 'Copy' }).click();

  clipboardText = await page.evaluate(() => navigator.clipboard.readText());
  expect(clipboardText).toBe(firstBody);
});
