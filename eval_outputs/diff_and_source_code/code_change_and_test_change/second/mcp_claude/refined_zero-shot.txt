import { test, expect } from '@playwright/test';

test.describe('Body character count', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
  });

  test('shows body character count and updates when typing', async ({ page }) => {
    await expect(page.getByText('0 / 5000')).toBeVisible();
    await page.getByLabel('Body').fill('Hello');
    await expect(page.getByText('5 / 5000')).toBeVisible();
    await page.getByLabel('Body').fill('Hello world');
    await expect(page.getByText('11 / 5000')).toBeVisible();
  });

  test('shows zero count when body is cleared', async ({ page }) => {
    await page.getByLabel('Body').fill('Some text');
    await expect(page.getByText('9 / 5000')).toBeVisible();
    await page.getByLabel('Body').fill('');
    await expect(page.getByText('0 / 5000')).toBeVisible();
  });

  test('character count is visible on initial page load', async ({ page }) => {
    const charCount = page.getByText('0 / 5000');
    await expect(charCount).toBeVisible();
  });

  test('counts multi-byte and special characters correctly', async ({ page }) => {
    await page.getByLabel('Body').fill('ab cd!');
    await expect(page.getByText('6 / 5000')).toBeVisible();
  });

  test('updates count for longer text', async ({ page }) => {
    const longText = 'a'.repeat(100);
    await page.getByLabel('Body').fill(longText);
    await expect(page.getByText('100 / 5000')).toBeVisible();
  });
});
