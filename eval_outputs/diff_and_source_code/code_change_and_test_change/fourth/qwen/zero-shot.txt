import { test, expect } from '@playwright/test';

test('filter by tag shows only matching notes', async ({ page }) => {
  // Navigate to the home page
  await page.goto('/');

  // Create a work note
  await page.fill('label[for="note-title"]', 'Work note');
  await page.fill('label[for="note-body"]', 'Body one');
  await page.fill('label[for="note-tags"]', 'work, urgent');
  await page.click('button[type="submit"]');

  // Create a personal note
  await page.fill('label[for="note-title"]', 'Personal note');
  await page.fill('label[for="note-body"]', 'Body two');
  await page.fill('label[for="note-tags"]', 'personal');
  await page.click('button[type="submit"]');

  // Verify that both notes are displayed
  await expect(page.locator('text=Your Notes (2)')).toBeVisible();

  // Filter by 'work' tag
  await page.fill('input[placeholder="Filter by tag"]', 'work');
  await expect(page.locator('text=Your Notes (1)')).toBeVisible();
  await expect(page.locator('button:has-text("Note #1")')).toBeVisible();
  await expect(page.locator('button:has-text("Note #2")')).not.toBeVisible();

  // Filter by a non-existent tag
  await page.fill('input[placeholder="Filter by tag"]', 'nonexistent');
  await expect(page.locator('text=No notes match the filter')).toBeVisible();
  await expect(page.locator('text=Your Notes (0)')).toBeVisible();

  // Clear the filter
  await page.fill('input[placeholder="Filter by tag"]', '');
  await expect(page.locator('text=Your Notes (2)')).toBeVisible();
});
