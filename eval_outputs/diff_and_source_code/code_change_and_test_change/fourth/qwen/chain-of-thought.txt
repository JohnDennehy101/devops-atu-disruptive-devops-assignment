import { test, expect } from '@playwright/test';

test('filter by tag shows only matching notes', async ({ page }) => {
  // Step 1: Navigate to the Home page
  await page.goto('/');

  // Step 2: Create a new note with the title "Work note", body "Body one", and tags "work, urgent"
  await page.fill('label[for="note-title"]', 'Work note');
  await page.fill('label[for="note-body"]', 'Body one');
  await page.fill('label[for="note-tags"]', 'work, urgent');
  await page.click('button[name="Create Note"]');

  // Step 3: Create another note with the title "Personal note", body "Body two", and tags "personal"
  await page.fill('label[for="note-title"]', 'Personal note');
  await page.fill('label[for="note-body"]', 'Body two');
  await page.fill('label[for="note-tags"]', 'personal');
  await page.click('button[name="Create Note"]');

  // Step 4: Verify that the total number of notes displayed is 2
  await expect(page.locator('text=Your Notes (2)')).toBeVisible();

  // Step 5: Filter notes by the tag "work"
  await page.fill('input[placeholder="Filter by tag"]', 'work');

  // Step 6: Verify that the total number of notes displayed is 1
  await expect(page.locator('text=Your Notes (1)')).toBeVisible();

  // Step 7: Verify that the note with the title "Work note" is visible
  await expect(page.locator('button[name="Note #1"]')).toBeVisible();

  // Step 8: Verify that the note with the title "Personal note" is not visible
  await expect(page.locator('button[name="Note #2"]')).not.toBeVisible();

  // Step 9: Filter notes by a non-existent tag "nonexistent"
  await page.fill('input[placeholder="Filter by tag"]', 'nonexistent');

  // Step 10: Verify that the message "No notes match the filter" is displayed
  await expect(page.locator('text=No notes match the filter')).toBeVisible();

  // Step 11: Verify that the total number of notes displayed is 0
  await expect(page.locator('text=Your Notes (0)')).toBeVisible();

  // Step 12: Clear the filter
  await page.fill('input[placeholder="Filter by tag"]', '');

  // Step 13: Verify that the total number of notes displayed is 2
  await expect(page.locator('text=Your Notes (2)')).toBeVisible();
});
