import { test, expect } from "@playwright/test"

test.describe("Note Taking App - Clear All Notes", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("can create multiple notes and see clear all button", async ({ page }) => {
    // Create multiple notes
    for (let i = 1; i <= 5; i++) {
      await page.getByLabel("Title").fill(`Note ${i}`)
      await page.getByLabel("Body").fill(`Content for note ${i}`)
      await page.getByLabel(/Tags/).fill(`tag${i}`)
      await page.getByRole("button", { name: "Create Note" }).click()
      await page.reload()
    }

    // Check clear all button is visible
    await expect(page.getByRole("button", { name: "Clear all" })).toBeVisible()
  })

  test("can clear all notes and see no notes message", async ({ page }) => {
    // Create a note
    await page.getByLabel("Title").fill("Note to delete")
    await page.getByLabel("Body").fill("Content to delete")
    await page.getByLabel(/Tags/).fill("tag1")
    await page.getByRole("button", { name: "Create Note" }).click()
    await page.reload()

    // Click clear all button
    await page.getByRole("button", { name: "Clear all" }).click()

    // Check no notes message is visible
    await expect(page.getByText("No notes yet")).toBeVisible()
  })
})
