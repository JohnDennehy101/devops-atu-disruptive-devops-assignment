import { test, expect } from "@playwright/test";

test.describe("Note Taking App - Clear All Notes", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175");
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test("shows 'Clear all' button when notes exist and clears all notes", async ({ page }) => {
    // Create the first note
    await page.getByLabel("Title").fill("Note 1");
    await page.getByLabel("Body").fill("Body of Note 1");
    await page.getByLabel(/Tags/).fill("tag1");
    await page.getByRole("button", { name: "Create Note" }).click();

    // Create the second note
    await page.getByLabel("Title").fill("Note 2");
    await page.getByLabel("Body").fill("Body of Note 2");
    await page.getByLabel(/Tags/).fill("tag2");
    await page.getByRole("button", { name: "Create Note" }).click();

    // Verify "Clear all" button is visible
    await expect(page.getByRole("button", { name: "Clear all" })).toBeVisible();

    // Click "Clear all" button
    await page.getByRole("button", { name: "Clear all" }).click();

    // Verify all notes are cleared
    await expect(page.getByText("No notes yet")).toBeVisible();
    await expect(page.getByText("Your Notes (0)")).toBeVisible();
  });

  test("does not show 'Clear all' button when no notes exist", async ({ page }) => {
    // Verify "Clear all" button is not visible
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible();
  });
});
