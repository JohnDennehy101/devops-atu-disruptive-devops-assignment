import { test, expect } from "@playwright/test"

test.describe("Clear All Notes", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("does not show Clear all button when there are no notes", async ({
    page,
  }) => {
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(
      page.getByRole("button", { name: "Clear all" }),
    ).not.toBeVisible()
  })

  test("shows Clear all button after creating a note", async ({ page }) => {
    await page.getByLabel("Title").fill("Test note")
    await page.getByLabel("Body").fill("Test body")
    await page.getByLabel(/Tags/).fill("demo")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(
      page.getByRole("button", { name: "Clear all" }),
    ).toBeVisible()
  })

  test("clears all notes when Clear all is clicked", async ({ page }) => {
    await page.getByLabel("Title").fill("Note one")
    await page.getByLabel("Body").fill("Body one")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByLabel("Title").fill("Note two")
    await page.getByLabel("Body").fill("Body two")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (2)")).toBeVisible()

    await page.getByRole("button", { name: "Clear all" }).click()

    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByText("No notes yet")).toBeVisible()
    await expect(
      page.getByRole("button", { name: "Clear all" }),
    ).not.toBeVisible()
    await expect(
      page.getByRole("button", { name: /Note #\d+/ }),
    ).not.toBeVisible()
  })

  test("clears localStorage when Clear all is clicked", async ({ page }) => {
    await page.getByLabel("Title").fill("Persistent note")
    await page.getByLabel("Body").fill("Should be cleared")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByRole("button", { name: "Clear all" }).click()

    const storedIds = await page.evaluate(() =>
      JSON.parse(localStorage.getItem("noteIds") || "[]"),
    )
    expect(storedIds).toEqual([])
  })

  test("resets form and selection state after Clear all", async ({ page }) => {
    await page.getByLabel("Title").fill("Editable note")
    await page.getByLabel("Body").fill("Some content")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByRole("button", { name: /Note #\d+/ }).click()
    await expect(
      page.getByRole("heading", { name: "Note Details" }),
    ).toBeVisible()

    await page.getByRole("button", { name: "Clear all" }).click()

    await expect(
      page.getByRole("heading", { name: "Note Details" }),
    ).not.toBeVisible()
    await expect(
      page.getByRole("heading", { name: /Create New Note/ }),
    ).toBeVisible()
    await expect(page.getByLabel("Title")).toHaveValue("")
    await expect(page.getByLabel("Body")).toHaveValue("")
  })

  test("shows descriptive subtitle text", async ({ page }) => {
    await expect(
      page.getByText("Create and manage notes. Your notes are stored locally."),
    ).toBeVisible()
  })
})
