import { test, expect } from "@playwright/test"

test.describe("Clear All Notes", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("clear all button is hidden when no notes exist", async ({ page }) => {
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByRole("button", { name: "Clear all" })).toBeHidden()
  })

  test("clear all button appears after creating a note", async ({ page }) => {
    await page.getByLabel("Title").fill("Test note")
    await page.getByLabel("Body").fill("Test body")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByRole("button", { name: "Clear all" })).toBeVisible()
  })

  test("clicking clear all removes all notes from the sidebar", async ({ page }) => {
    await page.getByLabel("Title").fill("Note one")
    await page.getByLabel("Body").fill("Body one")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByLabel("Title").fill("Note two")
    await page.getByLabel("Body").fill("Body two")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (2)")).toBeVisible()

    await page.getByRole("button", { name: "Clear all" }).click()

    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByText("No notes yet")).toBeVisible()
    await expect(page.getByRole("button", { name: "Clear all" })).toBeHidden()
  })

  test("clear all clears localStorage noteIds", async ({ page }) => {
    await page.getByLabel("Title").fill("Persisted note")
    await page.getByLabel("Body").fill("Persisted body")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByRole("button", { name: "Clear all" }).click()

    const storedIds = await page.evaluate(() => localStorage.getItem("noteIds"))
    expect(storedIds).toBe("[]")
  })

  test("shows subtitle text about local storage", async ({ page }) => {
    await expect(
      page.getByText("Create and manage notes. Your notes are stored locally.")
    ).toBeVisible()
  })
})
