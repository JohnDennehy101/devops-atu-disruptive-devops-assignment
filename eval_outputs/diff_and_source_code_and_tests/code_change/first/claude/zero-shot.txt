import { test, expect } from "@playwright/test"

test.describe("Clear All Notes", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("shows subtitle text about local storage", async ({ page }) => {
    await expect(
      page.getByText("Create and manage notes. Your notes are stored locally."),
    ).toBeVisible()
  })

  test("clear all button is not visible when there are no notes", async ({ page }) => {
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible()
  })

  test("clear all button appears after creating a note", async ({ page }) => {
    await page.getByLabel("Title").fill("Test note")
    await page.getByLabel("Body").fill("Test body")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByRole("button", { name: "Clear all" })).toBeVisible()
  })

  test("clear all removes all notes from the sidebar", async ({ page }) => {
    await page.getByLabel("Title").fill("Note one")
    await page.getByLabel("Body").fill("Body one")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByLabel("Title").fill("Note two")
    await page.getByLabel("Body").fill("Body two")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (2)")).toBeVisible()

    await page.getByRole("button", { name: "Clear all" }).click()

    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByText("No notes yet")).toBeVisible()
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible()
  })

  test("clear all resets the form and clears selected note", async ({ page }) => {
    await page.getByLabel("Title").fill("Selected note")
    await page.getByLabel("Body").fill("Selected body")
    await page.getByLabel(/Tags/).fill("tag1")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByRole("button", { name: /Note #\d+/ }).click()
    await expect(page.getByRole("heading", { name: "Note Details" })).toBeVisible()

    await page.getByRole("button", { name: "Clear all" }).click()

    await expect(page.getByRole("heading", { name: "Note Details" })).not.toBeVisible()
    await expect(page.getByLabel("Title")).toHaveValue("")
    await expect(page.getByLabel("Body")).toHaveValue("")
    await expect(page.getByLabel(/Tags/)).toHaveValue("")
  })

  test("clear all clears localStorage noteIds", async ({ page }) => {
    await page.getByLabel("Title").fill("Persist note")
    await page.getByLabel("Body").fill("Persist body")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByRole("button", { name: "Clear all" }).click()

    const storedIds = await page.evaluate(() => localStorage.getItem("noteIds"))
    expect(storedIds).toBe("[]")
  })

  test("clear all during edit mode resets to create mode", async ({ page }) => {
    await page.getByLabel("Title").fill("Edit note")
    await page.getByLabel("Body").fill("Edit body")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByRole("button", { name: /Note #\d+/ }).click()
    await page.getByRole("link", { name: "Edit" }).or(page.getByText("Edit")).click()

    await page.getByRole("button", { name: "Clear all" }).click()

    await expect(page.getByRole("heading", { name: /Create New Note/ })).toBeVisible()
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
  })
})
