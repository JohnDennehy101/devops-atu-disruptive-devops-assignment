import { test, expect } from "@playwright/test"

test.describe("Note Taking App - Character Count", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("displays character count in the body field", async ({ page }) => {
    await page.getByLabel("Body").fill("a".repeat(5000))

    await expect(page.getByText("5000 / 5000")).toBeVisible()
  })
})
