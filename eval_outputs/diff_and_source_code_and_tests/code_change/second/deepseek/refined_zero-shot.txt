import { test, expect } from '@playwright/test';

test.describe("Note Taking App", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175");
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test("displays character count in the form", async ({ page }) => {
    await page.getByLabel("Title").fill("Test Title");
    await page.getByLabel("Body").fill("Test Body");
    await page.getByLabel(/Tags/).fill("test, tag");

    const characterCount = await page.locator(".text-xs.text-gray-500.dark\\:text-gray-400.mt-1").innerText();
    expect(characterCount).toBe("7 / 5000");
  });
});
