import { test, expect } from "@playwright/test"

test.describe("Note Taking App", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  // Existing tests...

  test("character count limit is displayed correctly", async ({ page }) => {
    // Fill the body with 5000 characters
    const longText = "a".repeat(5000)
    await page.getByLabel("Body").fill(longText)

    // Check if the character count is displayed correctly
    const charCountElement = await page.getByText("5000 / 5000")
    await expect(charCountElement).toBeVisible()
  })
})
