import { test, expect } from '@playwright/test';

test.describe("Note Taking App", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175");
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test("can create a note and verify character count", async ({ page }) => {
    await page.getByLabel("Title").fill("Test Note");
    await page.getByLabel("Body").fill("This is a test body.");
    await page.getByLabel(/Tags/).fill("test, e2e");

    await page.getByRole("button", { name: "Create Note" }).click();

    await expect(page.getByText("Your Notes (1)")).toBeVisible();

    const charCount = await page.locator('p.text-xs.text-gray-500.dark:text-gray-400.mt-1').textContent();
    await expect(charCount).toBe("18 / 5000");

    await page.getByRole("button", { name: /Note #\d+/ }).click();

    const openedCharCount = await page.locator('p.text-xs.text-gray-500.dark:text-gray-400.mt-1').textContent();
    await expect(openedCharCount).toBe("18 / 5000");
  });
});
