import { test, expect } from '@playwright/test';

test.describe('Home Page', () => {
  test('displays formBody length', async ({ page }) => {
    await page.goto('/');
    await page.locator('textarea[id="note-body"]').fill('This is a test note.');
    await expect(page.locator('.text-xs')).toHaveText(/14\/5000/);
  });
});
