import { test, expect } from '@playwright/test';

test.describe('Home Page - Form Body Length', () => {
  test('verify form body length display', async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.getByLabel('Body').fill('This is a test note body with some text.');
    await expect(page.getByText(/0 \/ 5000/)).toBeVisible();
    await page.getByLabel('Body').fill('This is a test note body with some text. This is a test note body with some text. This is a test note body with some text. This is a test note body with some text. This is a test note body with some text.');
    await expect(page.getByText(/141 \/ 5000/)).toBeVisible();
  });
});
