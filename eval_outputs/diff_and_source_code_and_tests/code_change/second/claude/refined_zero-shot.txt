import { test, expect } from '@playwright/test';

test.describe("Character counter for note body", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175");
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test("displays character counter showing 0 / 5000 when body is empty", async ({ page }) => {
    await expect(page.getByText("0 / 5000")).toBeVisible();
  });

  test("updates character counter as user types in body", async ({ page }) => {
    await page.getByLabel("Body").fill("Hello");
    await expect(page.getByText("5 / 5000")).toBeVisible();
  });

  test("shows correct count for longer text", async ({ page }) => {
    const text = "A".repeat(100);
    await page.getByLabel("Body").fill(text);
    await expect(page.getByText("100 / 5000")).toBeVisible();
  });

  test("counter resets after creating a note", async ({ page }) => {
    await page.getByLabel("Title").fill("Counter test");
    await page.getByLabel("Body").fill("Some body text");
    await expect(page.getByText("14 / 5000")).toBeVisible();

    await page.getByRole("button", { name: "Create Note" }).click();
    await expect(page.getByText("0 / 5000")).toBeVisible();
  });
});
