import { test, expect } from '@playwright/test';

test.describe('Body character counter', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('displays character count that updates as user types', async ({ page }) => {
    await expect(page.getByText('0 / 5000')).toBeVisible();

    await page.getByLabel('Body').fill('Hello');
    await expect(page.getByText('5 / 5000')).toBeVisible();

    await page.getByLabel('Body').fill('Hello, world!');
    await expect(page.getByText('13 / 5000')).toBeVisible();
  });

  test('resets character count after creating a note', async ({ page }) => {
    await page.getByLabel('Title').fill('Counter test');
    await page.getByLabel('Body').fill('Some body text');
    await expect(page.getByText('14 / 5000')).toBeVisible();

    await page.getByRole('button', { name: 'Create Note' }).click();

    await expect(page.getByText('0 / 5000')).toBeVisible();
  });
});
