import { test, expect } from "@playwright/test"

test.describe("Body character counter", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("displays 0 / 5000 when body is empty", async ({ page }) => {
    await expect(page.getByText("0 / 5000")).toBeVisible()
  })

  test("updates character count as user types in body", async ({ page }) => {
    await page.getByLabel("Body").fill("Hello world")
    await expect(page.getByText("11 / 5000")).toBeVisible()
  })

  test("shows correct count for longer text", async ({ page }) => {
    const text = "a".repeat(100)
    await page.getByLabel("Body").fill(text)
    await expect(page.getByText("100 / 5000")).toBeVisible()
  })

  test("resets character count after creating a note", async ({ page }) => {
    await page.getByLabel("Title").fill("Test note")
    await page.getByLabel("Body").fill("Some content")
    await expect(page.getByText("12 / 5000")).toBeVisible()

    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toBeVisible()
    await expect(page.getByText("0 / 5000")).toBeVisible()
  })

  test("character count updates when editing an existing note", async ({ page }) => {
    await page.getByLabel("Title").fill("Edit test")
    await page.getByLabel("Body").fill("Original body")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByRole("button", { name: /Note #\d+/ }).click()
    await expect(page.getByText("Note Details")).toBeVisible()

    await page.getByRole("link", { name: "Edit" }).or(page.getByText("Edit")).click()
    await expect(page.getByText("13 / 5000")).toBeVisible()

    await page.getByLabel("Body").fill("Updated body content here")
    await expect(page.getByText("25 / 5000")).toBeVisible()
  })
})
