import { test, expect } from '@playwright/test';

test.describe('Body character counter', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('displays initial character count of 0 / 5000 when body is empty', async ({ page }) => {
    await expect(page.getByText('0 / 5000')).toBeVisible();
  });

  test('updates character count as user types in the body field', async ({ page }) => {
    await page.getByLabel('Body').fill('Hello world');
    await expect(page.getByText('11 / 5000')).toBeVisible();
  });

  test('updates character count for longer text', async ({ page }) => {
    const longText = 'A'.repeat(100);
    await page.getByLabel('Body').fill(longText);
    await expect(page.getByText('100 / 5000')).toBeVisible();
  });

  test('resets character count to 0 after creating a note', async ({ page }) => {
    await page.getByLabel('Title').fill('Counter test');
    await page.getByLabel('Body').fill('Some body text');
    await expect(page.getByText('14 / 5000')).toBeVisible();

    await page.getByRole('button', { name: 'Create Note' }).click();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toBeVisible();
    await expect(page.getByText('0 / 5000')).toBeVisible();
  });

  test('character counter is positioned below the body textarea', async ({ page }) => {
    const bodyTextarea = page.getByLabel('Body');
    const counter = page.getByText('0 / 5000');

    await expect(bodyTextarea).toBeVisible();
    await expect(counter).toBeVisible();

    const textareaBox = await bodyTextarea.boundingBox();
    const counterBox = await counter.boundingBox();

    expect(textareaBox).not.toBeNull();
    expect(counterBox).not.toBeNull();
    expect(counterBox!.y).toBeGreaterThan(textareaBox!.y);
  });
});
