import { test, expect } from '@playwright/test';

test.describe('Note Taking App - Character Counter', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('should display character counter for note body input', async ({ page }) => {
    const bodyTextarea = page.getByLabel('Body');
    await expect(bodyTextarea).toBeVisible();

    const counter = page.getByText('0 / 5000');
    await expect(counter).toBeVisible();
  });

  test('should update character counter when typing in body field', async ({ page }) => {
    const bodyTextarea = page.getByLabel('Body');
    const testText = 'Hello, this is a test note.';

    await bodyTextarea.fill(testText);
    
    const counter = page.getByText(`${testText.length} / 5000`);
    await expect(counter).toBeVisible();
  });

  test('should show correct counter after creating and editing a note', async ({ page }) => {
    await page.getByLabel('Title').fill('Test Note');
    await page.getByLabel('Body').fill('Initial body content');
    await page.getByLabel(/Tags/).fill('test');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByRole('button', { name: /Note #\d+/ }).click();
    await page.getByRole('button', { name: 'Edit' }).click();

    const bodyTextarea = page.getByLabel('Body');
    const initialCounter = page.getByText('20 / 5000');
    await expect(initialCounter).toBeVisible();

    const updatedText = 'This is updated content for the note body.';
    await bodyTextarea.fill(updatedText);
    
    const updatedCounter = page.getByText(`${updatedText.length} / 5000`);
    await expect(updatedCounter).toBeVisible();
  });

  test('should reset character counter when canceling edit', async ({ page }) => {
    await page.getByLabel('Title').fill('Test Note');
    await page.getByLabel('Body').fill('Some content');
    await page.getByLabel(/Tags/).fill('test');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByRole('button', { name: /Note #\d+/ }).click();
    await page.getByRole('button', { name: 'Edit' }).click();

    const bodyTextarea = page.getByLabel('Body');
    await bodyTextarea.fill('Much longer content that exceeds initial length');
    
    const counterAfterEdit = page.getByText('49 / 5000');
    await expect(counterAfterEdit).toBeVisible();

    await page.getByRole('button', { name: 'Cancel' }).click();
    
    const counterAfterCancel = page.getByText('0 / 5000');
    await expect(counterAfterCancel).toBeVisible();
  });

  test('should show correct counter when body is empty', async ({ page }) => {
    const bodyTextarea = page.getByLabel('Body');
    await bodyTextarea.fill('');
    
    const counter = page.getByText('0 / 5000');
    await expect(counter).toBeVisible();
  });
});
