import { test, expect } from "@playwright/test";

test.describe("Note Taking App - Character Counter", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175");
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test("displays character count for note body", async ({ page }) => {
    await page.getByLabel("Body").fill("This is a test note body.");
    await expect(page.getByText("27 / 5000")).toBeVisible();

    await page.getByLabel("Body").fill("Updated body content.");
    await expect(page.getByText("23 / 5000")).toBeVisible();

    await page.getByLabel("Body").fill("");
    await expect(page.getByText("0 / 5000")).toBeVisible();
  });
});
