import { test, expect } from '@playwright/test';

test.describe('Note Taking App - Character Counter', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('displays character count for note body', async ({ page }) => {
    // Verify the character counter is initially visible and shows "0 / 5000"
    const charCounter = page.locator('p.text-xs.text-gray-500');
    await expect(charCounter).toBeVisible();
    await expect(charCounter).toHaveText('0 / 5000');

    // Type into the "Body" textarea and verify the character count updates
    const bodyTextarea = page.getByLabel('Body');
    await bodyTextarea.fill('This is a test note body.');
    await expect(charCounter).toHaveText('25 / 5000');

    // Add more text and verify the updated character count
    await bodyTextarea.type(' Adding more text.');
    await expect(charCounter).toHaveText('43 / 5000');

    // Clear the textarea and verify the character count resets
    await bodyTextarea.fill('');
    await expect(charCounter).toHaveText('0 / 5000');
  });
});
