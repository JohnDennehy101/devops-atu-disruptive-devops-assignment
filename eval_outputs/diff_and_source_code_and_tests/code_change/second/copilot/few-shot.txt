import { test, expect } from "@playwright/test"

test.describe("Note Taking App - Character Counter", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("displays character count for the note body", async ({ page }) => {
    // Verify the initial character count is 0
    await expect(page.getByText("0 / 5000")).toBeVisible()

    // Fill the body with some text
    const bodyText = "This is a test note body."
    await page.getByLabel("Body").fill(bodyText)

    // Verify the character count updates correctly
    await expect(page.getByText(`${bodyText.length} / 5000`)).toBeVisible()

    // Add more text to the body
    const additionalText = " Adding more content."
    await page.getByLabel("Body").fill(bodyText + additionalText)

    // Verify the updated character count
    await expect(
      page.getByText(`${(bodyText + additionalText).length} / 5000`),
    ).toBeVisible()
  })
})
