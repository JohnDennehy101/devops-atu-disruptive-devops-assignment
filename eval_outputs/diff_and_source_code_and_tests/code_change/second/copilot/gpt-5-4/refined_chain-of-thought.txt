import { test, expect } from '@playwright/test';

test.describe('Note Taking App character counter', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('shows live body character count and resets after note creation', async ({ page }) => {
    const bodyField = page.getByLabel('Body');

    await expect(page.getByText('0 / 5000')).toBeVisible();

    await bodyField.fill('Hello');
    await expect(page.getByText('5 / 5000')).toBeVisible();

    await bodyField.fill('Hello world');
    await expect(page.getByText('11 / 5000')).toBeVisible();

    await page.getByLabel('Title').fill('Counter test');
    await page.getByLabel(/Tags/).fill('demo');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await expect(bodyField).toHaveValue('');
    await expect(page.getByText('0 / 5000')).toBeVisible();
  });

  test('shows body character count when entering edit mode and updates as body changes', async ({ page }) => {
    await page.getByLabel('Title').fill('Editable note');
    await page.getByLabel('Body').fill('Edit me');
    await page.getByLabel(/Tags/).fill('tag1');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByRole('button', { name: /Note #\d+/ }).click();
    await page.getByRole('button', { name: 'Edit' }).click();

    await expect(page.getByRole('heading', { name: 'Edit Note' })).toBeVisible();
    await expect(page.getByText('7 / 5000')).toBeVisible();

    await page.getByLabel('Body').fill('Edited body text');
    await expect(page.getByText('16 / 5000')).toBeVisible();
  });
});
