import { test, expect } from '@playwright/test';

test.describe('Note Taking App', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('shows body character count and updates as user types', async ({ page }) => {
    const bodyField = page.getByLabel('Body');

    await expect(page.getByText('0 / 5000')).toBeVisible();

    await bodyField.fill('Hello world');
    await expect(page.getByText('11 / 5000')).toBeVisible();

    await bodyField.fill('abc');
    await expect(page.getByText('3 / 5000')).toBeVisible();
  });

  test('resets body character count after creating a note', async ({ page }) => {
    await page.getByLabel('Title').fill('Count reset test');
    await page.getByLabel('Body').fill('Some body text');
    await expect(page.getByText('14 / 5000')).toBeVisible();

    await page.getByRole('button', { name: 'Create Note' }).click();

    await expect(page.getByText('0 / 5000')).toBeVisible();
  });
});
