import { test, expect } from "@playwright/test"

test("shows live body character count out of 5000", async ({ page }) => {
  await page.goto("http://localhost:5175")
  await page.evaluate(() => localStorage.clear())
  await page.reload()

  await expect(page.getByText("0 / 5000")).toBeVisible()

  await page.getByLabel("Body").fill("Hello world")
  await expect(page.getByText("11 / 5000")).toBeVisible()

  await page.getByLabel("Body").fill("abc")
  await expect(page.getByText("3 / 5000")).toBeVisible()
})
