import { test, expect } from '@playwright/test';

test.describe("Note Taking App", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175");
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test("displays character count for note body", async ({ page }) => {
    await page.getByLabel("Body").fill("This is a test note body.");
    await expect(page.locator("p.text-xs.text-gray-500")).toHaveText("26 / 5000");

    await page.getByLabel("Body").fill("Updated note body with more text.");
    await expect(page.locator("p.text-xs.text-gray-500")).toHaveText("35 / 5000");

    await page.getByLabel("Body").fill("");
    await expect(page.locator("p.text-xs.text-gray-500")).toHaveText("0 / 5000");
  });
});
