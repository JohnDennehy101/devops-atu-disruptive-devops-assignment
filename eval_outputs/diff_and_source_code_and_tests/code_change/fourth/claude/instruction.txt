import { test, expect } from "@playwright/test"

test.describe("Tag Filter", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("filter input is visible", async ({ page }) => {
    await expect(
      page.getByRole("textbox", { name: /filter by tag/i }),
    ).toBeVisible()
  })

  test("filters notes by tag", async ({ page }) => {
    // Create two notes with different tags
    await page.getByLabel("Title").fill("Note about cooking")
    await page.getByLabel("Body").fill("Recipe content")
    await page.getByLabel(/Tags/).fill("cooking, food")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByLabel("Title").fill("Note about coding")
    await page.getByLabel("Body").fill("Code content")
    await page.getByLabel(/Tags/).fill("coding, typescript")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (2)")).toBeVisible()

    // Filter by "cooking" — only one note should remain
    await page.getByRole("textbox", { name: /filter by tag/i }).fill("cooking")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #/ })).toHaveCount(1)
  })

  test("shows 'No notes match the filter' when no tags match", async ({
    page,
  }) => {
    await page.getByLabel("Title").fill("Some note")
    await page.getByLabel("Body").fill("Body text")
    await page.getByLabel(/Tags/).fill("music")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByRole("button", { name: /Note #/ })).toBeVisible()

    await page
      .getByRole("textbox", { name: /filter by tag/i })
      .fill("nonexistent")
    await expect(page.getByText("No notes match the filter")).toBeVisible()
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
  })

  test("clearing the filter shows all notes again", async ({ page }) => {
    await page.getByLabel("Title").fill("First note")
    await page.getByLabel("Body").fill("Body one")
    await page.getByLabel(/Tags/).fill("alpha")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByLabel("Title").fill("Second note")
    await page.getByLabel("Body").fill("Body two")
    await page.getByLabel(/Tags/).fill("beta")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (2)")).toBeVisible()

    // Filter then clear
    await page.getByRole("textbox", { name: /filter by tag/i }).fill("alpha")
    await expect(page.getByRole("button", { name: /Note #/ })).toHaveCount(1)

    await page.getByRole("textbox", { name: /filter by tag/i }).clear()
    await expect(page.getByText("Your Notes (2)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #/ })).toHaveCount(2)
  })
})
