test("can filter notes by tag", async ({ page }) => {
  await page.getByLabel("Title").fill("Tag Filter Test")
  await page.getByLabel("Body").fill("Content to verify")
  await page.getByLabel(/Tags/).fill("tag1, tag2")
  await page.getByRole("button", { name: "Create Note" }).click()

  await page.getByLabel("Title").fill("Another Note")
  await page.getByLabel("Body").fill("No tags")
  await page.getByRole("button", { name: "Create Note" }).click()

  await page.getByLabel("Title").fill("Yet Another Note")
  await page.getByLabel("Body").fill("Different tags")
  await page.getByLabel(/Tags/).fill("tag3, tag4")
  await page.getByRole("button", { name: "Create Note" }).click()

  await page.getByLabel("Filter by tag").fill("tag1")
  await expect(page.getByText("Your Notes (1)")).toBeVisible()
  await expect(page.getByText("No notes match the filter")).toBeVisible()

  await page.getByLabel("Filter by tag").fill("tag3")
  await expect(page.getByText("Your Notes (1)")).toBeVisible()
  await expect(page.getByText("No notes match the filter")).toBeVisible()

  await page.getByLabel("Filter by tag").fill("")
  await expect(page.getByText("Your Notes (3)")).toBeVisible()
  await expect(page.getByText("No notes yet")).toBeVisible()
})
