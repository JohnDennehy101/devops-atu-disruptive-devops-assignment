import { test, expect } from "@playwright/test"

test.describe("Note Taking App", () => {
  test("filters saved notes by tag and updates the sidebar count and empty state", async ({ page }) => {
    await page.goto("http://localhost:5175")
    await page.evaluate(() => localStorage.clear())
    await page.reload()

    await page.getByLabel("Title").fill("React note")
    await page.getByLabel("Body").fill("Body for react note")
    await page.getByLabel(/Tags/).fill("react, frontend")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByLabel("Title").fill("Backend note")
    await page.getByLabel("Body").fill("Body for backend note")
    await page.getByLabel(/Tags/).fill("api, server")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (2)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(2)

    await page.getByPlaceholder("Filter by tag").fill("react")

    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(1)
    await expect(page.getByText("No notes match the filter")).toBeHidden()

    await page.getByPlaceholder("Filter by tag").fill("missing-tag")

    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByText("No notes match the filter")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(0)

    await page.getByPlaceholder("Filter by tag").fill("")

    await expect(page.getByText("Your Notes (2)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(2)
  })
})
