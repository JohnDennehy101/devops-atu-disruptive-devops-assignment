import { test, expect } from '@playwright/test';

test.describe('Note Taking App with Tag Filter', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('displays tag filter input and updates note count', async ({ page }) => {
    await expect(page.getByPlaceholder('Filter by tag')).toBeVisible();
    await expect(page.getByText('Your Notes (0)')).toBeVisible();

    await page.getByLabel('Title').fill('Note with tag');
    await page.getByLabel('Body').fill('Body content');
    await page.getByLabel(/Tags/).fill('tag1, tag2');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByPlaceholder('Filter by tag').fill('tag1');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByPlaceholder('Filter by tag').fill('tag3');
    await expect(page.getByText('Your Notes (0)')).toBeVisible();
    await expect(page.getByText('No notes match the filter')).toBeVisible();
  });

  test('filters notes by tag and displays matching notes', async ({ page }) => {
    await page.getByLabel('Title').fill('First note');
    await page.getByLabel('Body').fill('Content for first note');
    await page.getByLabel(/Tags/).fill('tag1');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByLabel('Title').fill('Second note');
    await page.getByLabel('Body').fill('Content for second note');
    await page.getByLabel(/Tags/).fill('tag2');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await expect(page.getByText('Your Notes (2)')).toBeVisible();

    await page.getByPlaceholder('Filter by tag').fill('tag1');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveText(/First note/);

    await page.getByPlaceholder('Filter by tag').fill('tag2');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveText(/Second note/);

    await page.getByPlaceholder('Filter by tag').fill('');
    await expect(page.getByText('Your Notes (2)')).toBeVisible();
  });
});
