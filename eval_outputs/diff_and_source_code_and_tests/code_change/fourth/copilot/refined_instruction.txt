import { test, expect } from '@playwright/test';

test.describe('Note Taking App - Tag Filter', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('displays "No notes yet" when no notes exist', async ({ page }) => {
    await expect(page.getByText('No notes yet')).toBeVisible();
  });

  test('can filter notes by tag', async ({ page }) => {
    // Create first note
    await page.getByLabel('Title').fill('First Note');
    await page.getByLabel('Body').fill('Body of first note');
    await page.getByLabel(/Tags/).fill('work, important');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Create second note
    await page.getByLabel('Title').fill('Second Note');
    await page.getByLabel('Body').fill('Body of second note');
    await page.getByLabel(/Tags/).fill('personal, urgent');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Verify both notes are visible
    await expect(page.getByText('Your Notes (2)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(2);

    // Filter by "work"
    await page.getByRole('textbox', { name: 'Filter by tag' }).fill('work');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(1);
    await expect(page.getByRole('button', { name: /First Note/ })).toBeVisible();

    // Filter by "urgent"
    await page.getByRole('textbox', { name: 'Filter by tag' }).fill('urgent');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(1);
    await expect(page.getByRole('button', { name: /Second Note/ })).toBeVisible();

    // Filter by a non-matching tag
    await page.getByRole('textbox', { name: 'Filter by tag' }).fill('nonexistent');
    await expect(page.getByText('No notes match the filter')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(0);

    // Clear filter
    await page.getByRole('textbox', { name: 'Filter by tag' }).fill('');
    await expect(page.getByText('Your Notes (2)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(2);
  });
});
