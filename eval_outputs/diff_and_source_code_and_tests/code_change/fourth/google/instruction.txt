import { test, expect } from "@playwright/test";

test.describe("Note Taking App", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/");
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test("filter notes by tag", async ({ page }) => {
    // Assuming there are notes with tags to filter
    await page.getByLabel("Title").fill("Note with tag1");
    await page.getByLabel("Body").fill("Body of note 1");
    await page.getByLabel(/Tags/).fill("tag1");
    await page.getByRole("button", { name: "Create Note" }).click();

    await page.getByLabel("Title").fill("Note with tag2");
    await page.getByLabel("Body").fill("Body of note 2");
    await page.getByLabel(/Tags/).fill("tag2");
    await page.getByRole("button", { name: "Create Note" }).click();

    await page.getByLabel("Filter by tag").fill("tag1");

    await expect(page.getByRole("button", { name: /Note #\d+/ })).toBeVisible();
    await expect(page.getByRole("button", { name: /Note #\d+/ })).not().toHaveText("Note #2");
    await expect(page.getByText("Your Notes (1)")).toBeVisible();
  });
});
