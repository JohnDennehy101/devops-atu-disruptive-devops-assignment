import { test, expect } from "@playwright/test"

test.describe("Note Taking App - Filtering", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("can filter notes by tag", async ({ page }) => {
    // Create a note with a specific tag
    await page.getByLabel("Title").fill("Test Note")
    await page.getByLabel("Body").fill("Some content")
    await page.getByLabel(/Tags/).fill("testtag")
    await page.getByRole("button", { name: "Create Note" }).click()

    // Verify initial note count
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    // Enter a tag in the filter input
    await page.getByPlaceholder("Filter by tag").fill("testtag")

    // Verify filtered note count
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    // Verify filtered note display
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toBeVisible()

    // Clear the filter
    await page.getByPlaceholder("Filter by tag").clear()

    // Verify all notes are displayed again
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
  })

  test("displays 'No notes match the filter' message", async ({ page }) => {
    // Clear existing notes
    await page.evaluate(() => localStorage.clear());
    await page.reload();
