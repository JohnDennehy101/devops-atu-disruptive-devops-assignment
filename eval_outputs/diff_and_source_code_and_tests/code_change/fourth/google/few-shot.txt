import { test, expect } from "@playwright/test"

test.describe("Note Taking App", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("updates the number of notes displayed", async ({ page }) => {
    await page.getByLabel("Title").fill("Test Note 1")
    await page.getByLabel("Body").fill("Body 1")
    await page.getByLabel(/Tags/).fill("tag1")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByLabel("Title").fill("Test Note 2")
    await page.getByLabel("Body").fill("Body 2")
    await page.getByLabel(/Tags/).fill("tag2")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (2)")).toBeVisible()
  })

  test("can filter notes by tag", async ({ page }) => {
    await page.getByLabel("Title").fill("Note with tag1")
    await page.getByLabel("Body").fill("Body 1")
    await page.getByLabel(/Tags/).fill("tag1")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByLabel("Title").fill("Note with tag2")
    await page.getByLabel("Body").fill("Body 2")
    await page.getByLabel(/Tags/).fill("tag2")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.fill("placeholder=\"Filter by tag\"", "tag1")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByText("Note with tag2")).toBeHidden()

    await page.fill("placeholder=\"Filter by tag\"", "tag2")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByText("Note with tag1")).toBeHidden()

    await page.fill("placeholder=\"Filter by tag\"", "")
    await expect(page.getByText("Your Notes (2)")).toBeVisible()
  })
})
