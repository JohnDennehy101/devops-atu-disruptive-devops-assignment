import { test, expect } from '@playwright/test';

test.describe('Note Taking App - Tag Filter', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('should display tag filter input and update note count', async ({ page }) => {
    await expect(page.getByPlaceholder('Filter by tag')).toBeVisible();
    await expect(page.getByText('Your Notes (0)')).toBeVisible();
  });

  test('should filter notes by tag', async ({ page }) => {
    await page.getByLabel('Title').fill('Note with tag1');
    await page.getByLabel('Body').fill('Body 1');
    await page.getByLabel(/Tags/).fill('tag1, important');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByLabel('Title').fill('Note with tag2');
    await page.getByLabel('Body').fill('Body 2');
    await page.getByLabel(/Tags/).fill('tag2, other');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await expect(page.getByText('Your Notes (2)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(2);

    await page.getByPlaceholder('Filter by tag').fill('tag1');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
    await expect(page.getByRole('button', { name: /Note #\d+/ })).toHaveCount(1);

    await page.getByPlaceholder('Filter by tag').fill('important');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByPlaceholder('Filter by tag').fill('nonexistent');
    await expect(page.getByText('Your Notes (0)')).toBeVisible();
    await expect(page.getByText('No notes match the filter')).toBeVisible();
  });

  test('should show "No notes yet" when no notes exist', async ({ page }) => {
    await expect(page.getByText('No notes yet')).toBeVisible();
  });

  test('should show "No notes match the filter" when filter excludes all notes', async ({ page }) => {
    await page.getByLabel('Title').fill('Test Note');
    await page.getByLabel('Body').fill('Test Body');
    await page.getByLabel(/Tags/).fill('demo');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByPlaceholder('Filter by tag').fill('different');
    await expect(page.getByText('Your Notes (0)')).toBeVisible();
    await expect(page.getByText('No notes match the filter')).toBeVisible();
  });

  test('filter should be case-insensitive', async ({ page }) => {
    await page.getByLabel('Title').fill('Case Test');
    await page.getByLabel('Body').fill('Body');
    await page.getByLabel(/Tags/).fill('IMPORTANT');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByPlaceholder('Filter by tag').fill('important');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByPlaceholder('Filter by tag').fill('IMPORTANT');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByPlaceholder('Filter by tag').fill('ImPoRtAnT');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
  });

  test('filter should match partial tags', async ({ page }) => {
    await page.getByLabel('Title').fill('Partial Test');
    await page.getByLabel('Body').fill('Body');
    await page.getByLabel(/Tags/).fill('javascript, typescript');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByPlaceholder('Filter by tag').fill('script');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByPlaceholder('Filter by tag').fill('type');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByPlaceholder('Filter by tag').fill('java');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
  });
