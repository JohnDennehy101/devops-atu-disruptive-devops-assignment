import { test, expect } from "@playwright/test"

test.describe("Note Taking App - Tag Filter", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("tag filter input is visible and functional", async ({ page }) => {
    await expect(page.getByPlaceholder("Filter by tag")).toBeVisible()
    await page.getByPlaceholder("Filter by tag").fill("test")
    await expect(page.getByPlaceholder("Filter by tag")).toHaveValue("test")
  })

  test("displays correct count when filtering by tag", async ({ page }) => {
    await page.getByLabel("Title").fill("Note with tag")
    await page.getByLabel("Body").fill("Body")
    await page.getByLabel(/Tags/).fill("important, test")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByLabel("Title").fill("Another note")
    await page.getByLabel("Body").fill("Body")
    await page.getByLabel(/Tags/).fill("other")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (2)")).toBeVisible()

    await page.getByPlaceholder("Filter by tag").fill("important")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
  })

  test("shows filtered notes only", async ({ page }) => {
    await page.getByLabel("Title").fill("Filtered note")
    await page.getByLabel("Body").fill("Body")
    await page.getByLabel(/Tags/).fill("alpha")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByLabel("Title").fill("Unfiltered note")
    await page.getByLabel("Body").fill("Body")
    await page.getByLabel(/Tags/).fill("beta")
    await page.getByRole("button", { name: "Create Note" }).click()

    const noteButtons = page.getByRole("button", { name: /Note #\d+/ })
    await expect(noteButtons).toHaveCount(2)

    await page.getByPlaceholder("Filter by tag").fill("alpha")
    await expect(noteButtons).toHaveCount(1)
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toContainText("Note #")
  })

  test("shows 'No notes match the filter' when filter yields no results", async ({ page }) => {
    await page.getByLabel("Title").fill("First note")
    await page.getByLabel("Body").fill("Body")
    await page.getByLabel(/Tags/).fill("tag1")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await page.getByPlaceholder("Filter by tag").fill("nonexistent")
    await expect(page.getByText("No notes match the filter")).toBeVisible()
  })

  test("shows 'No notes yet' when there are no saved notes", async ({ page }) => {
    await expect(page.getByText("No notes yet")).toBeVisible()
    await page.getByPlaceholder("Filter by tag").fill("any")
    await expect(page.getByText("No notes yet")).toBeVisible()
  })

  test("filter is case-insensitive", async ({ page }) => {
    await page.getByLabel("Title").fill("Case test")
    await page.getByLabel("Body").fill("Body")
    await page.getByLabel(/Tags/).fill("IMPORTANT")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByPlaceholder("Filter by tag").fill("important")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByPlaceholder("Filter by tag").fill("IMPORTANT")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByPlaceholder("Filter by tag").fill("ImpOrTaNt")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
  })
