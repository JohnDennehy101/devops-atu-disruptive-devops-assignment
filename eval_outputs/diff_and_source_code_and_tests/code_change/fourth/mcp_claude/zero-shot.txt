import { test, expect } from "@playwright/test"

test.describe("Tag Filter Feature", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("shows filter by tag input in sidebar", async ({ page }) => {
    await expect(page.getByPlaceholder("Filter by tag")).toBeVisible()
  })

  test("shows 'No notes yet' when there are no notes and no filter", async ({ page }) => {
    await expect(page.getByText("No notes yet")).toBeVisible()
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
  })

  test("filter input does not affect empty notes list", async ({ page }) => {
    await page.getByPlaceholder("Filter by tag").fill("something")
    await expect(page.getByText("No notes yet")).toBeVisible()
  })

  test("shows all notes when filter is empty", async ({ page }) => {
    await page.getByLabel("Title").fill("Note A")
    await page.getByLabel("Body").fill("Body A")
    await page.getByLabel(/Tags/).fill("work, urgent")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByLabel("Title").fill("Note B")
    await page.getByLabel("Body").fill("Body B")
    await page.getByLabel(/Tags/).fill("personal")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (2)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(2)
  })

  test("filters notes by matching tag", async ({ page }) => {
    await page.getByLabel("Title").fill("Work Note")
    await page.getByLabel("Body").fill("Work body")
    await page.getByLabel(/Tags/).fill("work, important")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByLabel("Title").fill("Personal Note")
    await page.getByLabel("Body").fill("Personal body")
    await page.getByLabel(/Tags/).fill("personal, hobby")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (2)")).toBeVisible()

    await page.getByPlaceholder("Filter by tag").fill("work")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(1)
  })

  test("shows 'No notes match the filter' when filter matches nothing", async ({ page }) => {
    await page.getByLabel("Title").fill("Test Note")
    await page.getByLabel("Body").fill("Test body")
    await page.getByLabel(/Tags/).fill("alpha")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByPlaceholder("Filter by tag").fill("nonexistent")
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByText("No notes match the filter")).toBeVisible()
  })

  test("clearing filter restores all notes", async ({ page }) => {
    await page.getByLabel("Title").fill("Note One")
    await page.getByLabel("Body").fill("Body one")
    await page.getByLabel(/Tags/).fill("tagA")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByLabel("Title").fill("Note Two")
    await page.getByLabel("Body").fill("Body two")
    await page.getByLabel(/Tags/).fill("tagB")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (2)")).toBeVisible()

    await page.getByPlaceholder("Filter by tag").fill("tagA")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByPlaceholder("Filter by tag").fill("")
    await expect(page.getByText("Your Notes (2)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(2)
  })

  test("filter is case-insensitive", async ({ page }) => {
    await page.getByLabel("Title").fill("Case Test")
    await page.getByLabel("Body").fill("Body")
    await page.getByLabel(/Tags/).fill("JavaScript")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByPlaceholder("Filter by tag").fill("javascript")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(1)

    await page.getByPlaceholder("Filter by tag").fill("JAVASCRIPT")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
  })

  test("filter matches partial tag names", async ({ page }) => {
    await page.getByLabel("Title").fill("Partial Test")
    await page.getByLabel("Body").fill("Body")
    await page.getByLabel(/Tags/).fill("development")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByPlaceholder("Filter by tag").fill("dev")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(1)
  })

  test("filter updates count correctly with multiple matching notes", async ({ page }) => {
    await page.getByLabel("Title").fill("Note 1")
    await page.getByLabel("Body").fill("Body 1")
    await page.getByLabel(/Tags/).fill("shared, unique1")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByLabel("Title").fill("Note 2")
    await page.getByLabel("Body").fill("Body 2")
    await page.getByLabel(/Tags/).fill("shared, unique2")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByLabel("Title").fill("Note 3")
    await page.getByLabel("Body").fill("Body 3")
    await page.getByLabel(/Tags/).fill("other")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (3)")).toBeVisible()

    await page.getByPlaceholder("Filter by tag").fill("shared")
    await expect(page.getByText("Your Notes (2)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(2)

    await page.getByPlaceholder("Filter by tag").fill("unique1")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(1)
  })

  test("filtered notes are still clickable to view details", async ({ page }) => {
    await page.getByLabel("Title").fill("Clickable Note")
    await page.getByLabel("Body").fill("Click body")
    await page.getByLabel(/Tags/).fill("clicktag")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByLabel("Title").fill("Other Note")
    await page.getByLabel("Body").fill("Other body")
    await page.getByLabel(/Tags/).fill("othertag")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByPlaceholder("Filter by tag").fill("clicktag")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByRole("button", { name: /Note #\d+/ }).click()
    await expect(page.getByRole("heading", { name: "Note Details" })).toBeVisible()
    await expect(page.getByText("Title: Clickable Note")).toBeVisible()
    await expect(page.getByText("Body: Click body")).toBeVisible()
  })
})
