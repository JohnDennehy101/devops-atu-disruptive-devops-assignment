import { test, expect } from "@playwright/test"

test.describe("Tag Filter Feature", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("displays the tag filter input in the sidebar", async ({ page }) => {
    await expect(
      page.getByPlaceholder("Filter by tag"),
    ).toBeVisible()
  })

  test("shows 'No notes yet' when there are no notes and no filter", async ({ page }) => {
    await expect(page.getByText("No notes yet")).toBeVisible()
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
  })

  test("filters notes by tag", async ({ page }) => {
    // Create first note with "javascript" tag
    await page.getByLabel("Title").fill("JS Note")
    await page.getByLabel("Body").fill("JavaScript content")
    await page.getByLabel(/Tags/).fill("javascript, react")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    // Create second note with "python" tag
    await page.getByLabel("Title").fill("Python Note")
    await page.getByLabel("Body").fill("Python content")
    await page.getByLabel(/Tags/).fill("python, django")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (2)")).toBeVisible()

    // Filter by "python" — should show only one note
    await page.getByPlaceholder("Filter by tag").fill("python")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(1)

    // Filter by "javascript" — should show only the other note
    await page.getByPlaceholder("Filter by tag").fill("javascript")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(1)
  })

  test("shows 'No notes match the filter' when filter has no results", async ({ page }) => {
    // Create a note first
    await page.getByLabel("Title").fill("Tagged Note")
    await page.getByLabel("Body").fill("Some body")
    await page.getByLabel(/Tags/).fill("existing-tag")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    // Apply a filter that matches nothing
    await page.getByPlaceholder("Filter by tag").fill("nonexistent")
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByText("No notes match the filter")).toBeVisible()
  })

  test("shows all notes when filter is cleared", async ({ page }) => {
    // Create two notes
    await page.getByLabel("Title").fill("Note One")
    await page.getByLabel("Body").fill("Body one")
    await page.getByLabel(/Tags/).fill("alpha")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByLabel("Title").fill("Note Two")
    await page.getByLabel("Body").fill("Body two")
    await page.getByLabel(/Tags/).fill("beta")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (2)")).toBeVisible()

    // Filter to show one
    await page.getByPlaceholder("Filter by tag").fill("alpha")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    // Clear filter — should show all again
    await page.getByPlaceholder("Filter by tag").fill("")
    await expect(page.getByText("Your Notes (2)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(2)
  })

  test("tag filter is case-insensitive", async ({ page }) => {
    await page.getByLabel("Title").fill("Case Test")
    await page.getByLabel("Body").fill("Body text")
    await page.getByLabel(/Tags/).fill("JavaScript")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    // Filter with lowercase should match uppercase tag
    await page.getByPlaceholder("Filter by tag").fill("javascript")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(1)

    // Filter with uppercase should also match
    await page.getByPlaceholder("Filter by tag").fill("JAVASCRIPT")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(1)
  })

  test("tag filter supports partial tag matching", async ({ page }) => {
    await page.getByLabel("Title").fill("Partial Match")
    await page.getByLabel("Body").fill("Body")
    await page.getByLabel(/Tags/).fill("javascript")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    // Partial match "java" should match "javascript"
    await page.getByPlaceholder("Filter by tag").fill("java")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByRole("button", { name: /Note #\d+/ })).toHaveCount(1)
  })

  test("note count updates dynamically as filter text changes", async ({ page }) => {
    // Create notes with different tags
    await page.getByLabel("Title").fill("React Note")
    await page.getByLabel("Body").fill("React body")
    await page.getByLabel(/Tags/).fill("react")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByLabel("Title").fill("Redux Note")
    await page.getByLabel("Body").fill("Redux body")
    await page.getByLabel(/Tags/).fill("redux")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByLabel("Title").fill("Vue Note")
    await page.getByLabel("Body").fill("Vue body")
    await page.getByLabel(/Tags/).fill("vue")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (3)")).toBeVisible()

    // "re" matches both "react" and "redux"
    await page.getByPlaceholder("Filter by tag").fill("re")
    await expect(page.getByText("Your Notes (2)")).toBeVisible()

    // "react" matches only one
    await page.getByPlaceholder("Filter by tag").fill("react")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    // "vue" matches only one
    await page.getByPlaceholder("Filter by tag").fill("vue")
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
  })
})
