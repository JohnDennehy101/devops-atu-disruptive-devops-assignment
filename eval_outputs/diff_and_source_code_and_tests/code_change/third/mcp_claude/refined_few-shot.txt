import { test, expect } from "@playwright/test"

test.describe("Note Taking App - Copy Note Body", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("copy button appears in note details after selecting a note", async ({ page }) => {
    // Create a note first
    await page.getByLabel("Title").fill("Copy test note")
    await page.getByLabel("Body").fill("Body text to copy")
    await page.getByLabel(/Tags/).fill("test")
    await page.getByRole("button", { name: "Create Note" }).click()

    // Select the note to view its details
    await page.getByRole("button", { name: /Note #\d+/ }).click()

    await expect(page.getByRole("heading", { name: "Note Details" })).toBeVisible()
    await expect(page.getByRole("button", { name: "Copy" })).toBeVisible()
  })

  test("copy button copies note body to clipboard", async ({ page, context }) => {
    // Grant clipboard permissions
    await context.grantPermissions(["clipboard-read", "clipboard-write"])

    // Create a note
    await page.getByLabel("Title").fill("Clipboard note")
    await page.getByLabel("Body").fill("This body should be copied")
    await page.getByLabel(/Tags/).fill("clipboard")
    await page.getByRole("button", { name: "Create Note" }).click()

    // Select the note
    await page.getByRole("button", { name: /Note #\d+/ }).click()
    await expect(page.getByText("Body: This body should be copied")).toBeVisible()

    // Click the Copy button
    await page.getByRole("button", { name: "Copy" }).click()

    // Verify clipboard contents
    const clipboardText = await page.evaluate(() => navigator.clipboard.readText())
    expect(clipboardText).toBe("This body should be copied")
  })

  test("copy button is not visible when no note is selected", async ({ page }) => {
    await expect(page.getByRole("button", { name: "Copy" })).not.toBeVisible()
  })

  test("copy button is not visible when editing a note", async ({ page }) => {
    // Create a note
    await page.getByLabel("Title").fill("Edit mode note")
    await page.getByLabel("Body").fill("Some body content")
    await page.getByRole("button", { name: "Create Note" }).click()

    // Select the note to view details
    await page.getByRole("button", { name: /Note #\d+/ }).click()
    await expect(page.getByRole("button", { name: "Copy" })).toBeVisible()

    // Switch to edit mode
    await page.getByRole("button", { name: "Edit" }).click()

    // Copy button should no longer be visible in edit mode
    await expect(page.getByRole("button", { name: "Copy" })).not.toBeVisible()
  })
})
