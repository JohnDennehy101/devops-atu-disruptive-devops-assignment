import { test, expect } from '@playwright/test';

test('copy note body functionality', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await page.getByRole('button', { name: 'Create Note' }).click();
  await page.getByLabel('Title').fill('Test Note');
  await page.getByLabel('Body').fill('This is a test body');
  await page.getByLabel(/Tags/).fill('test,copy');
  await page.getByRole('button', { name: 'Create Note' }).click();

  await page.getByRole('button', { name: /Note #\d+/ }).click();
  await page.getByRole('button', { name: 'Copy' }).click();

  await expect(page.evaluate(() => navigator.clipboard.readText())).resolves().toBe('This is a test body');
});
