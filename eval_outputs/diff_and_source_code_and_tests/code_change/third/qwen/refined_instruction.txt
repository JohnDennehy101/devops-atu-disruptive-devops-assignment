import { test, expect } from '@playwright/test';

test('Verify that the \'Copy\' button copies the note body to the clipboard', async ({ page }) => {
  // Step 1: Create a new note
  await page.getByLabel("Title").fill("Copy Test Note");
  await page.getByLabel("Body").fill("This is the body of the copy test note.");
  await page.getByLabel(/Tags/).fill("copy, test");
  await page.getByRole("button", { name: "Create Note" }).click();

  // Step 2: Open the note to view its details
  await page.getByRole("button", { name: /Note #\d+/ }).click();

  // Step 3: Click the 'Copy' button
  await page.getByRole("button", { name: "Copy" }).click();

  // Step 4: Verify that the note body is copied to the clipboard
  const clipboardText = await page.evaluate(() => navigator.clipboard.readText());
  await expect(clipboardText).toBe("This is the body of the copy test note.");
});
