import { test, expect } from '@playwright/test';

test.describe('Copy Note Button', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('should display the copy button when a note is selected', async ({ page }) => {
    await page.getByLabel('Title').fill('Test Note');
    await page.getByLabel('Body').fill('This is a test note body.');
    await page.getByLabel(/Tags/).fill('test');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByRole('button', { name: /Note #\d+/ }).click();

    await expect(page.getByRole('button', { name: 'Copy' })).toBeVisible();
  });

  test('should copy the note body to the clipboard when the copy button is clicked', async ({ page }) => {
    await page.getByLabel('Title').fill('Test Note');
    await page.getByLabel('Body').fill('This is a test note body.');
    await page.getByLabel(/Tags/).fill('test');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByRole('button', { name: /Note #\d+/ }).click();

    await page.getByRole('button', { name: 'Copy' }).click();

    const clipboardContent = await page.evaluate(() => navigator.clipboard.readText());
    await expect(clipboardContent).toBe('This is a test note body.');
  });
});
