import { test, expect } from '@playwright/test';

test.describe('Note Taking App - Copy note body', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('shows Copy button for a selected note and copies the note body to clipboard', async ({ page, context }) => {
    await context.grantPermissions(['clipboard-read', 'clipboard-write'], {
      origin: 'http://localhost:5175',
    });

    await page.getByRole('textbox', { name: 'Title' }).fill('Clipboard note');
    await page.getByRole('textbox', { name: 'Body' }).fill('This body should be copied');
    await page.getByRole('textbox', { name: 'Tags (comma-separated)' }).fill('copy, test');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    const noteButton = page.getByRole('button', { name: /Note #\d+/ });
    await expect(noteButton).toBeVisible();
    await noteButton.click();

    await expect(page.getByRole('heading', { name: 'Note Details' })).toBeVisible();
    await expect(page.getByText('Title: Clipboard note')).toBeVisible();
    await expect(page.getByText('Body: This body should be copied')).toBeVisible();

    const copyButton = page.getByRole('button', { name: 'Copy' });
    await expect(copyButton).toBeVisible();
    await copyButton.click();

    const clipboardText = await page.evaluate(async () => await navigator.clipboard.readText());
    expect(clipboardText).toBe('This body should be copied');
  });
});
