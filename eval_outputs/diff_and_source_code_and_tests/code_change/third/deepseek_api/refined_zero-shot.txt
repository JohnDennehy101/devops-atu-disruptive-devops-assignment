import { test, expect } from '@playwright/test';

test.describe('Note Taking App - Copy functionality', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('should have a Copy button when a note is selected', async ({ page }) => {
    await page.getByLabel('Title').fill('Note to copy');
    await page.getByLabel('Body').fill('This text should be copied');
    await page.getByLabel(/Tags/).fill('copy, test');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByRole('button', { name: /Note #\d+/ }).click();

    await expect(page.getByRole('heading', { name: 'Note Details' })).toBeVisible();
    await expect(page.getByRole('button', { name: 'Copy' })).toBeVisible();
  });

  test('should copy note body to clipboard when Copy button is clicked', async ({ page, context }) => {
    await page.getByLabel('Title').fill('Clipboard test');
    await page.getByLabel('Body').fill('This is the copied content');
    await page.getByLabel(/Tags/).fill('clipboard');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByRole('button', { name: /Note #\d+/ }).click();

    await expect(page.getByText('Body: This is the copied content')).toBeVisible();

    await page.getByRole('button', { name: 'Copy' }).click();

    const clipboardText = await page.evaluate(() => navigator.clipboard.readText());
    expect(clipboardText).toBe('This is the copied content');
  });

  test('Copy button should only appear when a note is selected and not in edit mode', async ({ page }) => {
    await page.getByLabel('Title').fill('Test note');
    await page.getByLabel('Body').fill('Body text');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await page.getByRole('button', { name: /Note #\d+/ }).click();

    await expect(page.getByRole('button', { name: 'Copy' })).toBeVisible();
    await expect(page.getByRole('button', { name: 'Edit' })).toBeVisible();

    await page.getByRole('button', { name: 'Edit' }).click();

    await expect(page.getByRole('button', { name: 'Copy' })).not.toBeVisible();
    await expect(page.getByRole('heading', { name: 'Edit Note' })).toBeVisible();
  });
});
