import { test, expect } from "@playwright/test"

test.describe("Note Taking App", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("has title and create form visible", async ({ page }) => {
    await expect(
      page.getByRole("heading", { name: "Note Taking App" }),
    ).toBeVisible()
    await expect(
      page.getByRole("heading", { name: /Create New Note/ }),
    ).toBeVisible()
    await expect(
      page.getByRole("button", { name: "Create Note" }),
    ).toBeVisible()
  })

  test("shows subtitle describing local storage", async ({ page }) => {
    await expect(
      page.getByText("Create and manage notes. Your notes are stored locally."),
    ).toBeVisible()
  })

  test("can create a note and see it in the sidebar", async ({ page }) => {
    await page.getByLabel("Title").fill("My first note")
    await page.getByLabel("Body").fill("Some body text")
    await page.getByLabel(/Tags/).fill("demo, e2e")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByRole("button", { name: /Note #\d+/ })).toBeVisible()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
  })

  test("can open a note and see its details", async ({ page }) => {
    await page.getByLabel("Title").fill("Detail test")
    await page.getByLabel("Body").fill("Content to verify")
    await page.getByLabel(/Tags/).fill("tag1")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByRole("button", { name: /Note #\d+/ }).click()

    await expect(
      page.getByRole("heading", { name: "Note Details" }),
    ).toBeVisible()
    await expect(page.getByText("Title: Detail test")).toBeVisible()
    await expect(page.getByText("Body: Content to verify")).toBeVisible()
  })

  test("can clear all notes from sidebar", async ({ page }) => {
    await page.getByLabel("Title").fill("To clear")
    await page.getByLabel("Body").fill("Body")
    await page.getByLabel(/Tags/).fill("x")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await page.getByRole("button", { name: "Clear all" }).click()
    await expect(page.getByText("No notes yet")).toBeVisible()
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
  })

  test("clear all button is not visible when there are no notes", async ({ page }) => {
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible()
  })

  test("clear all button appears after creating a note", async ({ page }) => {
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible()

    await page.getByLabel("Title").fill("Trigger clear btn")
    await page.getByLabel("Body").fill("Body text")
    await page.getByLabel(/Tags/).fill("test")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByRole("button", { name: "Clear all" })).toBeVisible()
  })

  test("clear all removes multiple notes", async ({ page }) => {
    for (let i = 1; i <= 3; i++) {
      await page.getByLabel("Title").fill(`Note ${i}`)
      await page.getByLabel("Body").fill(`Body ${i}`)
      await page.getByLabel(/Tags/).fill(`tag${i}`)
      await page.getByRole("button", { name: "Create Note" }).click()
    }

    await expect(page.getByText("Your Notes (3)")).toBeVisible()
    await page.getByRole("button", { name: "Clear all" }).click()

    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByText("No notes yet")).toBeVisible()
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible()
  })

  test("clear all resets form fields and editing state", async ({ page }) => {
    await page.getByLabel("Title").fill("Edit me")
    await page.getByLabel("Body").fill("Edit body")
    await page.getByLabel(/Tags/).fill("edit")
    await page.getByRole("button", { name: "Create Note" }).click()

    await page.getByRole("button", { name: /Note #\d+/ }).click()
    await expect(page.getByRole("heading", { name: "Note Details" })).toBeVisible()

    await page.getByRole("button", { name: "Clear all" }).click()

    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByRole("heading", { name: "Note Details" })).not.toBeVisible()
    await expect(page.getByRole("heading", { name: /Create New Note/ })).toBeVisible()
  })

  test("clear all clears localStorage noteIds", async ({ page }) => {
    await page.getByLabel("Title").fill("Persist test")
    await page.getByLabel("Body").fill("Body")
    await page.getByLabel(/Tags/).fill("persist")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByRole("button", { name: "Clear all" }).click()

    const noteIds = await page.evaluate(() => localStorage.getItem("noteIds"))
    expect(noteIds).toBe("[]")

    await page.reload()
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByText("No notes yet")).toBeVisible()
  })
})
