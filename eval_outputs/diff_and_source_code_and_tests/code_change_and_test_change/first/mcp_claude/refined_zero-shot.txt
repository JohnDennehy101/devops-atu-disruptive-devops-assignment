import { test, expect } from '@playwright/test';

test.describe("Note Taking App - Clear All and Subtitle", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175");
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test("displays subtitle text about local storage", async ({ page }) => {
    await expect(
      page.getByText("Create and manage notes. Your notes are stored locally."),
    ).toBeVisible();
  });

  test("clear all button is not visible when there are no notes", async ({ page }) => {
    await expect(page.getByText("No notes yet")).toBeVisible();
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible();
  });

  test("can clear all notes from sidebar", async ({ page }) => {
    await page.getByLabel("Title").fill("To clear");
    await page.getByLabel("Body").fill("Body");
    await page.getByLabel(/Tags/).fill("x");
    await page.getByRole("button", { name: "Create Note" }).click();
    await expect(page.getByText("Your Notes (1)")).toBeVisible();
    await page.getByRole("button", { name: "Clear all" }).click();
    await expect(page.getByText("No notes yet")).toBeVisible();
    await expect(page.getByText("Your Notes (0)")).toBeVisible();
  });

  test("clear all button appears only when notes exist", async ({ page }) => {
    // Initially no clear all button
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible();

    // Create a note
    await page.getByLabel("Title").fill("Test note");
    await page.getByLabel("Body").fill("Test body");
    await page.getByLabel(/Tags/).fill("tag");
    await page.getByRole("button", { name: "Create Note" }).click();
    await expect(page.getByText("Your Notes (1)")).toBeVisible();

    // Now clear all button should be visible
    await expect(page.getByRole("button", { name: "Clear all" })).toBeVisible();

    // Click clear all
    await page.getByRole("button", { name: "Clear all" }).click();

    // Button should disappear again
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible();
  });

  test("clear all removes multiple notes", async ({ page }) => {
    // Create first note
    await page.getByLabel("Title").fill("Note one");
    await page.getByLabel("Body").fill("Body one");
    await page.getByLabel(/Tags/).fill("a");
    await page.getByRole("button", { name: "Create Note" }).click();
    await expect(page.getByText("Your Notes (1)")).toBeVisible();

    // Create second note
    await page.getByLabel("Title").fill("Note two");
    await page.getByLabel("Body").fill("Body two");
    await page.getByLabel(/Tags/).fill("b");
    await page.getByRole("button", { name: "Create Note" }).click();
    await expect(page.getByText("Your Notes (2)")).toBeVisible();

    // Clear all
    await page.getByRole("button", { name: "Clear all" }).click();
    await expect(page.getByText("Your Notes (0)")).toBeVisible();
    await expect(page.getByText("No notes yet")).toBeVisible();
  });

  test("clear all clears localStorage noteIds", async ({ page }) => {
    // Create a note
    await page.getByLabel("Title").fill("Persist test");
    await page.getByLabel("Body").fill("Body");
    await page.getByLabel(/Tags/).fill("t");
    await page.getByRole("button", { name: "Create Note" }).click();
    await expect(page.getByText("Your Notes (1)")).toBeVisible();

    // Clear all
    await page.getByRole("button", { name: "Clear all" }).click();

    // Verify localStorage is cleared
    const noteIds = await page.evaluate(() => localStorage.getItem("noteIds"));
    expect(noteIds).toBe("[]");

    // Reload and verify notes are still gone
    await page.reload();
    await expect(page.getByText("Your Notes (0)")).toBeVisible();
    await expect(page.getByText("No notes yet")).toBeVisible();
  });
});
