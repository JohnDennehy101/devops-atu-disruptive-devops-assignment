import { test, expect } from '@playwright/test';

test('shows body character count and updates when typing', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await expect(page.locator('text=0 / 5000')).toBeVisible();
  await page.locator('label=Body').fill('Hello');
  await expect(page.locator('text=5 / 5000')).toBeVisible();
  await page.locator('label=Body').fill('Hello world');
  await expect(page.locator('text=11 / 5000')).toBeVisible();
});
