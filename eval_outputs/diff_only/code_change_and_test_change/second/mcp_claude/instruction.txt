import { test, expect } from "@playwright/test"

test.describe("Note Taking App", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("shows body character count with initial zero value", async ({ page }) => {
    await expect(page.getByText("0 / 5000")).toBeVisible()
  })

  test("shows body character count and updates when typing", async ({ page }) => {
    await expect(page.getByText("0 / 5000")).toBeVisible()
    await page.getByLabel("Body").fill("Hello")
    await expect(page.getByText("5 / 5000")).toBeVisible()
    await page.getByLabel("Body").fill("Hello world")
    await expect(page.getByText("11 / 5000")).toBeVisible()
  })

  test("resets body character count when body is cleared", async ({ page }) => {
    await page.getByLabel("Body").fill("Some text")
    await expect(page.getByText("9 / 5000")).toBeVisible()
    await page.getByLabel("Body").fill("")
    await expect(page.getByText("0 / 5000")).toBeVisible()
  })

  test("character count reflects long body text", async ({ page }) => {
    const longText = "a".repeat(500)
    await page.getByLabel("Body").fill(longText)
    await expect(page.getByText("500 / 5000")).toBeVisible()
  })

  test("character count persists correctly after creating a note", async ({ page }) => {
    await page.getByLabel("Title").fill("Count test")
    await page.getByLabel("Body").fill("Test body")
    await expect(page.getByText("9 / 5000")).toBeVisible()
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("0 / 5000")).toBeVisible()
  })
})
