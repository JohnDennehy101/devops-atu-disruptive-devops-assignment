import { test, expect } from '@playwright/test';

test('shows body character count and updates when typing', async ({ page }) => {
  await page.goto('/'); // Assuming the home page is the root path

  // Check if the initial character count is displayed correctly
  await expect(page.getByText('0 / 5000')).toBeVisible();

  // Type "Hello" into the text area
  await page.getByLabel('Body').fill('Hello');

  // Check if the character count updates correctly after typing "Hello"
  await expect(page.getByText('5 / 5000')).toBeVisible();

  // Type " world" into the text area
  await page.getByLabel('Body').fill('Hello world');

  // Check if the character count updates correctly after typing " world"
  await expect(page.getByText('11 / 5000')).toBeVisible();
});
