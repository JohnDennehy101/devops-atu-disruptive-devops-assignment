import { test, expect } from '@playwright/test';

test.describe('Body Character Count', () => {
  test('shows body character count and updates when typing', async ({ page }) => {
    // Navigate to the page
    await page.goto('/');

    // Assert initial character count is visible
    await expect(page.getByText('0 / 5000')).toBeVisible();

    // Type "Hello" into the "Body" field and assert updated character count
    await page.getByRole('textbox', { name: 'Body' }).fill('Hello');
    await expect(page.getByText('5 / 5000')).toBeVisible();

    // Type "Hello world" into the "Body" field and assert updated character count
    await page.getByRole('textbox', { name: 'Body' }).fill('Hello world');
    await expect(page.getByText('11 / 5000')).toBeVisible();
  });
});
