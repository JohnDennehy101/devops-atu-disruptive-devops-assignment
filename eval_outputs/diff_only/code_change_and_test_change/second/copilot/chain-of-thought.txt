import { test, expect } from "@playwright/test";

test.describe("Body Character Count", () => {
  test("shows body character count and updates when typing", async ({ page }) => {
    // Step 1: Navigate to the Home Page
    await page.goto("/");

    // Step 2: Verify Initial State
    await expect(page.getByText("0 / 5000")).toBeVisible();

    // Step 3: Type Text into the "Body" Input Field
    await page.getByLabel("Body").fill("Hello");
    await expect(page.getByText("5 / 5000")).toBeVisible();

    // Step 4: Update the Text in the "Body" Input Field
    await page.getByLabel("Body").fill("Hello world");
    await expect(page.getByText("11 / 5000")).toBeVisible();

    // Optional Step 5: Test Edge Cases
    // Clear the input field and verify the count resets
    await page.getByLabel("Body").fill("");
    await expect(page.getByText("0 / 5000")).toBeVisible();

    // Test behavior when input exceeds 5000 characters
    const longText = "a".repeat(5001);
    await page.getByLabel("Body").fill(longText);
    await expect(page.getByText("5001 / 5000")).toBeVisible(); // Assuming it shows the actual count even if it exceeds the limit
  });
});
