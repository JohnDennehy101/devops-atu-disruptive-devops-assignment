import { test, expect } from '@playwright/test';

test('filter by tag shows only matching notes', async ({ page }) => {
  await page.goto('http://localhost:5175');

  await page.getByText('Home').waitFor();

  await page.getByRole('textbox', { name: 'Title' }).fill('Work note');
  await page.getByRole('textbox', { name: 'Body' }).fill('Body one');
  await page.getByRole('textbox', { name: /Tags/ }).fill('work, urgent');
  await page.getByRole('button', { name: 'Create Note' }).click();

  await page.getByRole('textbox', { name: 'Title' }).fill('Personal note');
  await page.getByRole('textbox', { name: 'Body' }).fill('Body two');
  await page.getByRole('textbox', { name: /Tags/ }).fill('personal');
  await page.getByRole('button', { name: 'Create Note' }).click();

  await expect(page.getByText('Your Notes (2)')).toBeVisible();

  await page.getByRole('textbox', { name: 'Filter by tag' }).fill('work');
  await expect(page.getByText('Your Notes (1)')).toBeVisible();
  await expect(page.getByRole('button', { name: 'Note #1' })).toBeVisible();
  await expect(page.getByRole('button', { name: 'Note #2' })).not.toBeVisible();

  await page.getByRole('textbox', { name: 'Filter by tag' }).fill('nonexistent');
  await expect(page.getByText('No notes match the filter')).toBeVisible();
  await expect(page.getByText('Your Notes (0)')).toBeVisible();

  await page.getByRole('textbox', { name: 'Filter by tag' }).fill('');
  await expect(page.getByText('Your Notes (2)')).toBeVisible();
});
