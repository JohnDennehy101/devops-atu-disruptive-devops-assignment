import { test, expect } from '@playwright/test';

test('should display character count and update on textarea input', async ({ page }) => {
  await page.goto('http://localhost:5175');

  const textarea = page.locator('textarea');
  const characterCount = page.locator('.text-xs.text-gray-500.dark:text-gray-400.mt-1');

  await expect(characterCount).toHaveText('0 / 5000');

  await textarea.fill('Hello, world!');
  await expect(characterCount).toHaveText('13 / 5000');

  await textarea.fill('a'.repeat(5000));
  await expect(characterCount).toHaveText('5000 / 5000');
});
