import { test, expect } from '@playwright/test';

test('should display character count below form input', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Assuming the form input has a role of 'textbox'
  const formInput = await page.getByRole('textbox');
  await formInput.fill('Hello, world!');

  // Locate the character count paragraph
  const characterCount = await page.getByText(/\/ 5000/);

  // Assert that the character count is displayed correctly
  await expect(characterCount).toHaveText('13 / 5000');
});
