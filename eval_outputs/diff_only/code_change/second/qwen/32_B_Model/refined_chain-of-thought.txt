import { test, expect } from '@playwright/test';

test('character count updates correctly', async ({ page }) => {
  await page.goto('http://localhost:5175');
  const formInput = page.locator('input[name="formBody"]');
  await formInput.type('Hello, world!');
  const charCount = page.locator('.text-xs.text-gray-500.dark:text-gray-400.mt-1');
  await expect(charCount).toHaveText('13 / 5000');
  await formInput.type(' '.repeat(4887));
  await expect(charCount).toHaveText('5000 / 5000');
});
