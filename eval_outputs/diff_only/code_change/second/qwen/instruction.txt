import { test, expect } from '@playwright/test';

test('Home page character count display', async ({ page }) => {
    // Step 1: Navigate to the Home page
    await page.goto('/');

    // Step 2: Check if the character count display is present and displays "0 / 5000"
    const charCountDisplay = await page.getByRole('paragraph', { name: '0 / 5000' });
    await expect(charCountDisplay).toBeVisible();
    await expect(charCountDisplay).toHaveText('0 / 5000');
});
