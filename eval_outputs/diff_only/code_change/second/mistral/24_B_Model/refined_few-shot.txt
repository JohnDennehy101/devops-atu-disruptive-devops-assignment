import { test, expect } from '@playwright/test';

test('form body length is displayed', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await page.fill('#form-body', 'This is a test');
  await expect(page.locator('.text-xs.text-gray-500.dark\\:text-gray-400.mt-1')).toContainText('14 / 5000');
});
