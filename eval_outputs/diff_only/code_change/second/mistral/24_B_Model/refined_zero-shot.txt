import { test, expect } from '@playwright/test';

test.describe('Home Page', () => {
  test('should display the character count', async ({ page }) => {
    await page.goto('http://localhost:5175');
    const characterCount = await page.locator('p.text-xs.text-gray-500.dark\\:text-gray-400.mt-1').textContent();
    expect(characterCount).toContain('/ 5000');
  });
});

test.describe('About Page', () => {
  test('should display the character count', async ({ page }) => {
    await page.goto('http://localhost:5175/about');
    const characterCount = await page.locator('p.text-xs.text-gray-500.dark\\:text-gray-400.mt-1').textContent();
    expect(characterCount).toContain('/ 5000');
  });
});
