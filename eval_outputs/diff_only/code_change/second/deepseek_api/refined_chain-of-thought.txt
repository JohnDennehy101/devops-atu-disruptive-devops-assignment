import { test, expect } from '@playwright/test';

test('character counter displays correctly on Home page', async ({ page }) => {
  await page.goto('http://localhost:5175');

  const textarea = page.locator('textarea[name="formBody"]');
  await expect(textarea).toBeVisible();

  const counter = page.locator('p.text-xs.text-gray-500.dark\\:text-gray-400.mt-1');
  await expect(counter).toBeVisible();
  await expect(counter).toHaveText('0 / 5000');

  await textarea.fill('Hello');
  await expect(counter).toHaveText('5 / 5000');

  const longText = 'A'.repeat(5000);
  await textarea.fill(longText);
  await expect(counter).toHaveText('5000 / 5000');
});
