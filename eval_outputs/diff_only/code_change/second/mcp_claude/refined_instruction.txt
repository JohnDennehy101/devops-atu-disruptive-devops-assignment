import { test, expect } from '@playwright/test';

test.describe('Body character counter', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
  });

  test('should display 0 / 5000 when body textarea is empty', async ({ page }) => {
    await expect(page.getByText('0 / 5000')).toBeVisible();
  });

  test('should update character count when typing in the body textarea', async ({ page }) => {
    const bodyTextbox = page.getByRole('textbox', { name: 'Body' });
    await bodyTextbox.fill('Hello world');
    await expect(page.getByText('11 / 5000')).toBeVisible();
  });

  test('should update character count for longer text', async ({ page }) => {
    const bodyTextbox = page.getByRole('textbox', { name: 'Body' });
    const longText = 'A'.repeat(100);
    await bodyTextbox.fill(longText);
    await expect(page.getByText('100 / 5000')).toBeVisible();
  });

  test('should reset character count to 0 when body is cleared', async ({ page }) => {
    const bodyTextbox = page.getByRole('textbox', { name: 'Body' });
    await bodyTextbox.fill('Some text');
    await expect(page.getByText('9 / 5000')).toBeVisible();
    await bodyTextbox.fill('');
    await expect(page.getByText('0 / 5000')).toBeVisible();
  });

  test('should show character counter below the body textarea', async ({ page }) => {
    const bodyTextbox = page.getByRole('textbox', { name: 'Body' });
    const counter = page.getByText('0 / 5000');
    await expect(counter).toBeVisible();

    // Verify the counter is rendered after the textarea (sibling relationship)
    const counterBox = await counter.boundingBox();
    const textareaBox = await bodyTextbox.boundingBox();
    expect(counterBox).not.toBeNull();
    expect(textareaBox).not.toBeNull();
    expect(counterBox!.y).toBeGreaterThan(textareaBox!.y);
  });
});
