import { test, expect } from '@playwright/test';

test('character counter displays 0 / 5000 by default', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await expect(page.getByText('0 / 5000')).toBeVisible();
});

test('character counter updates as user types in body', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await page.locator('#note-body').fill('Hello');
  await expect(page.getByText('5 / 5000')).toBeVisible();
});

test('character counter reflects longer text length', async ({ page }) => {
  await page.goto('http://localhost:5175');
  const text = 'A'.repeat(100);
  await page.locator('#note-body').fill(text);
  await expect(page.getByText('100 / 5000')).toBeVisible();
});

test('character counter resets after clearing body', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await page.locator('#note-body').fill('Some text');
  await expect(page.getByText('9 / 5000')).toBeVisible();
  await page.locator('#note-body').fill('');
  await expect(page.getByText('0 / 5000')).toBeVisible();
});
