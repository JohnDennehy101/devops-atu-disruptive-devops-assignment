import { test, expect } from "@playwright/test";

test.describe("Body character counter", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175");
  });

  test("displays initial character count as 0 / 5000 when body is empty", async ({ page }) => {
    const counter = page.locator("text=/ 5000").first();
    await expect(counter).toBeVisible();
    await expect(counter).toHaveText("0 / 5000");
  });

  test("updates character count as user types in the body field", async ({ page }) => {
    const bodyTextarea = page.getByRole("textbox", { name: "Body" });
    const counter = page.locator("text=/ 5000").first();

    await bodyTextarea.fill("Hello");
    await expect(counter).toHaveText("5 / 5000");

    await bodyTextarea.fill("Hello world");
    await expect(counter).toHaveText("11 / 5000");
  });

  test("updates character count for longer text input", async ({ page }) => {
    const bodyTextarea = page.getByRole("textbox", { name: "Body" });
    const counter = page.locator("text=/ 5000").first();

    const longText = "a".repeat(100);
    await bodyTextarea.fill(longText);
    await expect(counter).toHaveText("100 / 5000");
  });

  test("resets character count when body field is cleared", async ({ page }) => {
    const bodyTextarea = page.getByRole("textbox", { name: "Body" });
    const counter = page.locator("text=/ 5000").first();

    await bodyTextarea.fill("Some text");
    await expect(counter).toHaveText("9 / 5000");

    await bodyTextarea.fill("");
    await expect(counter).toHaveText("0 / 5000");
  });

  test("character counter is positioned below the body textarea", async ({ page }) => {
    const bodyTextarea = page.getByRole("textbox", { name: "Body" });
    const counter = page.locator("text=/ 5000").first();

    const textareaBox = await bodyTextarea.boundingBox();
    const counterBox = await counter.boundingBox();

    expect(textareaBox).not.toBeNull();
    expect(counterBox).not.toBeNull();
    expect(counterBox!.y).toBeGreaterThan(textareaBox!.y);
  });
});
