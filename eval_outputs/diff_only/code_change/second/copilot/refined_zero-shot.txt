import { test, expect } from '@playwright/test';

test('should display character count in form', async ({ page }) => {
  await page.goto('http://localhost:5175');
  const formBody = await page.locator('textarea'); // Assuming the form body is a textarea
  await formBody.fill('Test input');
  const charCount = await page.locator('p.text-xs.text-gray-500.dark\\:text-gray-400.mt-1');
  await expect(charCount).toHaveText('10 / 5000');
});
