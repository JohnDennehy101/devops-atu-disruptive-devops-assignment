import { test, expect } from "@playwright/test";

test.describe("Character counter for note body", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/");
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test("should display character counter with 0 / 5000 initially", async ({ page }) => {
    await expect(page.getByText("0 / 5000")).toBeVisible();
  });

  test("should update character count as user types in body", async ({ page }) => {
    const body = page.getByLabel("Body");
    await body.fill("Hello");
    await expect(page.getByText("5 / 5000")).toBeVisible();
  });

  test("should update character count for longer text", async ({ page }) => {
    const body = page.getByLabel("Body");
    const text = "a".repeat(100);
    await body.fill(text);
    await expect(page.getByText("100 / 5000")).toBeVisible();
  });
});
