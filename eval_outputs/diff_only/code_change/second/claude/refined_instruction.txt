import { test, expect } from '@playwright/test';

test('character counter is displayed below the form body textarea', async ({ page }) => {
  await page.goto('http://localhost:5175');

  const textarea = page.getByRole('textbox', { name: /body/i });
  await expect(textarea).toBeVisible();

  await expect(page.getByText('0 / 5000')).toBeVisible();

  await textarea.fill('Hello');
  await expect(page.getByText('5 / 5000')).toBeVisible();
});
