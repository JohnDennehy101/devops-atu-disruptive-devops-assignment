import { test, expect } from '@playwright/test';

test.describe('Character counter for form body', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/');
  });

  test('displays initial character count of 0 / 5000', async ({ page }) => {
    const counter = page.locator('text=0 / 5000');
    await expect(counter).toBeVisible();
  });

  test('updates character count as user types', async ({ page }) => {
    const textarea = page.locator('textarea');
    await textarea.fill('Hello World');

    const counter = page.locator('text=11 / 5000');
    await expect(counter).toBeVisible();
  });

  test('resets character count when textarea is cleared', async ({ page }) => {
    const textarea = page.locator('textarea');
    await textarea.fill('Some text');
    await expect(page.locator('text=9 / 5000')).toBeVisible();

    await textarea.fill('');
    await expect(page.locator('text=0 / 5000')).toBeVisible();
  });
});
