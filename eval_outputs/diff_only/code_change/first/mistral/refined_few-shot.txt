import { test, expect } from '@playwright/test';

test('should display the number of saved notes', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await expect(page.locator('h3')).toHaveText('Your Notes (0)');
});

test('should display a warning when trying to clear all notes', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await page.locator('button').first().click();
  await expect(page.locator('p')).toHaveText('Are you sure you want to clear all notes?');
});
