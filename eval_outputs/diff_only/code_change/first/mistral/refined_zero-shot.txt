import { test, expect } from '@playwright/test';

test('should display the correct number of notes and clear all button', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Check the initial state of the page
  await expect(page.locator('h3')).toHaveText('Your Notes (0)');
  await expect(page.locator('button')).toHaveText('Clear all');

  // Add some notes and check the updated state
  await page.locator('input[name="title"]').fill('Test Note');
  await page.locator('input[name="body"]').fill('This is a test note.');
  await page.locator('input[name="tags"]').fill('test,note');
  await page.locator('button[type="submit"]').click();

  await expect(page.locator('h3')).toHaveText('Your Notes (1)');
  await expect(page.locator('button')).toHaveText('Clear all');

  // Click the clear all button and check the final state
  await page.locator('button').click();
  await expect(page.locator('h3')).toHaveText('Your Notes (0)');
  await expect(page.locator('button')).toHaveText('Clear all');
});
