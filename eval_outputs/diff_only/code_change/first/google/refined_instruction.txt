+++ b/app/src/components/NoteForm.tsx
+import { useState } from'react';
+
+interface NoteFormProps {
+  title: string;
+  body: string;
+  tags: string;
+  isEditing: boolean;
+  onSave: (title: string, body: string, tags: string) => void;
+  onCancel: () => void;
+}
+
+const NoteForm: React.FC<NoteFormProps> = ({ title, body, tags, isEditing, onSave, onCancel }) => {
+  const [formTitle, setFormTitle] = useState(title);
+  const [formBody, setFormBody] = useState(body);
+  const [formTags, setFormTags] = useState(tags);
+
+  const handleSubmit = (e: React.FormEvent) => {
+    e.preventDefault();
+    onSave(formTitle, formBody, formTags);
+  };
+
+  return (
+    <form onSubmit={handleSubmit} className="space-y-4">
+      <div>
+        <label htmlFor="title" className="block text-gray-700 dark:text-gray-400">Title:</label>
+        <input
+          type="text"
+          id="title"
+          value={formTitle}
+          onChange={(e) => setFormTitle(e.target.value)}
+          className="mt-1 block w-full px-3 py-2 text-gray-900 dark:text-white rounded-md"
+        />
+      </div>
+      <div>
+        <label htmlFor="body" className="block text-gray-700 dark:text-gray-400">Body:</label>
+        <textarea
+          id="body"
+          value={formBody}
+          onChange={(e) => setFormBody(e.target.value)}
+          className="mt-1 block w-full px-3 py-2 text-gray-900 dark:text-white rounded-md"
+        />
+      </div>
+      <div>
+        <label htmlFor="tags" className="block text-gray-700 dark:text-gray-400">Tags:</label>
+        <input
+          type="text"
+          id="tags"
+          value={formTags}
+          onChange={(e) => setFormTags(e.target.value)}
+          className="mt-1 block w-full px-3 py-2 text-gray-900 dark:text-white rounded-md"
+        />
+      </div>
+      <button type="submit" className="bg-blue-500 hover:bg-blue-700 text-white font-bold py-2 px-4 rounded">
+        Save
+      </button>
+      <button type="button" onClick={onCancel} className="bg-gray-300 hover:bg-gray-400 text-gray-800 font-bold py-2 px-4 rounded">
+        Cancel
+      </button>
+    </form>
+  );
+};
