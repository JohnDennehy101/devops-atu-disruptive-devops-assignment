import { test, expect } from '@playwright/test';

test('clear all button clears all notes', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await page.fill('#note-title', 'Test Note');
  await page.fill('#note-body', 'Test Body');
  await page.click('#save-note');
  await expect(page.getByText('Test Note')).toBeVisible();
  await page.locator('text=Clear all').click();
  await expect(page.getByText('Your Notes (0)')).toBeVisible();
});
