import { test, expect } from '@playwright/test';

test('clear all notes button is functional', async ({ page }) => {
  await page.goto('http://localhost:5175');
  await page.fill('#note-title', 'Test Note');
  await page.fill('#note-body', 'This is a test note.');
  await page.click('#save-note-button');
  await expect(page.getByText('Your Notes (1)')).toBeVisible();
  await page.click('button:text("Clear all")');
  await expect(page.getByText('Your Notes (0)')).toBeVisible();
});
