import { test, expect } from '@playwright/test';

test.describe('Clear All Notes', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
  });

  test('should not show Clear all button when no notes exist', async ({ page }) => {
    await expect(page.getByRole('button', { name: 'Clear all' })).not.toBeVisible();
  });

  test('should show Clear all button when notes exist and remove all notes on click', async ({ page }) => {
    await page.getByPlaceholder('Note title').fill('Test Note 1');
    await page.getByPlaceholder('Note body').fill('Body 1');
    await page.getByRole('button', { name: /save/i }).click();

    await expect(page.getByText('Your Notes (1)')).toBeVisible();
    await expect(page.getByRole('button', { name: 'Clear all' })).toBeVisible();

    await page.getByPlaceholder('Note title').fill('Test Note 2');
    await page.getByPlaceholder('Note body').fill('Body 2');
    await page.getByRole('button', { name: /save/i }).click();

    await expect(page.getByText('Your Notes (2)')).toBeVisible();

    await page.getByRole('button', { name: 'Clear all' }).click();

    await expect(page.getByText('Your Notes (0)')).toBeVisible();
    await expect(page.getByRole('button', { name: 'Clear all' })).not.toBeVisible();
    await expect(page.getByText('Test Note 1')).not.toBeVisible();
    await expect(page.getByText('Test Note 2')).not.toBeVisible();
  });

  test('should display subtitle text', async ({ page }) => {
    await expect(page.getByText('Create and manage notes. Your notes are stored locally.')).toBeVisible();
  });

  test('should clear form fields after clearing all notes', async ({ page }) => {
    await page.getByPlaceholder('Note title').fill('My Note');
    await page.getByPlaceholder('Note body').fill('My Body');
    await page.getByRole('button', { name: /save/i }).click();

    await page.getByText('My Note').click();

    await page.getByRole('button', { name: 'Clear all' }).click();

    await expect(page.getByPlaceholder('Note title')).toHaveValue('');
    await expect(page.getByPlaceholder('Note body')).toHaveValue('');
  });
});
