import { test, expect } from '@playwright/test';

test('clear all notes button is functional', async ({ page }) => {
  await page.goto('http://localhost:5175');
  
  // Assuming there is a way to add notes for testing
  await page.evaluate(() => {
    localStorage.setItem('noteIds', JSON.stringify(['note1', 'note2']));
  });
  await page.reload();
  
  const clearAllButton = page.locator('text=Clear all');
  await expect(clearAllButton).toBeVisible();
  
  await clearAllButton.click();
  
  await expect(page.locator('text=Your Notes (0)')).toBeVisible();
  await expect(clearAllButton).toBeHidden();
  
  const noteIds = await page.evaluate(() => localStorage.getItem('noteIds'));
  expect(noteIds).toBe(JSON.stringify([]));
});
