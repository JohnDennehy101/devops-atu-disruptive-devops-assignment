import { test, expect } from '@playwright/test';

test('should display and clear notes', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Check if the paragraph with the text is present
  const paragraph = await page.locator('p.text-gray-500.dark\\:text-gray-400.text-sm.mb-6');
  await expect(paragraph).toHaveText('Create and manage notes. Your notes are stored locally.');

  // Check if the notes header is present
  const notesHeader = await page.locator('h3.text-lg.font-semibold.text-gray-900.dark\\:text-white');
  await expect(notesHeader).toHaveText(/Your Notes \(\d+\)/);

  // Check if the "Clear all" button is present when there are notes
  const clearAllButton = page.locator('button.text-sm.text-red-600.dark\\:text-red-400.hover\\:underline');
  if (await clearAllButton.isVisible()) {
    await clearAllButton.click();

    // Verify that notes are cleared
    await expect(notesHeader).toHaveText('Your Notes (0)');
  }
});
