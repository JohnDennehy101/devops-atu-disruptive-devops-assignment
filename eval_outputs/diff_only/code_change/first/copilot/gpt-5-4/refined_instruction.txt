import { test, expect } from '@playwright/test';

test('can clear all notes from the notes list', async ({ page }) => {
  await page.goto('http://localhost:5175');

  await expect(page.getByText('Create and manage notes. Your notes are stored locally.')).toBeVisible();

  const titleInput = page.getByRole('textbox', { name: /title/i });
  const bodyInput = page.getByRole('textbox', { name: /body/i });
  const tagsInput = page.getByRole('textbox', { name: /tags/i });
  const saveButton = page.getByRole('button', { name: /save/i });

  await titleInput.fill('First note');
  await bodyInput.fill('Body for first note');
  await tagsInput.fill('work');
  await saveButton.click();

  await titleInput.fill('Second note');
  await bodyInput.fill('Body for second note');
  await tagsInput.fill('personal');
  await saveButton.click();

  await expect(page.getByRole('heading', { name: /Your Notes \(2\)/i })).toBeVisible();

  const clearAllButton = page.getByRole('button', { name: /clear all/i });
  await expect(clearAllButton).toBeVisible();
  await clearAllButton.click();

  await expect(page.getByRole('heading', { name: /Your Notes \(0\)/i })).toBeVisible();
  await expect(page.getByRole('button', { name: /clear all/i })).toHaveCount(0);
  await expect(titleInput).toHaveValue('');
  await expect(bodyInput).toHaveValue('');
  await expect(tagsInput).toHaveValue('');
});
