import { test, expect } from '@playwright/test';

test('clear all removes saved notes and resets the notes list', async ({ page }) => {
  await page.goto('http://localhost:5175');

  await page.fill('input[placeholder*="title" i]', 'First Note');
  await page.fill('textarea[placeholder*="body" i]', 'This is the first note');
  await page.fill('input[placeholder*="tag" i]', 'work');
  await page.getByRole('button', { name: /save/i }).click();

  await page.fill('input[placeholder*="title" i]', 'Second Note');
  await page.fill('textarea[placeholder*="body" i]', 'This is the second note');
  await page.fill('input[placeholder*="tag" i]', 'personal');
  await page.getByRole('button', { name: /save/i }).click();

  await expect(page.getByText(/Your Notes \(2\)/)).toBeVisible();
  await expect(page.getByRole('button', { name: 'Clear all' })).toBeVisible();

  await page.getByRole('button', { name: 'Clear all' }).click();

  await expect(page.getByText(/Your Notes \(0\)/)).toBeVisible();
  await expect(page.getByRole('button', { name: 'Clear all' })).toHaveCount(0);
  await expect(page.getByText('First Note')).toHaveCount(0);
  await expect(page.getByText('Second Note')).toHaveCount(0);

  await page.reload();
  await expect(page.getByText(/Your Notes \(0\)/)).toBeVisible();
  await expect(page.getByText('First Note')).toHaveCount(0);
  await expect(page.getByText('Second Note')).toHaveCount(0);
});
