test('should call clearAllNotes when clear all button is clicked', async ({ page }) => {
  await page.goto('/');
  await page.locator('#note-1').click();
  await page.locator('.clear-all-button').click();
  await expect(page.getByText('Your Notes (0)')).toBeVisible();
});
