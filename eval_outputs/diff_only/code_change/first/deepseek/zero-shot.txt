import { test, expect } from '@playwright/test';

test.describe('Home Page', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:3000'); // Adjust the URL if your app runs on a different port or domain
  });

  test('should display "Clear all" button when there are saved notes', async ({ page }) => {
    // Assuming there are saved notes in localStorage
    await page.evaluate(() => {
      localStorage.setItem("noteIds", JSON.stringify([1, 2, 3]));
    });

    await page.reload();

    const clearAllButton = page.locator('button', { hasText: 'Clear all' });
    await expect(clearAllButton).toBeVisible();
  });

  test('should not display "Clear all" button when there are no saved notes', async ({ page }) => {
    await page.evaluate(() => {
      localStorage.setItem("noteIds", JSON.stringify([]));
    });

    await page.reload();

    const clearAllButton = page.locator('button', { hasText: 'Clear all' });
    await expect(clearAllButton).not.toBeVisible();
  });

  test('should clear all notes, reset the form, and update local storage when "Clear all" button is clicked', async ({ page }) => {
    // Set up initial state with saved notes
    await page.evaluate(() => {
      localStorage.setItem("noteIds", JSON.stringify([1, 2, 3]));
    });

    await page.reload();

    // Click the "Clear all" button
    const clearAllButton = page.locator('button', { hasText: 'Clear all' });
    await clearAllButton.click();

    // Verify that the button is no longer visible
    await expect(clearAllButton).not.toBeVisible();

    // Verify that the form is reset
    const formTitle = page.locator('input[name="title"]');
    const formBody = page.locator('textarea[name="body"]');
    const formTags = page.locator('input[name="tags"]');
    await expect(formTitle).toBeEmpty();
    await expect(formBody).toBeEmpty();
    await expect(formTags).toBeEmpty();

    // Verify that local storage is updated
    const updatedNoteIds = await page.evaluate(() => {
      return JSON.parse(localStorage.getItem("noteIds"));
    });
    expect(updatedNoteIds).toEqual([]);
  });
});
