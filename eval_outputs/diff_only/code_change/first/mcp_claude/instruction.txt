import { test, expect } from "@playwright/test";

test.describe("Clear All Notes functionality", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175");
  });

  test("should display subtitle text about local storage", async ({ page }) => {
    await expect(
      page.getByText("Create and manage notes. Your notes are stored locally.")
    ).toBeVisible();
  });

  test("should not show Clear all button when there are no notes", async ({
    page,
  }) => {
    await expect(page.getByRole("heading", { name: /Your Notes \(0\)/ })).toBeVisible();
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible();
  });

  test("should show Clear all button when at least one note exists", async ({
    page,
  }) => {
    await page.getByRole("textbox", { name: "Title" }).fill("Test Note");
    await page.getByRole("textbox", { name: "Body" }).fill("Test body");
    await page.getByRole("textbox", { name: "Tags (comma-separated)" }).fill("tag1");
    await page.getByRole("button", { name: "Create Note" }).click();

    await expect(page.getByRole("heading", { name: /Your Notes \(1\)/ })).toBeVisible();
    await expect(page.getByRole("button", { name: "Clear all" })).toBeVisible();
  });

  test("should clear all notes when Clear all button is clicked", async ({
    page,
  }) => {
    // Create first note
    await page.getByRole("textbox", { name: "Title" }).fill("First Note");
    await page.getByRole("textbox", { name: "Body" }).fill("First body");
    await page.getByRole("textbox", { name: "Tags (comma-separated)" }).fill("tag1");
    await page.getByRole("button", { name: "Create Note" }).click();

    // Create second note
    await page.getByRole("textbox", { name: "Title" }).fill("Second Note");
    await page.getByRole("textbox", { name: "Body" }).fill("Second body");
    await page.getByRole("textbox", { name: "Tags (comma-separated)" }).fill("tag2");
    await page.getByRole("button", { name: "Create Note" }).click();

    await expect(page.getByRole("heading", { name: /Your Notes \(2\)/ })).toBeVisible();

    // Click Clear all
    await page.getByRole("button", { name: "Clear all" }).click();

    // Verify all notes are removed
    await expect(page.getByRole("heading", { name: /Your Notes \(0\)/ })).toBeVisible();
    await expect(page.getByText("No notes yet")).toBeVisible();
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible();
  });

  test("should reset form fields and editing state after clearing all notes", async ({
    page,
  }) => {
    // Create a note
    await page.getByRole("textbox", { name: "Title" }).fill("Note to clear");
    await page.getByRole("textbox", { name: "Body" }).fill("Body to clear");
    await page.getByRole("textbox", { name: "Tags (comma-separated)" }).fill("clear");
    await page.getByRole("button", { name: "Create Note" }).click();

    // Click Clear all
    await page.getByRole("button", { name: "Clear all" }).click();

    // Verify form fields are empty
    await expect(page.getByRole("textbox", { name: "Title" })).toHaveValue("");
    await expect(page.getByRole("textbox", { name: "Body" })).toHaveValue("");
    await expect(page.getByRole("textbox", { name: "Tags (comma-separated)" })).toHaveValue("");
  });

  test("should clear localStorage when Clear all is clicked", async ({
    page,
  }) => {
    // Create a note
    await page.getByRole("textbox", { name: "Title" }).fill("Storage Note");
    await page.getByRole("textbox", { name: "Body" }).fill("Storage body");
    await page.getByRole("textbox", { name: "Tags (comma-separated)" }).fill("storage");
    await page.getByRole("button", { name: "Create Note" }).click();

    // Click Clear all
    await page.getByRole("button", { name: "Clear all" }).click();

    // Verify localStorage is cleared
    const noteIds = await page.evaluate(() => localStorage.getItem("noteIds"));
    expect(JSON.parse(noteIds!)).toEqual([]);
  });

  test("should persist cleared state after page reload", async ({ page }) => {
    // Create a note
    await page.getByRole("textbox", { name: "Title" }).fill("Persist Note");
    await page.getByRole("textbox", { name: "Body" }).fill("Persist body");
    await page.getByRole("textbox", { name: "Tags (comma-separated)" }).fill("persist");
    await page.getByRole("button", { name: "Create Note" }).click();

    await expect(page.getByRole("heading", { name: /Your Notes \(1\)/ })).toBeVisible();

    // Click Clear all
    await page.getByRole("button", { name: "Clear all" }).click();

    // Reload the page
    await page.reload();

    // Verify notes are still cleared after reload
    await expect(page.getByRole("heading", { name: /Your Notes \(0\)/ })).toBeVisible();
    await expect(page.getByText("No notes yet")).toBeVisible();
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible();
  });
});
