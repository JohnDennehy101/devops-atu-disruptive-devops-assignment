import { test, expect } from "@playwright/test";

test.describe("Clear All Notes functionality", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175");
    // Clear any existing state
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test("should display subtitle text", async ({ page }) => {
    await expect(
      page.getByText("Create and manage notes. Your notes are stored locally.")
    ).toBeVisible();
  });

  test("should not show Clear all button when there are no notes", async ({
    page,
  }) => {
    await expect(page.getByRole("heading", { name: /Your Notes \(0\)/ })).toBeVisible();
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible();
  });

  test("should show Clear all button when notes exist", async ({ page }) => {
    await page.getByRole("textbox", { name: "Title" }).fill("Test Note");
    await page.getByRole("textbox", { name: "Body" }).fill("Test body");
    await page.getByRole("textbox", { name: "Tags (comma-separated)" }).fill("tag1");
    await page.getByRole("button", { name: "Create Note" }).click();

    await expect(page.getByRole("heading", { name: /Your Notes \(1\)/ })).toBeVisible();
    await expect(page.getByRole("button", { name: "Clear all" })).toBeVisible();
  });

  test("should clear all notes when Clear all is clicked", async ({ page }) => {
    // Create two notes
    for (const title of ["Note A", "Note B"]) {
      await page.getByRole("textbox", { name: "Title" }).fill(title);
      await page.getByRole("textbox", { name: "Body" }).fill("Body for " + title);
      await page.getByRole("textbox", { name: "Tags (comma-separated)" }).fill("test");
      await page.getByRole("button", { name: "Create Note" }).click();
    }

    await expect(page.getByRole("heading", { name: /Your Notes \(2\)/ })).toBeVisible();
    await expect(page.getByRole("button", { name: "Clear all" })).toBeVisible();

    // Click Clear all
    await page.getByRole("button", { name: "Clear all" }).click();

    // Verify all notes are removed
    await expect(page.getByRole("heading", { name: /Your Notes \(0\)/ })).toBeVisible();
    await expect(page.getByText("No notes yet")).toBeVisible();
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible();
  });

  test("should clear form fields and editing state when Clear all is clicked", async ({
    page,
  }) => {
    // Create a note
    await page.getByRole("textbox", { name: "Title" }).fill("Edit Me");
    await page.getByRole("textbox", { name: "Body" }).fill("Body text");
    await page.getByRole("textbox", { name: "Tags (comma-separated)" }).fill("t1");
    await page.getByRole("button", { name: "Create Note" }).click();

    // Select the note to populate the form
    await page.getByRole("button", { name: /Edit Me|Note #/ }).click();

    // Click Clear all
    await page.getByRole("button", { name: "Clear all" }).click();

    // Verify form fields are cleared
    await expect(page.getByRole("textbox", { name: "Title" })).toHaveValue("");
    await expect(page.getByRole("textbox", { name: "Body" })).toHaveValue("");
    await expect(page.getByRole("textbox", { name: "Tags (comma-separated)" })).toHaveValue("");
  });

  test("should clear localStorage when Clear all is clicked", async ({ page }) => {
    // Create a note
    await page.getByRole("textbox", { name: "Title" }).fill("Persist Test");
    await page.getByRole("textbox", { name: "Body" }).fill("Body");
    await page.getByRole("textbox", { name: "Tags (comma-separated)" }).fill("tag");
    await page.getByRole("button", { name: "Create Note" }).click();

    await expect(page.getByRole("heading", { name: /Your Notes \(1\)/ })).toBeVisible();

    // Click Clear all
    await page.getByRole("button", { name: "Clear all" }).click();

    // Verify localStorage is cleared
    const noteIds = await page.evaluate(() => JSON.parse(localStorage.getItem("noteIds") || "[]"));
    expect(noteIds).toEqual([]);

    // Reload and verify notes stay cleared
    await page.reload();
    await expect(page.getByRole("heading", { name: /Your Notes \(0\)/ })).toBeVisible();
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible();
  });

  test("should display Clear all button with correct styling", async ({ page }) => {
    // Create a note so the button appears
    await page.getByRole("textbox", { name: "Title" }).fill("Style Test");
    await page.getByRole("textbox", { name: "Body" }).fill("Body");
    await page.getByRole("textbox", { name: "Tags (comma-separated)" }).fill("tag");
    await page.getByRole("button", { name: "Create Note" }).click();

    const clearButton = page.getByRole("button", { name: "Clear all" });
    await expect(clearButton).toBeVisible();
    await expect(clearButton).toHaveClass(/text-red-600/);
    await expect(clearButton).toHaveClass(/text-sm/);
  });
});
