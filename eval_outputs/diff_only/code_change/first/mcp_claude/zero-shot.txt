import { test, expect } from "@playwright/test"

test.describe("Note Taking App - Clear All & Description", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/")
    await page.evaluate(() => localStorage.clear())
    await page.reload()
  })

  test("displays description text below the heading", async ({ page }) => {
    await expect(
      page.getByText("Create and manage notes. Your notes are stored locally."),
    ).toBeVisible()
  })

  test("does not show Clear all button when there are no notes", async ({ page }) => {
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible()
  })

  test("shows Clear all button after creating a note", async ({ page }) => {
    await page.getByLabel("Title").fill("Test note")
    await page.getByLabel("Body").fill("Body text")
    await page.getByLabel(/Tags/).fill("tag1")
    await page.getByRole("button", { name: "Create Note" }).click()

    await expect(page.getByText("Your Notes (1)")).toBeVisible()
    await expect(page.getByRole("button", { name: "Clear all" })).toBeVisible()
  })

  test("Clear all removes all notes and resets the UI", async ({ page }) => {
    // Create two notes
    await page.getByLabel("Title").fill("Note A")
    await page.getByLabel("Body").fill("Body A")
    await page.getByLabel(/Tags/).fill("a")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    await page.getByLabel("Title").fill("Note B")
    await page.getByLabel("Body").fill("Body B")
    await page.getByLabel(/Tags/).fill("b")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (2)")).toBeVisible()

    // Click Clear all
    await page.getByRole("button", { name: "Clear all" }).click()

    // Verify all notes are removed
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByText("No notes yet")).toBeVisible()
    await expect(page.getByRole("button", { name: "Clear all" })).not.toBeVisible()
    await expect(page.getByRole("button", { name: /Note #/ })).not.toBeVisible()
  })

  test("Clear all clears the form fields", async ({ page }) => {
    // Create a note and select it to populate form
    await page.getByLabel("Title").fill("Editable note")
    await page.getByLabel("Body").fill("Editable body")
    await page.getByLabel(/Tags/).fill("edit")
    await page.getByRole("button", { name: "Create Note" }).click()

    // Select the note to populate the form for editing
    await page.getByRole("button", { name: /Note #/ }).click()

    // Click Clear all
    await page.getByRole("button", { name: "Clear all" }).click()

    // Verify form fields are cleared
    await expect(page.getByLabel("Title")).toHaveValue("")
    await expect(page.getByLabel("Body")).toHaveValue("")
    await expect(page.getByLabel(/Tags/)).toHaveValue("")
  })

  test("Clear all clears localStorage noteIds", async ({ page }) => {
    // Create a note
    await page.getByLabel("Title").fill("Persist note")
    await page.getByLabel("Body").fill("Persist body")
    await page.getByLabel(/Tags/).fill("persist")
    await page.getByRole("button", { name: "Create Note" }).click()
    await expect(page.getByText("Your Notes (1)")).toBeVisible()

    // Click Clear all
    await page.getByRole("button", { name: "Clear all" }).click()

    // Verify localStorage is cleared
    const noteIds = await page.evaluate(() => JSON.parse(localStorage.getItem("noteIds") || "[]"))
    expect(noteIds).toEqual([])

    // Verify notes stay cleared after reload
    await page.reload()
    await expect(page.getByText("Your Notes (0)")).toBeVisible()
    await expect(page.getByText("No notes yet")).toBeVisible()
  })

  test("Your Notes heading and Clear all button are on the same row", async ({ page }) => {
    await page.getByLabel("Title").fill("Layout note")
    await page.getByLabel("Body").fill("Layout body")
    await page.getByLabel(/Tags/).fill("layout")
    await page.getByRole("button", { name: "Create Note" }).click()

    const heading = page.getByRole("heading", { name: /Your Notes/ })
    const clearButton = page.getByRole("button", { name: "Clear all" })

    // Both should be visible and vertically aligned (same row)
    await expect(heading).toBeVisible()
    await expect(clearButton).toBeVisible()

    const headingBox = await heading.boundingBox()
    const buttonBox = await clearButton.boundingBox()
    expect(headingBox).not.toBeNull()
    expect(buttonBox).not.toBeNull()

    // They should share the same vertical space (same row)
    expect(Math.abs(headingBox!.y - buttonBox!.y)).toBeLessThan(20)
  })
})
