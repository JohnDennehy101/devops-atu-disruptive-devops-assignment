import { test, expect } from '@playwright/test';

test.describe('Home page copy functionality', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
  });

  test('should copy note body when copy button is clicked', async ({ page, context }) => {
    await page.waitForSelector('[data-testid^="note-item-"]');
    const firstNote = page.locator('[data-testid^="note-item-"]').first();
    await firstNote.click();

    await page.waitForSelector('button:has-text("Copy")');
    
    const copyButton = page.locator('button:has-text("Copy")');
    await expect(copyButton).toBeVisible();

    await context.grantPermissions(['clipboard-read', 'clipboard-write']);

    const noteBody = await page.locator('[data-testid="note-body"]').textContent();
    
    await copyButton.click();

    const clipboardText = await page.evaluate(() => navigator.clipboard.readText());
    expect(clipboardText).toBe(noteBody);
  });

  test('copy button should be visible when a note is selected', async ({ page }) => {
    await page.waitForSelector('[data-testid^="note-item-"]');
    const firstNote = page.locator('[data-testid^="note-item-"]').first();
    await firstNote.click();

    await page.waitForSelector('button:has-text("Copy")');
    const copyButton = page.locator('button:has-text("Copy")');
    
    await expect(copyButton).toBeVisible();
    await expect(copyButton).toHaveClass(/text-blue-600/);
  });

  test('copy button should not be visible when no note is selected', async ({ page }) => {
    const copyButton = page.locator('button:has-text("Copy")');
    await expect(copyButton).not.toBeVisible();
  });
});
