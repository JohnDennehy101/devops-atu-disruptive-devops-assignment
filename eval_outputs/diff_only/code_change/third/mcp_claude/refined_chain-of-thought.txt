import { test, expect } from '@playwright/test';

test.describe('Copy Note Body Button', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
  });

  test('Copy button is visible when a note is selected', async ({ page }) => {
    // Create a note first
    await page.getByLabel('Title').fill('Copy Test Note');
    await page.getByLabel('Body').fill('Body text to copy');
    await page.getByLabel('Tags (comma-separated)').fill('test');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Wait for the note to appear in the sidebar and click it
    const noteButton = page.getByRole('button', { name: /Note #/ }).last();
    await noteButton.click();

    // Verify the Note Details section appears with the Copy button
    await expect(page.getByRole('heading', { name: 'Note Details' })).toBeVisible();
    await expect(page.getByRole('button', { name: 'Copy' })).toBeVisible();
  });

  test('Copy button is not visible when no note is selected', async ({ page }) => {
    await expect(page.getByRole('button', { name: 'Copy' })).not.toBeVisible();
  });

  test('Copy button copies the note body to clipboard', async ({ context, page }) => {
    // Grant clipboard permissions
    await context.grantPermissions(['clipboard-read', 'clipboard-write']);

    // Create a note with specific body text
    const bodyText = 'This is the body text that should be copied';
    await page.getByLabel('Title').fill('Clipboard Test');
    await page.getByLabel('Body').fill(bodyText);
    await page.getByLabel('Tags (comma-separated)').fill('clipboard');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Select the newly created note
    const noteButton = page.getByRole('button', { name: /Note #/ }).last();
    await noteButton.click();

    // Wait for Note Details to appear
    await expect(page.getByText(`Body: ${bodyText}`)).toBeVisible();

    // Click the Copy button
    await page.getByRole('button', { name: 'Copy' }).click();

    // Verify clipboard contents
    const clipboardText = await page.evaluate(() => navigator.clipboard.readText());
    expect(clipboardText).toBe(bodyText);
  });

  test('Copy button appears alongside Edit button in Note Details', async ({ page }) => {
    // Create a note
    await page.getByLabel('Title').fill('Layout Test Note');
    await page.getByLabel('Body').fill('Some body content');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Select the note
    const noteButton = page.getByRole('button', { name: /Note #/ }).last();
    await noteButton.click();

    // Verify both Edit and Copy buttons are visible
    await expect(page.getByRole('button', { name: 'Edit' })).toBeVisible();
    await expect(page.getByRole('button', { name: 'Copy' })).toBeVisible();
  });

  test('Copy button is hidden when in edit mode', async ({ page }) => {
    // Create a note
    await page.getByLabel('Title').fill('Edit Mode Test');
    await page.getByLabel('Body').fill('Body for edit mode test');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Select the note
    const noteButton = page.getByRole('button', { name: /Note #/ }).last();
    await noteButton.click();

    // Verify Copy is visible
    await expect(page.getByRole('button', { name: 'Copy' })).toBeVisible();

    // Click Edit to enter edit mode
    await page.getByRole('button', { name: 'Edit' }).click();

    // Note Details section (and Copy button) should be hidden in edit mode
    await expect(page.getByRole('button', { name: 'Copy' })).not.toBeVisible();
  });
});
