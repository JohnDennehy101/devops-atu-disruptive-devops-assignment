import { test, expect } from '@playwright/test';

test('should copy note body to clipboard', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Assuming there's a note with id 1 and its body is "Hello, World!"
  await page.locate('button[data-testid="note-1-copy-button"]').click();

  // Check if "Hello, World!" is copied to clipboard
  const text = await page.evaluate(() => navigator.clipboard.readText());
  expect(text).toBe('Hello, World!');
});
