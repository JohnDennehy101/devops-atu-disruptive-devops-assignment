import { test, expect } from '@playwright/test';

test('copy button copies note body to clipboard', async ({ page }) => {
  await page.goto('http://localhost:5175');
  
  // Assuming there's a way to select a note, this part might need adjustment based on actual UI
  await page.click('.note-item'); // Replace with actual selector to select a note
  
  await page.locator('button:has-text("Copy")').click();
  
  // Verify clipboard content
  const clipboardText = await page.evaluate(() => navigator.clipboard.readText());
  await expect(clipboardText).toBe('Expected note body text'); // Replace with the expected note body text
});
