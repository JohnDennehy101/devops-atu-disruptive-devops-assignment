import { test, expect } from '@playwright/test';

test('should copy note body when copy button is clicked', async ({ page }) => {
  // Assuming the app is already loaded and the selectedNote is set
  await page.evaluate(() => {
    window.selectedNote = { body: 'Sample note body' };
  });

  await page.getByRole('button', { name: 'Copy' }).click();

  const copiedText = await navigator.clipboard.readText();
  expect(copiedText).toBe('Sample note body');
});
