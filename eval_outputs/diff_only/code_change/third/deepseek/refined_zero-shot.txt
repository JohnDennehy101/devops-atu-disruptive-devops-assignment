import { test, expect } from '@playwright/test';

test('Copy button should be present on the Home page', async ({ page }) => {
  await page.goto('http://localhost:5175');
  const copyButton = page.locator('button', { hasText: 'Copy' });
  await expect(copyButton).toBeVisible();
});

test('Copy button should not be present on the Note page', async ({ page }) => {
  await page.goto('http://localhost:5175/note');
  const copyButton = page.locator('button', { hasText: 'Copy' });
  await expect(copyButton).not.toBeVisible();
});
