import { test, expect } from '@playwright/test';

test('Copy note body to clipboard', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Ensure the "Copy" button is visible
  await expect(page.locator('button', { hasText: 'Copy' })).toBeVisible();

  // Click the "Copy" button
  await page.locator('button', { hasText: 'Copy' }).click();

  // Verify that the note body has been copied to the clipboard
  // This step is more complex and would typically require server-side code to verify clipboard content
  // For demonstration purposes, we'll just check if the button text changes or if an alert appears
  // Note: This is a simplified check and may not be reliable in all environments
  const buttonText = await page.locator('button', { hasText: 'Copy' }).innerText();
  expect(buttonText).toBe('Copied');
});
