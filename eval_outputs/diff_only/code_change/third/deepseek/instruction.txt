import { test, expect } from '@playwright/test';

test('Copy button functionality', async ({ page }) => {
  // Navigate to the home page
  await page.goto('http://localhost:3000');

  // Select a note (assuming there's a way to select a note programmatically)
  await page.getByRole('button', { name: 'Copy' }).click();

  // Expect the note body to be copied to the clipboard
  const copiedText = await navigator.clipboard.readText();
  await expect(copiedText).toBe('selectedNote.body');
});
