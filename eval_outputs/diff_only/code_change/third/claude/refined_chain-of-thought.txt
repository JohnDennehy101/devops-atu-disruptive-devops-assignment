import { test, expect } from '@playwright/test';

test('Copy button copies the selected note body to clipboard', async ({ page, context }) => {
  await context.grantPermissions(['clipboard-read', 'clipboard-write']);

  await page.goto('http://localhost:5175');

  const noteTitle = 'Clipboard Test Note';
  const noteBody = 'This is the body text to be copied.';

  await page.fill('#note-title', noteTitle);
  await page.fill('#note-body', noteBody);
  await page.click('button:has-text("Create Note")');

  await page.waitForSelector('button:has-text("Note #")');
  await page.click('button:has-text("Note #")');

  await page.waitForSelector('text=Note Details');
  await expect(page.locator('text=Note Details')).toBeVisible();
  await expect(page.locator(`text=${noteBody}`)).toBeVisible();

  await page.click('button:has-text("Copy")');

  const clipboardText = await page.evaluate(() => navigator.clipboard.readText());
  expect(clipboardText).toBe(noteBody);
});
