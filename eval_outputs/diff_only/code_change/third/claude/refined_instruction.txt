import { test, expect } from '@playwright/test';

test('copy note body button copies text to clipboard', async ({ page, context }) => {
  await context.grantPermissions(['clipboard-read', 'clipboard-write']);

  await page.goto('http://localhost:5175');

  // Create a note first so there's something to copy
  await page.getByRole('button', { name: /new note/i }).click();
  const noteBody = 'Test note body for clipboard copy';
  await page.getByRole('textbox', { name: /body/i }).fill(noteBody);
  await page.getByRole('button', { name: /save/i }).click();

  // Click the Copy button
  await page.getByRole('button', { name: 'Copy' }).click();

  // Verify clipboard contents
  const clipboardText = await page.evaluate(() => navigator.clipboard.readText());
  expect(clipboardText).toBe(noteBody);
});

test('copy button is visible when a note is selected', async ({ page }) => {
  await page.goto('http://localhost:5175');

  await page.getByRole('button', { name: /new note/i }).click();
  await page.getByRole('textbox', { name: /body/i }).fill('Some content');
  await page.getByRole('button', { name: /save/i }).click();

  await expect(page.getByRole('button', { name: 'Copy' })).toBeVisible();
});
