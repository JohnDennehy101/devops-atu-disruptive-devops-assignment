import { test, expect } from '@playwright/test';
import { Page } from '@playwright/test';

async function navigateToHome(page: Page) {
  await page.goto('http://localhost:3000');
}

test('Copy button copies note body to clipboard', async ({ page }) => {
  await navigateToHome(page);

  // Find a note element and click it to select it
  const noteElement = await page.locator('text=Test Note');
  await noteElement.click();

  // Find the copy button and click it
  const copyButton = await page.locator('text=Copy');
  await copyButton.click();

  // Check if the note body is in the clipboard
  const clipboardText = await page.evaluate(() => navigator.clipboard.readText());
  expect(clipboardText).toBe('Test Note');
});

test('Clicking a note selects it', async ({ page }) => {
  await navigateToHome(page);

  // Find a note element and click it
  const noteElement = await page.locator('text=Test Note');
  await noteElement.click();

  // Check if the selected note is displayed
  const selectedNoteElement = await page.locator('text=Test Note');
  expect(selectedNoteElement).toBeVisible();
});
