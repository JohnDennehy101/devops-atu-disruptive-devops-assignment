import { test, expect } from '@playwright/test';

test('test home page', async ({ page }) => {
  await page.goto('http://localhost:5175/');

  // Verify initial text
  await expect(page.locator('h1')).toHaveText('Your Notes (0)');

  // Test case 1
  await page.locator('input').fill('test');
  await expect(page.locator('h1')).toHaveText('Your Notes (0)');

  // Test case 2
  await page.locator('input').fill('test');
  await expect(page.locator('h1')).toHaveText('Your Notes (1)');

  // Test case 3
  await page.locator('input').fill('Test');
  await expect(page.locator('h1')).toHaveText('Your Notes (0)');

  // Test case 4
  await page.locator('input').fill('TEST');
  await expect(page.locator('h1')).toHaveText('Your Notes (0)');

  // Test case 5
  await page.locator('input').fill('test');
  await expect(page.locator('h1')).toHaveText('Your Notes (1)');

  // Test case 6
  await page.locator('input').fill('test');
  await expect(page.locator('h1')).toHaveText('Your Notes (2)');

  // Test case 7
  await page.locator('input').fill('Test');
  await expect(page.locator('h1')).toHaveText('Your Notes (1)');

  // Test case 8
  await page.locator('input').fill('TEST');
  await expect(page.locator('h1')).toHaveText('Your Notes (0)');
});
