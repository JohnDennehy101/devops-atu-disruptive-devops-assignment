import { test, expect } from "@playwright/test"
import { HomePage } from "../pages/HomePage"

test("Home page - tag filter", async ({ page }) => {
  const homePage = new HomePage(page)
  await homePage.goto()

  // Verify initial text
  await expect(homePage.getTitle()).toHaveText("Your Notes (0)")

  // Type "test" in the filter input
  await homePage.typeInFilterInput("test")

  // Verify text changes to "Your Notes (0)"
  await expect(homePage.getTitle()).toHaveText("Your Notes (0)")

  // Add some notes with the tag "test"
  await homePage.addNotesWithTag("test")

  // Verify text changes to "Your Notes (1)"
  await expect(homePage.getTitle()).toHaveText("Your Notes (1)")

  // Type "wrong" in the filter input
  await homePage.typeInFilterInput("wrong")

  // Verify text changes to "Your Notes (0)"
  await expect(homePage.getTitle()).toHaveText("Your Notes (0)")

  // Type "test" in the filter input again
  await homePage.typeInFilterInput("test")

  // Verify text changes back to "Your Notes (1)"
  await expect(homePage.getTitle()).toHaveText("Your Notes (1)")
})
