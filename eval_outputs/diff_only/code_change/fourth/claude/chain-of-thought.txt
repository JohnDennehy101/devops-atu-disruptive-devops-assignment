import { test, expect } from "@playwright/test";

test.describe("Tag filter on Your Notes sidebar", () => {
  test.beforeEach(async ({ page }) => {
    // Assumes two notes are already saved with tags "work" and "personal"
    await page.goto("/");
  });

  test("displays all saved notes when filter is empty", async ({ page }) => {
    const filterInput = page.getByPlaceholder("Filter by tag");
    await expect(filterInput).toBeVisible();
    await expect(filterInput).toHaveValue("");

    const heading = page.getByText(/Your Notes \(\d+\)/);
    await expect(heading).toContainText("Your Notes (2)");
  });

  test("filters notes by matching tag", async ({ page }) => {
    const filterInput = page.getByPlaceholder("Filter by tag");
    await filterInput.fill("work");

    await expect(page.getByText(/Your Notes \(1\)/)).toBeVisible();
  });

  test("shows 'No notes match the filter' when no tags match", async ({ page }) => {
    const filterInput = page.getByPlaceholder("Filter by tag");
    await filterInput.fill("nonexistenttag");

    await expect(page.getByText("No notes match the filter")).toBeVisible();
    await expect(page.getByText(/Your Notes \(0\)/)).toBeVisible();
  });

  test("restores full list when filter is cleared", async ({ page }) => {
    const filterInput = page.getByPlaceholder("Filter by tag");
    await filterInput.fill("work");
    await expect(page.getByText(/Your Notes \(1\)/)).toBeVisible();

    await filterInput.clear();
    await expect(page.getByText(/Your Notes \(2\)/)).toBeVisible();
  });

  test("shows 'No notes yet' when there are no saved notes", async ({ page }) => {
    // Navigate as a user with no saved notes (or clear saved notes)
    await page.evaluate(() => localStorage.clear());
    await page.reload();

    await expect(page.getByText("No notes yet")).toBeVisible();
  });
});
