import { test, expect } from "@playwright/test";

test.describe("Tag filter on saved notes", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("http://localhost:5175");
  });

  test("filter by tag input is visible", async ({ page }) => {
    await expect(
      page.getByRole("textbox", { name: "Filter by tag" })
    ).toBeVisible();
  });

  test("shows all notes when filter is empty", async ({ page }) => {
    const heading = page.getByRole("heading", { level: 3 });
    await expect(heading).toContainText("Your Notes (2)");
    await expect(page.getByRole("button", { name: "Note #18" })).toBeVisible();
    await expect(page.getByRole("button", { name: "Note #19" })).toBeVisible();
  });

  test("filters notes by matching tag", async ({ page }) => {
    await page.getByRole("textbox", { name: "Filter by tag" }).fill("javascript");

    const heading = page.getByRole("heading", { level: 3 });
    await expect(heading).toContainText("Your Notes (1)");
    await expect(page.getByRole("button", { name: "Note #18" })).toBeVisible();
    await expect(page.getByRole("button", { name: "Note #19" })).not.toBeVisible();
  });

  test("filters notes by a different tag", async ({ page }) => {
    await page.getByRole("textbox", { name: "Filter by tag" }).fill("python");

    const heading = page.getByRole("heading", { level: 3 });
    await expect(heading).toContainText("Your Notes (1)");
    await expect(page.getByRole("button", { name: "Note #19" })).toBeVisible();
    await expect(page.getByRole("button", { name: "Note #18" })).not.toBeVisible();
  });

  test("shows 'No notes match the filter' when no tags match", async ({ page }) => {
    await page.getByRole("textbox", { name: "Filter by tag" }).fill("nonexistenttag");

    const heading = page.getByRole("heading", { level: 3 });
    await expect(heading).toContainText("Your Notes (0)");
    await expect(page.getByText("No notes match the filter")).toBeVisible();
  });

  test("restores all notes when filter is cleared", async ({ page }) => {
    const filterInput = page.getByRole("textbox", { name: "Filter by tag" });

    await filterInput.fill("javascript");
    await expect(page.getByRole("heading", { level: 3 })).toContainText("Your Notes (1)");

    await filterInput.fill("");
    await expect(page.getByRole("heading", { level: 3 })).toContainText("Your Notes (2)");
    await expect(page.getByRole("button", { name: "Note #18" })).toBeVisible();
    await expect(page.getByRole("button", { name: "Note #19" })).toBeVisible();
  });

  test("tag filter is case-insensitive", async ({ page }) => {
    await page.getByRole("textbox", { name: "Filter by tag" }).fill("JAVASCRIPT");

    const heading = page.getByRole("heading", { level: 3 });
    await expect(heading).toContainText("Your Notes (1)");
    await expect(page.getByRole("button", { name: "Note #18" })).toBeVisible();
  });

  test("tag filter supports partial tag matching", async ({ page }) => {
    await page.getByRole("textbox", { name: "Filter by tag" }).fill("dev");

    const heading = page.getByRole("heading", { level: 3 });
    await expect(heading).toContainText("Your Notes (1)");
    await expect(page.getByRole("button", { name: "Note #19" })).toBeVisible();
  });
});
