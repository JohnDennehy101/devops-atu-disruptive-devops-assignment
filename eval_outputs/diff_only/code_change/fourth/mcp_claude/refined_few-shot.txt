import { test, expect } from '@playwright/test';

test.describe('Tag filter functionality', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
    // Clear localStorage to start fresh
    await page.evaluate(() => localStorage.clear());
    await page.reload();
  });

  test('filter by tag input is visible', async ({ page }) => {
    const filterInput = page.getByPlaceholder('Filter by tag');
    await expect(filterInput).toBeVisible();
  });

  test('shows "No notes yet" when no notes exist', async ({ page }) => {
    await expect(page.getByText('No notes yet')).toBeVisible();
  });

  test('displays note count in header as Your Notes (0) initially', async ({ page }) => {
    await expect(page.getByText('Your Notes (0)')).toBeVisible();
  });

  test('filters notes by tag and shows correct count', async ({ page }) => {
    // Create a note with tag "work"
    await page.locator('#note-title').fill('Work Note');
    await page.locator('#note-body').fill('This is a work note');
    await page.locator('#note-tags').fill('work, important');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Wait for note to appear in the sidebar
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    // Create a second note with tag "personal"
    await page.locator('#note-title').fill('Personal Note');
    await page.locator('#note-body').fill('This is a personal note');
    await page.locator('#note-tags').fill('personal, health');
    await page.getByRole('button', { name: 'Create Note' }).click();

    // Wait for both notes to appear
    await expect(page.getByText('Your Notes (2)')).toBeVisible();

    // Filter by "work" tag
    await page.getByPlaceholder('Filter by tag').fill('work');

    // Should show only 1 note matching the tag
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    // Filter by "personal" tag
    await page.getByPlaceholder('Filter by tag').fill('personal');

    // Should show only 1 note matching the tag
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    // Filter by non-existent tag
    await page.getByPlaceholder('Filter by tag').fill('nonexistent');

    // Should show 0 notes and display "No notes match the filter"
    await expect(page.getByText('Your Notes (0)')).toBeVisible();
    await expect(page.getByText('No notes match the filter')).toBeVisible();

    // Clear filter to show all notes again
    await page.getByPlaceholder('Filter by tag').fill('');
    await expect(page.getByText('Your Notes (2)')).toBeVisible();
  });

  test('tag filter is case-insensitive', async ({ page }) => {
    // Create a note with mixed-case tag
    await page.locator('#note-title').fill('Case Test Note');
    await page.locator('#note-body').fill('Testing case sensitivity');
    await page.locator('#note-tags').fill('JavaScript, React');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    // Filter with different cases
    await page.getByPlaceholder('Filter by tag').fill('javascript');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByPlaceholder('Filter by tag').fill('JAVASCRIPT');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByPlaceholder('Filter by tag').fill('react');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
  });

  test('tag filter supports partial matching', async ({ page }) => {
    // Create a note with a tag
    await page.locator('#note-title').fill('Partial Match Note');
    await page.locator('#note-body').fill('Testing partial match');
    await page.locator('#note-tags').fill('programming');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    // Filter with partial tag text
    await page.getByPlaceholder('Filter by tag').fill('prog');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    await page.getByPlaceholder('Filter by tag').fill('gram');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
  });

  test('shows "No notes match the filter" when notes exist but none match filter', async ({ page }) => {
    // Create a note
    await page.locator('#note-title').fill('Test Note');
    await page.locator('#note-body').fill('Test body');
    await page.locator('#note-tags').fill('alpha');
    await page.getByRole('button', { name: 'Create Note' }).click();

    await expect(page.getByText('Your Notes (1)')).toBeVisible();

    // Apply a filter that doesn't match
    await page.getByPlaceholder('Filter by tag').fill('zzz');

    // Should show the "no match" message, not the "no notes yet" message
    await expect(page.getByText('No notes match the filter')).toBeVisible();
    await expect(page.getByText('No notes yet')).not.toBeVisible();
  });
});
