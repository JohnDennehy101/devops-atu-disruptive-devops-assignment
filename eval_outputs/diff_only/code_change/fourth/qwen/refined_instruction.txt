import { test, expect } from '@playwright/test';

test('filter notes by tag', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Check initial state
  const initialNotesCount = await page.getByRole('heading', { name: /Your Notes \(\d+\)/ }).textContent();
  expect(initialNotesCount).toContain('Your Notes (0)');

  // Enter a tag filter
  await page.getByPlaceholder('Filter by tag').fill('example');
  await page.waitForTimeout(500); // Wait for filtering to occur

  // Check filtered state
  const filteredNotesCount = await page.getByRole('heading', { name: /Your Notes \(\d+\)/ }).textContent();
  expect(filteredNotesCount).toContain('Your Notes (0)'); // Assuming no notes match the filter

  // Clear the filter
  await page.getByPlaceholder('Filter by tag').clear();
  await page.waitForTimeout(500); // Wait for filtering to occur

  // Check state after clearing filter
  const clearedNotesCount = await page.getByRole('heading', { name: /Your Notes \(\d+\)/ }).textContent();
  expect(clearedNotesCount).toContain('Your Notes (0)');
});
