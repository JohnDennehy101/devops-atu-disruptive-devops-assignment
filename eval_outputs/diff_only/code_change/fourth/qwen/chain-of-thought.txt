import { test, expect } from '@playwright/test';

test('filter notes by tag', async ({ page }) => {
  await page.goto('/home');

  // Step 2: Check initial state
  await expect(page.locator('h1')).toHaveText('Your Notes (0)');
  await expect(page.locator('p')).toHaveText('No notes yet');

  // Step 3: Enter a tag into the filter input
  await page.fill('input[placeholder="Filter by tag"]', 'exampleTag');

  // Step 4: Verify that the displayed count updates based on the filter
  await expect(page.locator('h1')).toHaveText('Your Notes (X)'); // Replace X with the expected count

  // Step 5: Ensure that the correct notes are displayed based on the filter
  await expect(page.locator('.note')).toHaveCount(X); // Replace X with the expected count
});
