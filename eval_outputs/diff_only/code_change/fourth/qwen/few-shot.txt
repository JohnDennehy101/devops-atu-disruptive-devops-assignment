test('filter notes by tag', async ({ page }) => {
  await page.goto('/home');
  await page.fill('#tag-filter', 'work');
  await expect(page.getByText('Your Notes (2)')).toBeVisible();
  await expect(page.getByText('Note 1')).toBeVisible();
  await expect(page.getByText('Note 2')).toBeVisible();
  await expect(page.getByText('No notes match the filter')).not.toBeVisible();
});
