import { test, expect } from '@playwright/test';

test('Home page note filtering', async ({ page }) => {
  // Step 1: Navigate to the Home page
  await page.goto('/home');

  // Step 2: Check that the initial count of notes is displayed correctly
  const initialNoteCount = await page.getByText(/Your Notes \(\d+\)/).textContent();
  const initialCount = parseInt(initialNoteCount.match(/\d+/)?.[0] || '0', 10);
  expect(initialCount).toBeGreaterThan(0); // Assuming there are notes initially

  // Step 3: Enter a tag in the filter input
  const filterInput = await page.getByPlaceholder('Filter by tag');
  await filterInput.fill('exampleTag');

  // Step 4: Verify that the note count updates based on the filter
  await page.waitForTimeout(1000); // Wait for the filter to apply
  const filteredNoteCount = await page.getByText(/Your Notes \(\d+\)/).textContent();
  const filteredCount = parseInt(filteredNoteCount.match(/\d+/)?.[0] || '0', 10);
  expect(filteredCount).toBeLessThanOrEqual(initialCount);

  // Step 5: Clear the filter input
  await filterInput.clear();

  // Step 6: Verify that the note count returns to the original count
  await page.waitForTimeout(1000); // Wait for the filter to clear
  const finalNoteCount = await page.getByText(/Your Notes \(\d+\)/).textContent();
  const finalCount = parseInt(finalNoteCount.match(/\d+/)?.[0] || '0', 10);
  expect(finalCount).toBe(initialCount);
});
