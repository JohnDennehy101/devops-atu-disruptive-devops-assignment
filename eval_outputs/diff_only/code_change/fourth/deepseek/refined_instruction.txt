import { test, expect } from '@playwright/test';

test('should display notes based on filter', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Check initial state with no notes
  await expect(page.getByText('No notes yet')).toBeVisible();

  // Add a note to trigger the displayedIds update
  await page.evaluate(() => {
    api.getNotes = jest.fn().mockResolvedValue([{ id: 1, tags: ['test'] }]);
  });
  await page.reload();

  // Check if notes are displayed after adding
  await expect(page.getByText('Your Notes (1)')).toBeVisible();

  // Filter by tag
  await page.getByPlaceholder('Filter by tag').fill('test');
  await expect(page.getByText('Your Notes (1)')).toBeVisible();

  // Check if no matches are displayed
  await page.getByPlaceholder('Filter by tag').fill('nonexistenttag');
  await expect(page.getByText('No notes match the filter')).toBeVisible();
});
