import { test, expect } from '@playwright/test';

test.describe('Tag Filter Functionality', () => {
  test.beforeEach(async ({ page }) => {
    // Navigate to the Home page before each test
    await page.goto('/home');
  });

  test('should display the filter input field', async ({ page }) => {
    // Verify the presence of the filter input field
    const filterInput = page.locator('input[placeholder="Filter by tag"]');
    await expect(filterInput).toBeVisible();
  });

  test('should update note count based on the filter', async ({ page }) => {
    // Verify initial note count
    const initialCount = await page.locator('text=Your Notes (').innerText();
    expect(initialCount).toContain('Your Notes (');

    // Enter a tag in the filter input field
    const filterInput = page.locator('input[placeholder="Filter by tag"]');
    await filterInput.fill('example-tag');

    // Verify the updated note count
    const filteredCount = await page.locator('text=Your Notes (').innerText();
    expect(filteredCount).not.toEqual(initialCount);
