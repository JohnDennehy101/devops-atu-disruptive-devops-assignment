import { test, expect } from '@playwright/test';

test('should filter notes by tag', async ({ page }) => {
  await page.goto('http://localhost:5175');

  // Check initial state with no filter
  const initialNoteCount = await page.locator('text=Your Notes (0)').count();
  expect(initialNoteCount).toBe(1);

  // Simulate API response for notes
  await page.evaluate(() => {
    window['api'] = {
      getNotes: () => Promise.resolve([
        { id: '1', tags: ['work', 'urgent'] },
        { id: '2', tags: ['personal'] },
        { id: '3', tags: ['work'] }
      ])
    };
  });

  // Trigger useEffect to fetch notes
  await page.reload();

  // Check notes count after fetching
  const fetchedNoteCount = await page.locator('text=Your Notes (3)').count();
  expect(fetchedNoteCount).toBe(1);

  // Apply tag filter
  await page.fill('input[placeholder="Filter by tag"]', 'work');

  // Check filtered notes count
  const filteredNoteCount = await page.locator('text=Your Notes (2)').count();
  expect(filteredNoteCount).toBe(1);

  // Clear filter
  await page.fill('input[placeholder="Filter by tag"]', '');

  // Check notes count after clearing filter
  const clearedFilterNoteCount = await page.locator('text=Your Notes (3)').count();
  expect(clearedFilterNoteCount).toBe(1);
});
