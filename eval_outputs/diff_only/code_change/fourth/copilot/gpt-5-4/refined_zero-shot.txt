import { test, expect } from '@playwright/test';

test('filters saved notes by tag on the home page', async ({ page }) => {
  await page.goto('http://localhost:5175');

  const saveButtons = page.getByRole('button', { name: /save/i });
  const saveCount = await saveButtons.count();

  if (saveCount >= 2) {
    await saveButtons.nth(0).click();
    await saveButtons.nth(1).click();
  } else if (saveCount === 1) {
    await saveButtons.nth(0).click();
  }

  const notesHeading = page.getByText(/Your Notes \(\d+\)/);
  await expect(notesHeading).toBeVisible();

  const filterInput = page.getByPlaceholder('Filter by tag');
  await expect(filterInput).toBeVisible();

  const noteLinks = page.locator('a[href^="/note/"]');
  const savedNoteCount = await noteLinks.count();

  if (savedNoteCount === 0) {
    await expect(page.getByText('No notes yet')).toBeVisible();
    return;
  }

  const savedTitles: string[] = [];
  for (let i = 0; i < savedNoteCount; i++) {
    const text = await noteLinks.nth(i).textContent();
    savedTitles.push((text || '').trim());
  }

  await filterInput.fill('zzzzunlikelytagmatch');
  await expect(page.getByText('No notes match the filter')).toBeVisible();
  await expect(page.getByText(/Your Notes \(0\)/)).toBeVisible();

  await filterInput.fill('');
  await expect(page.getByText(new RegExp(`Your Notes \\(${savedNoteCount}\\)`))).toBeVisible();

  for (const title of savedTitles) {
    await expect(page.getByText(title, { exact: true })).toBeVisible();
  }
});
