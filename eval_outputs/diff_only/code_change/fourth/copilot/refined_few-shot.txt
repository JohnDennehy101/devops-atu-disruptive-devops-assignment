import { test, expect } from '@playwright/test';

test('filter notes by tag', async ({ page }) => {
  await page.goto('http://localhost:5175');
  
  // Ensure the input field for tag filtering is present
  const tagFilterInput = page.locator('input[placeholder="Filter by tag"]');
  await expect(tagFilterInput).toBeVisible();
  
  // Initially, no filter is applied, so all notes should be displayed
  const initialNotesCount = await page.locator('.note-item').count();
  await expect(page.getByText(`Your Notes (${initialNotesCount})`)).toBeVisible();
  
  // Apply a tag filter
  await tagFilterInput.fill('exampleTag');
  
  // Check if the notes are filtered correctly
  const filteredNotesCount = await page.locator('.note-item').count();
  await expect(page.getByText(`Your Notes (${filteredNotesCount})`)).toBeVisible();
  
  // If no notes match the filter, the appropriate message should be displayed
  if (filteredNotesCount === 0) {
    await expect(page.getByText('No notes match the filter')).toBeVisible();
  }
});
