import { test, expect } from '@playwright/test';

test('Home page displays notes and allows filtering by tag', async ({ page }) => {
  await page.goto('http://localhost:5175');

  await page.getByText('Your Notes (0)');
  await page.getByText('No notes yet');

  await page.locator('input[type="text"]').fill('test');
  await page.getByText('No notes match the filter');

  await page.locator('input[type="text"]').fill('');
  await page.getByText('Your Notes (0)');
  await page.getByText('No notes yet');
});
