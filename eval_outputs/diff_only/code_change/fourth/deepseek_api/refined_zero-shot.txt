import { test, expect } from '@playwright/test';

test.describe('Home page tag filtering', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('http://localhost:5175');
  });

  test('should show tag filter input', async ({ page }) => {
    const filterInput = page.getByPlaceholder('Filter by tag');
    await expect(filterInput).toBeVisible();
    await expect(filterInput).toBeEmpty();
  });

  test('should filter notes by tag', async ({ page }) => {
    // Mock API response with notes containing tags
    await page.route('**/api/notes', async route => {
      const mockNotes = [
        { id: '1', title: 'Note 1', content: 'Content 1', tags: ['work', 'important'] },
        { id: '2', title: 'Note 2', content: 'Content 2', tags: ['personal'] },
        { id: '3', title: 'Note 3', content: 'Content 3', tags: ['work'] }
      ];
      await route.fulfill({ json: mockNotes });
    });

    // Simulate saving some notes
    await page.evaluate(() => {
      window.localStorage.setItem('savedNoteIds', JSON.stringify(['1', '2', '3']));
    });

    await page.reload();

    const filterInput = page.getByPlaceholder('Filter by tag');
    await filterInput.fill('work');
    
    // Should show only notes with 'work' tag
    await expect(page.getByText('Your Notes (2)')).toBeVisible();
    await expect(page.getByText('Note 1')).toBeVisible();
    await expect(page.getByText('Note 3')).toBeVisible();
    await expect(page.getByText('Note 2')).not.toBeVisible();
  });

  test('should show "No notes match the filter" when filter excludes all notes', async ({ page }) => {
    await page.route('**/api/notes', async route => {
      const mockNotes = [
        { id: '1', title: 'Note 1', content: 'Content 1', tags: ['work'] },
        { id: '2', title: 'Note 2', content: 'Content 2', tags: ['personal'] }
      ];
      await route.fulfill({ json: mockNotes });
    });

    await page.evaluate(() => {
      window.localStorage.setItem('savedNoteIds', JSON.stringify(['1', '2']));
    });

    await page.reload();

    const filterInput = page.getByPlaceholder('Filter by tag');
    await filterInput.fill('nonexistent');
    
    await expect(page.getByText('Your Notes (0)')).toBeVisible();
    await expect(page.getByText('No notes match the filter')).toBeVisible();
    await expect(page.getByText('Note 1')).not.toBeVisible();
    await expect(page.getByText('Note 2')).not.toBeVisible();
  });

  test('should show all notes when filter is cleared', async ({ page }) => {
    await page.route('**/api/notes', async route => {
      const mockNotes = [
        { id: '1', title: 'Note 1', content: 'Content 1', tags: ['work'] },
        { id: '2', title: 'Note 2', content: 'Content 2', tags: ['personal'] }
      ];
      await route.fulfill({ json: mockNotes });
    });

    await page.evaluate(() => {
      window.localStorage.setItem('savedNoteIds', JSON.stringify(['1', '2']));
    });

    await page.reload();

    const filterInput = page.getByPlaceholder('Filter by tag');
    await filterInput.fill('work');
    await expect(page.getByText('Your Notes (1)')).toBeVisible();
    
    await filterInput.clear();
    await expect(page.getByText('Your Notes (2)')).toBeVisible();
    await expect(page.getByText('Note 1')).toBeVisible();
    await expect(page.getByText('Note 2')).toBeVisible();
  });
